88     private boolean add(long x, long y) {
89

91       if (!region.contains(x, y)) return false;
92 
93       // The point is within this region and there is room for it.
94       if (ptCount < NUM_POINTS) {
95

97         Y[ptCount] = y;
98         ptCount++;
99 
100         return true;
101 
102         // This region is full, so subdivide the region into four
103         // quadrants and try adding the point to these new regions
104       } else {
105 
106         // Find the center of this region at (cx, cy)
107         long cx = (region.x1 + region.x2) / 2;
108         long cy = (region.y1 + region.y2) / 2;
109 
110         // Lazily subdivide each of the regions into four parts
111         // one by one as needed to save memory.
112

114         if (sw.add(x, y)) return true;
115

117         if (nw.add(x, y)) return true;

120         if (ne.add(x, y)) return true;

333     // check if the two rectangles do not intersect and negate the logic afterwards.
334     public boolean intersects(Rect r) {
335       return r != null && !(r.x2 < x1 || r.x1 > x2 || r.y1 > y2 || r.y2 < y1);
336     }

338     // Check if a point (x, y) is within this rectangle, this
339     // includes the boundary of the rectangle.

341       return (x1 <= x && x <= x2) && (y1 <= y && y <= y2);
342     }

344     // Check if another rectangle is strictly contained within this rectangle.
345     public boolean contains(Rect r) {
346       return r != null && contains(r.x1, r.y1) && contains(r.x2, r.y2);
