30   static long run(int capacity, int[] weights, int[] values) {
31 
32     power = 1.0;
33     final int N = weights.length;
34 
35     // Create initial population
36     Individual[] generation = new Individual[P + 1];
37     Individual[] nextGeneration = new Individual[P + 1];
38     for (int i = 1; i <= P; i++) generation[i] = new Individual(N);
39 
40     // Stores the ranges of individuals in the selection roulette
41     double[] lo = new double[P + 1];
42     double[] hi = new double[P + 1];
43 
44     long[] fitness = new long[P + 1];
45     long bestFitness = 0;
46 
47     for (int epoch = 1; epoch <= MAX_EPOCH; epoch++, power += POWER_INC) {
48 
49       // Compute the total fitness sum across all individuals in order
50       // to be able to normalize and assign importance percentages
51       double fitnessSum = 0;
52 
53       for (int i = 1; i <= P; i++) {
54         Individual in = generation[i];
55         fitness[i] = fitness(in, weights, values, capacity, N);
56         fitnessSum += fitness[i];
57         lo[i] = hi[i] = 0;
58       }
59 
60       // Track the fittest individual
61       long bestEpochFitness = 0;
62 
63       // Setup selection roulette
64       for (int i = 1; i <= P; i++) {
65 
66         double norm = fitness[i] / fitnessSum;
67 
68         lo[i] = hi[i - 1] = lo[i - 1] + norm;
69 
70         if (fitness[i] > bestEpochFitness) {
71           bestEpochFitness = fitness[i];
72           if (bestEpochFitness > bestFitness) bestFitness = bestEpochFitness;
73         }
74       }
75 
76       if (epoch % 50 == 0)

78 
79       // Selection process
80       for (int i = 1; i <= P; i++) {

102   static long fitness(Individual in, int[] weights, int[] values, int capacity, int n) {
103     long value = 0, weight = 0;
104     for (int i = 0; i < n; i++) {
105       if (in.bits.get(i)) {
106         value += values[i];
107         weight += weights[i];
108       }
109       if (weight > capacity) return 0;
110     }
111     return value;
112   }

118   static Individual selectIndividual(Individual[] generation, double[] lo, double[] hi) {
119 
120     double r = Math.random();
121 
122     // Binary search to find individual
123     int mid, l = 0, h = P - 1;
124     while (true) {
125       mid = (l + h) >>> 1;
126       if (lo[mid] <= r && r < hi[mid]) return generation[mid + 1];
127       if (r < lo[mid]) h = mid - 1;
128       else l = mid + 1;
129     }
130   }

132   static Individual crossover(Individual p1, Individual p2, int n) {
133     int splitPoint = RANDOM.nextInt(n);
134     BitSet newBitSet = new BitSet(n);
135     for (int i = 0; i < splitPoint; i++) if (p1.bits.get(i)) newBitSet.flip(i);
136     for (int i = splitPoint; i < n; i++) if (p2.bits.get(i)) newBitSet.flip(i);
137     return new Individual(newBitSet);
138   }

188   static class Knapsack_01 {

190     /**
191      * @param maxWeight - The maximum weight of the knapsack
192      * @param W - The weights of the items
193      * @param V - The values of the items
194      * @return The maximum achievable profit of selecting a subset of the elements such that the
195      *     capacity of the knapsack is not exceeded
196      */
197     public static int knapsack(int maxWeight, int[] W, int[] V) {
198 
199       if (W == null || V == null || W.length != V.length || maxWeight < 0) {
200         throw new IllegalArgumentException("Invalid input");
201       }
202 
203       final int N = W.length;
204 
205       // Initialize a table where individual rows represent items
206       // and columns represent the weight of the knapsack
207       int[][] DP = new int[N + 1][maxWeight + 1];
208 
209       for (int i = 1; i <= N; i++) {
210 
211         // Get the value and weight of the item
212         int w = W[i - 1], v = V[i - 1];
213 
214         for (int sz = 1; sz <= maxWeight; sz++) {
215 
216           // Consider not picking this element
217           DP[i][sz] = DP[i - 1][sz];
218 
219           // Consider including the current element and
220           // see if this would be more profitable
221           if (sz >= w && DP[i - 1][sz - w] + v > DP[i][sz]) {
222             DP[i][sz] = DP[i - 1][sz - w] + v;
223           }
224         }
225       }
226 
227       // Return the maximum profit
228       return DP[N][maxWeight];
229     }
230   }
