39     for (int i = 1; i <= POPULATION_SZ; i++) {
40       generation[i] = new Individual();
41     }
42

44     double[] lo = new double[POPULATION_SZ + 1];
45     double[] hi = new double[POPULATION_SZ + 1];
46 
47     for (int epoch = 1; epoch <= MAX_EPOCH; epoch++, power += POWER_INC) {
48 
49       // Compute the total fitness sum across all individuals in order
50       // to be able to normalize and assign importance percentages
51       double fitnessSum = 0;
52 
53       for (int i = 1; i <= POPULATION_SZ; i++) {
54         Individual in = generation[i];

56         lo[i] = hi[i] = 0;
57       }
58 
59       // Track the fittest individual
60       Individual fittest = null;
61       double maxFitness = 0;
62 
63       // Setup selection roulette
64       for (int i = 1; i <= POPULATION_SZ; i++) {
65 
66         Individual in = generation[i];
67         double norm = in.fitness / fitnessSum;
68 
69         lo[i] = hi[i - 1] = lo[i - 1] + norm;
70 
71         if (in.fitness > maxFitness) {
72           maxFitness = in.fitness;
73           fittest = in;
74         }
75       }
76 
77       if (epoch % 50 == 0) System.out.printf("Epoch: %d,       %s\n", epoch, fittest);
78 
79       // Selection process
80       for (int i = 1; i <= POPULATION_SZ; i++) {
81

83         Individual parent1 = selectIndividual(generation, lo, hi);
84         Individual parent2 = selectIndividual(generation, lo, hi);
85         Individual child = crossover(parent1, parent2);
86

88         // according to a predefined mutation rate
89         for (int j = 0; j < TL; j++) {

91             child = mutate(child, j);
92           }
93         }

102         }
103       }
104

106     }

153   static Individual selectIndividual(Individual[] generation, double[] lo, double[] hi) {
154 
155     double r = Math.random();
156 
157     // Binary search to find individual
