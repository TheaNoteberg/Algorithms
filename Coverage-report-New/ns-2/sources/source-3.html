173   static Individual selectIndividual(Individual[] generation, double[] lo, double[] hi) {
174 
175     double r = Math.random();
176 
177     // Binary search to find individual
178     int mid, l = 0, h = P - 1;
179     while (true) {
180       mid = (l + h) >>> 1;
181       if (lo[mid] <= r && r < hi[mid]) return generation[mid + 1];
182       if (r < lo[mid]) h = mid - 1;
183       else l = mid + 1;
184     }
185   }

188 
189     int[] newPath = new int[n];
190     int start = RANDOM.nextInt(n);
191     int end = RANDOM.nextInt(n);
192     int minimum = Math.min(start, end);
193     int maximum = Math.max(start, end);
194 
195     int[] missing = new int[n - ((maximum - minimum) + 1)];
196 
197     int j = 0;
198     for (int i = 0; i < n; i++) {
199       if (i >= minimum && i <= maximum) {
200         newPath[i] = p1.cities[i];
201       } else {
202         missing[j++] = p1.cities[i];
203       }
204     }
205 
206     Individual.shuffleArray(missing);
207 
208     j = 0;
209     for (int i = 0; i < n; i++) {
210       if (i < minimum || i > maximum) {
211         newPath[i] = missing[j++];
212       }
213     }
214     return new Individual(newPath);
215   }
