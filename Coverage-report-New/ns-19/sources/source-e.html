87   // The cost matrix should be a symmetric (i.e cost[i][j] = cost[j][i]) and have a cost of `null`

97   // Sets the cost matrix. If the number of nodes in the graph is odd, add an artificial
98   // node that connects to every other node with a cost of infinity. This will make it easy
99   // to find a perfect matching and remove in the artificial node in the end.
100   private void setCostMatrix(Double[][] inputMatrix) {
101     isOdd = (n % 2 == 0) ? false : true;
102     Double[][] newCostMatrix = null;
103 
104     if (isOdd) {
105       newCostMatrix = new Double[n + 1][n + 1];
106     } else {
107       newCostMatrix = new Double[n][n];
108     }
109 
110     for (int i = 0; i < n; ++i) {
111       for (int j = 0; j < n; ++j) {
112         newCostMatrix[i][j] = inputMatrix[i][j];
113       }
114     }

156   private void solve() {
157     if (solved) return;
158     MatchingCost[] dp = new MatchingCost[1 << n];
159     int[] history = new int[1 << n];
160 
161     MatchingCost matchingCost = f(FULL_STATE, dp, history);
162     minWeightCost = matchingCost.cost;
163 
164     reconstructMatching(history);
165     solved = true;
166   }

168   private MatchingCost f(int state, MatchingCost[] dp, int[] history) {
169     if (dp[state] != null) {
170       return dp[state];
171     }
172     if (state == 0) {
173       return new MatchingCost();
174     }
175     int p1, p2;
176     // Seek to find active bit position (p1)
177     for (p1 = 0; p1 < n; p1++) {
178       if ((state & (1 << p1)) > 0) {
179         break;
180       }
181     }
182 
183     int bestState = -1;
184     MatchingCost bestMatchingCost = MatchingCost.createInfiniteValueMatchingCost();
185 
186     for (p2 = p1 + 1; p2 < n; p2++) {
187       // Position `p2` is on. Try matching the pair (p1, p2) together.
188       if ((state & (1 << p2)) > 0) {
189         int reducedState = state ^ (1 << p1) ^ (1 << p2);
190 
191         MatchingCost matchCost = new MatchingCost(f(reducedState, dp, history));
192         matchCost.updateMatchingCost(cost[p1][p2]);
193 
194         if (matchCost.isBetterMatchingCost(bestMatchingCost)) {
195           bestMatchingCost = matchCost;
196           bestState = reducedState;
197         }
198       }
199     }
200 
201     history[state] = bestState;
202     return dp[state] = bestMatchingCost;
203   }

205   // Populates the `matching` array with a sorted deterministic matching sorted by lowest node
206   // index. For example, if the perfect matching consists of the pairs (3, 4), (1, 5), (0, 2).
207   // The matching is sorted such that the pairs appear in the ordering: (0, 2), (1, 5), (3, 4).
208   // Furthermore, it is guaranteed that for any pair (a, b) that a < b.
209   private void reconstructMatching(int[] history) {
210     // A map between pairs of nodes that were matched together.
211     int[] map = new int[n];
212     int[] leftNodes = new int[n / 2];
213 
214     int matchingSize = 0;
215 
216     // Reconstruct the matching of pairs of nodes working backwards through computed states.
217     for (int i = 0, state = FULL_STATE; state != 0; state = history[state]) {
218       // Isolate the pair used by xoring the state with the state used to generate it.
219       int pairUsed = state ^ history[state];
220 
221       int leftNode = getBitPosition(Integer.lowestOneBit(pairUsed));
222       int rightNode = getBitPosition(Integer.highestOneBit(pairUsed));
223 
224       leftNodes[i++] = leftNode;
225       map[leftNode] = rightNode;
226 
227       if (cost[leftNode][rightNode] != null) {
228         matchingSize++;
229       }
230     }
231 
232     // Sort the left nodes in ascending order.
233     java.util.Arrays.sort(leftNodes);
234 
235     matchingSize = matchingSize * 2;
236     matching = new int[matchingSize];
237 
238     for (int i = 0, j = 0; i < n / 2; i++) {
239       int leftNode = leftNodes[i];
240       int rightNode = map[leftNodes[i]];
241       // Ignore the artificial node when there is an odd number of nodes.
242       if (isOdd && (leftNode == artificialNodeId || rightNode == artificialNodeId)) {
243         continue;
244       }
245       // Only match edges which actually exist
246       if (cost[leftNode][rightNode] != null) {
247         matching[2 * j] = leftNode;
248         matching[2 * j + 1] = rightNode;
249         j++;
250       }
251     }
252   }

254   // Gets the zero base index position of the 1 bit in `k`. `k` must be a power of 2, so there is
255   // only ever 1 bit in the binary representation of k.
256   private int getBitPosition(int k) {
257     int count = -1;
258     while (k > 0) {
259       count++;
260       k >>= 1;
261     }
262     return count;
263   }
