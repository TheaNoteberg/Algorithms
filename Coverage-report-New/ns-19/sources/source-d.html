35   public WeightedMaximumCardinalityMatchingIterative(double[][] cost) {

76   private void solve() {
77     if (solved) return;
78 
79     // The DP state is encoded as a bitmask where the i'th bit is flipped on if the i'th node is
80     // included in the state. Encoding the state this way allows us to compactly represent selecting
81     // a subset of the nodes present in the matching. Furthermore, it allows using the '&' binary
82     // operator to compare states to see if they overlap and the '|' operator to combine states.
83     //
84     // dp[i] contains the optimal cost of the MWPM for the nodes captured in the binary
85     // representation of `i`. The dp table is always half empty because all states with an odd
86     // number of nodes do not have a MWPM.
87     Double[] dp = new Double[1 << n];
88 
89     // Memo table to save the history of the chosen states. This table is used to reconstruct the
90     // chosen pairs of nodes after the algorithm has executed.
91     int[] history = new int[1 << n];
92 
93     // All the states consisting of pairs of nodes are the building blocks of this algorithm.
94     // In every iteration, we try to add a pair of nodes to previous state to construct a larger
95     // matching.
96     final int numPairs = (n * (n - 1)) / 2;
97     int[] pairStates = new int[numPairs];
98     double[] pairCost = new double[numPairs];
99 
100     int k = 0;
101     for (int i = 0; i < n; i++) {
102       for (int j = i + 1; j < n; j++) {
103         int state = (1 << i) | (1 << j);
104         dp[state] = cost[i][j];
105         pairStates[k] = state;
106         pairCost[k] = cost[i][j];
107         k++;
108       }
109     }
110 
111     for (int state = 0b11; state < (1 << n); state++) { // O(2^n)
112       // Skip states with an odd number of bits (nodes). It's easier (and faster) to
113       // check dp[state] instead of calling `Integer.bitCount` for the bit count.
114       if (dp[state] == null) {
115         continue;
116       }
117       for (int i = 0; i < numPairs; i++) { // O(n^2)
118         int pair = pairStates[i];
119         // Ignore states which overlap
120         if ((state & pair) != 0) continue;
121 
122         int newState = state | pair;
123         double newCost = dp[state] + pairCost[i];
124         if (dp[newState] == null || newCost < dp[newState]) {
125           dp[newState] = newCost;
126           // Save the fact that we went from 'state' -> 'newState'. From this we will be able to
127           // reconstruct which pairs of nodes were taken by looking at 'state' xor 'newState' which
128           // should give us the binary representation (state) of the pair used.
129           history[newState] = state;
130         }
131       }
132     }
133 
134     reconstructMatching(history);
135 
136     minWeightCost = dp[END_STATE];
137     solved = true;
138   }

140   // Populates the `matching` array with a sorted deterministic matching sorted by lowest node
141   // index. For example, if the perfect matching consists of the pairs (3, 4), (1, 5), (0, 2).
142   // The matching is sorted such that the pairs appear in the ordering: (0, 2), (1, 5), (3, 4).
143   // Furthermore, it is guaranteed that for any pair (a, b) that a < b.
144   private void reconstructMatching(int[] history) {
145     // A map between pairs of nodes that were matched together.
146     int[] map = new int[n];
147     int[] leftNodes = new int[n / 2];
148 
149     // Reconstruct the matching of pairs of nodes working backwards through computed states.
150     for (int i = 0, state = END_STATE; state != 0; state = history[state]) {
151       // Isolate the pair used by xoring the state with the state used to generate it.
152       int pairUsed = state ^ history[state];
153 
154       int leftNode = getBitPosition(Integer.lowestOneBit(pairUsed));
155       int rightNode = getBitPosition(Integer.highestOneBit(pairUsed));
156 
157       leftNodes[i++] = leftNode;
158       map[leftNode] = rightNode;
159     }
160 
161     // Sort the left nodes in ascending order.
162     java.util.Arrays.sort(leftNodes);
163 
164     matching = new int[n];
165     for (int i = 0; i < n / 2; i++) {
166       matching[2 * i] = leftNodes[i];
167       int rightNode = map[leftNodes[i]];
168       matching[2 * i + 1] = rightNode;
169     }
170   }

172   // Gets the zero base index position of the 1 bit in `k`. `k` must be a power of 2, so there is
173   // only ever 1 bit in the binary representation of k.
174   private int getBitPosition(int k) {
175     int count = -1;
176     while (k > 0) {
177       count++;
178       k >>= 1;
179     }
180     return count;
181   }
