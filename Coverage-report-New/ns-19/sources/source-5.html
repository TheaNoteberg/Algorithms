16 public class KnapsackUnbounded {

18   /**
19    * @param maxWeight - The maximum weight of the knapsack
20    * @param W - The weights of the items
21    * @param V - The values of the items
22    * @return The maximum achievable profit of selecting a subset of the elements such that the
23    *     capacity of the knapsack is not exceeded
24    */
25   public static int unboundedKnapsack(int maxWeight, int[] W, int[] V) {
26 
27     if (W == null || V == null || W.length != V.length || maxWeight < 0)
28       throw new IllegalArgumentException("Invalid input");
29 
30     final int N = W.length;
31 
32     // Initialize a table where individual rows represent items
33     // and columns represent the weight of the knapsack
34     int[][] DP = new int[N + 1][maxWeight + 1];
35 
36     // Loop through items
37     for (int i = 1; i <= N; i++) {
38 
39       // Get the value and weight of the item
40       int w = W[i - 1], v = V[i - 1];
41 
42       // Consider all possible knapsack sizes
43       for (int sz = 1; sz <= maxWeight; sz++) {
44 
45         // Try including the current element
46         if (sz >= w) DP[i][sz] = DP[i][sz - w] + v;
47 
48         // Check if not selecting this item at all is more profitable
49         if (DP[i - 1][sz] > DP[i][sz]) DP[i][sz] = DP[i - 1][sz];
50       }
51     }
52 
53     // Return the best value achievable
54     return DP[N][maxWeight];
55   }

57   public static int unboundedKnapsackSpaceEfficient(int maxWeight, int[] W, int[] V) {
58 
59     if (W == null || V == null || W.length != V.length)
60       throw new IllegalArgumentException("Invalid input");
61 
62     final int N = W.length;
63 
64     // Initialize a table where we will only keep track of
65     // the best possible value for each knapsack weight
66     int[] DP = new int[maxWeight + 1];
67 
68     // Consider all possible knapsack sizes
69     for (int sz = 1; sz <= maxWeight; sz++) {
70 
71       // Loop through items
72       for (int i = 0; i < N; i++) {
73 
74         // First check that we can include this item (we can't include it if
75         // it's too heavy for our knapsack). Assumming it fits inside the
76         // knapsack check if including this element would be profitable.
77         if (sz - W[i] >= 0 && DP[sz - W[i]] + V[i] > DP[sz]) DP[sz] = DP[sz - W[i]] + V[i];
78       }
79     }
80 
81     // Return the best value achievable
82     return DP[maxWeight];
83   }
