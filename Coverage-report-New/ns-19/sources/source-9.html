20   public static int lis(int[] ar) {
21 
22     if (ar == null || ar.length == 0) return 0;
23     int n = ar.length, len = 0;
24 
25     // When starting, each individual element has a LIS
26     // of exactly one, so each index is initialized to 1
27     int[] dp = new int[n];
28     java.util.Arrays.fill(dp, 1);
29 
30     // Processing the array left to right update the value of dp[j] if two
31     // conditions hold 1) The value at i is less than that of the one at j
32     // and 2) updating the value of dp[j] to dp[i]+1 is better
33     for (int i = 0; i < n; i++) {
34       for (int j = i + 1; j < n; j++) {
35         if (ar[i] < ar[j] && dp[j] < dp[i] + 1) {
36           dp[j] = dp[i] + 1;
37         }
38       }
39       // Track the LIS
40       if (dp[i] > len) len = dp[i];
41     }
42 
43     return len;
