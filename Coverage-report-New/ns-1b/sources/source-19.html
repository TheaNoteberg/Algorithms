17   // Given three sides of a triangle use Heron's formula
18   // to compute the area of the triangle.
19   public static double triangleArea(double side1, double side2, double side3) {
20 
21     // Semi-perimeter
22     double s = (side1 + side2 + side3) / 2.0;
23 
24     // Square root makes this method slower than the others
25     return sqrt(s * (s - side1) * (s - side2) * (s - side3));

28   // Given three points a, b, c find the area of the triangle
29   public static double triangleArea(
30       double ax, double ay, double bx, double by, double cx, double cy) {
31 
32     // Let v1 = <bx-ax, by-ay> and v2 = <cx-ax, cy-ay> be vectors
33     // Take the determinant of v1 and v2 to find the area of the
34     // parallelogram of the two vectors.
35 
36     double v1_x = bx - ax;
37     double v1_y = by - ay;
38     double v2_x = cx - ax;
39     double v2_y = cy - ay;
40 
41     //                | v1_x v2_x |
42     // Determinant of | v1_y v2_y | is v1_x*v2_y - v2_x*v1_y
43     double determinant = v1_x * v2_y - v2_x * v1_y;
44 
45     // Divide by two because we computed the area of the parallelogram
46     return abs(determinant) / 2.0;
47   }
