39   // Finds the intersection point(s) of two line segments. Unlike regular line
40   // segments, segments which are points (x1 = x2 and y1 = y2) are allowed.
41   public static Pt[] lineSegmentLineSegmentIntersection(Pt p1, Pt p2, Pt p3, Pt p4) {
42 
43     // No intersection.
44     if (!segmentsIntersect(p1, p2, p3, p4)) return new Pt[] {};
45 
46     // Both segments are a single point.
47     if (p1.equals(p2) && p2.equals(p3) && p3.equals(p4)) return new Pt[] {p1};
48 
49     List<Pt> endpoints = getCommonEndpoints(p1, p2, p3, p4);
50     int n = endpoints.size();
51 
52     // One of the line segments is an intersecting single point.
53     // NOTE: checking only n == 1 is insufficient to return early
54     // because the solution might be a sub segment.
55     boolean singleton = p1.equals(p2) || p3.equals(p4);
56     if (n == 1 && singleton) return new Pt[] {endpoints.get(0)};
57 
58     // Segments are equal.
59     if (n == 2) return new Pt[] {endpoints.get(0), endpoints.get(1)};
60 
61     boolean collinearSegments = (orientation(p1, p2, p3) == 0) && (orientation(p1, p2, p4) == 0);
62 
63     // The intersection will be a sub-segment of the two
64     // segments since they overlap each other.
65     if (collinearSegments) {
66 
67       // Segment #2 is enclosed in segment #1
68       if (pointOnLine(p1, p2, p3) && pointOnLine(p1, p2, p4)) return new Pt[] {p3, p4};
69 
70       // Segment #1 is enclosed in segment #2
71       if (pointOnLine(p3, p4, p1) && pointOnLine(p3, p4, p2)) return new Pt[] {p1, p2};
72 
73       // The subsegment is part of segment #1 and part of segment #2.
74       // Find the middle points which correspond to this segment.
75       Pt midPoint1 = pointOnLine(p1, p2, p3) ? p3 : p4;
76       Pt midPoint2 = pointOnLine(p3, p4, p1) ? p1 : p2;
77 
78       // There is actually only one middle point!
79       if (midPoint1.equals(midPoint2)) return new Pt[] {midPoint1};
80 
81       return new Pt[] {midPoint1, midPoint2};
82     }
83 
84     /* Beyond this point there is a unique intersection point. */
85 
86     // Segment #1 is a vertical line.
87     if (abs(p1.x - p2.x) < EPS) {
88       double m = (p4.y - p3.y) / (p4.x - p3.x);
89       double b = p3.y - m * p3.x;
90       return new Pt[] {new Pt(p1.x, m * p1.x + b)};
91     }
92 
93     // Segment #2 is a vertical line.
94     if (abs(p3.x - p4.x) < EPS) {
95       double m = (p2.y - p1.y) / (p2.x - p1.x);
96       double b = p1.y - m * p1.x;
97       return new Pt[] {new Pt(p3.x, m * p3.x + b)};
98     }
99 
100     double m1 = (p2.y - p1.y) / (p2.x - p1.x);
101     double m2 = (p4.y - p3.y) / (p4.x - p3.x);
102     double b1 = p1.y - m1 * p1.x;
103     double b2 = p3.y - m2 * p3.x;
104     double x = (b2 - b1) / (m1 - m2);
105     double y = (m1 * b2 - m2 * b1) / (m1 - m2);
106 
107     return new Pt[] {new Pt(x, y)};
108   }

110   // Finds the orientation of point 'c' relative to the line segment (a, b)
111   // Returns  0 if all three points are collinear.
112   // Returns -1 if 'c' is clockwise to segment (a, b), i.e right of line formed by the segment.
113   // Returns +1 if 'c' is counter clockwise to segment (a, b), i.e left of line
114   // formed by the segment.
115   private static int orientation(Pt a, Pt b, Pt c) {
116     double value = (b.y - a.y) * (c.x - b.x) - (b.x - a.x) * (c.y - b.y);
117     if (abs(value) < EPS) return 0;
118     return (value > 0) ? -1 : +1;
119   }

121   // Tests whether point 'c' is on the line segment (a, b).
122   // Ensure first that point c is collinear to segment (a, b) and
123   // then check whether c is within the rectangle formed by (a, b)
124   private static boolean pointOnLine(Pt a, Pt b, Pt c) {
125     return orientation(a, b, c) == 0
126         && min(a.x, b.x) <= c.x
127         && c.x <= max(a.x, b.x)
128         && min(a.y, b.y) <= c.y
129         && c.y <= max(a.y, b.y);
130   }

132   // Determines whether two segments intersect.
133   private static boolean segmentsIntersect(Pt p1, Pt p2, Pt p3, Pt p4) {
134 
135     // Get the orientation of points p3 and p4 in relation
136     // to the line segment (p1, p2)
137     int o1 = orientation(p1, p2, p3);
138     int o2 = orientation(p1, p2, p4);
139     int o3 = orientation(p3, p4, p1);
140     int o4 = orientation(p3, p4, p2);
141 
142     // If the points p1, p2 are on opposite sides of the infinite
143     // line formed by (p3, p4) and conversly p3, p4 are on opposite
144     // sides of the infinite line formed by (p1, p2) then there is
145     // an intersection.
146     if (o1 != o2 && o3 != o4) return true;
147 
148     // Collinear special cases (perhaps these if checks can be simplified?)
149     if (o1 == 0 && pointOnLine(p1, p2, p3)) return true;
150     if (o2 == 0 && pointOnLine(p1, p2, p4)) return true;
151     if (o3 == 0 && pointOnLine(p3, p4, p1)) return true;
152     if (o4 == 0 && pointOnLine(p3, p4, p2)) return true;
153 
154     return false;
155   }
