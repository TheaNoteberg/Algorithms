23     int len = polygon.length;
24     if (len < 3) return 0;

26     double[][] dp = new double[len][len];
27     for (int i = 2; i < len; i++) {
28       for (int j = 0; j + i < len; j++) {
29         dp[j][j + i] = Integer.MAX_VALUE;
30         for (int k = j + 1; k < j + i; k++) {
31           dp[j][j + i] =
32               Math.min(
33                   dp[j][j + i],
34                   dp[j][k] + dp[k][j + i] + cost(polygon[j], polygon[j + i], polygon[k]));
35         }
36       }
37     }
38     return dp[0][len - 1];
