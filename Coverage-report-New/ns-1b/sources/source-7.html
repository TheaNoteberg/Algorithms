16   public static Stack<Point2D> createConvexHull(Point2D[] pts) {
17     int k1, k2, N = pts.length;

19     Arrays.sort(pts, new PointOrder());
20     Arrays.sort(pts, 1, N, new PolarOrder(pts[0]));
21     hull.push(pts[0]);
22     for (k1 = 1; k1 < N; k1++) {
23       if (!pts[0].equals(pts[k1])) {
24         break;
25       }
26     }
27     if (k1 == N) return null;
28     for (k2 = k1 + 1; k2 < N; k2++) {
29       if (collinear(pts[0], pts[k1], pts[k2]) != 0) {
30         break;
31       }
32     }
33     hull.push(pts[k2 - 1]);
34     for (int i = k2; i < N; i++) {
35       Point2D top = hull.pop();
36       while (collinear(hull.peek(), top, pts[i]) <= 0) {
37         top = hull.pop();
38       }
39       hull.push(top);
40       hull.push(pts[i]);
41     }
42     return hull;
43   }

46   private static class PolarOrder implements Comparator<Point2D> {

52 
53     @Override
54     public int compare(Point2D q1, Point2D q2) {
55       double dx1 = q1.getX() - pt.getX(), dy1 = q1.getY() - pt.getY();
56       double dx2 = q2.getX() - pt.getX(), dy2 = q2.getY() - pt.getY();
57       if (dy1 >= 0 && dy2 < 0) return -1;
58       else if (dy2 >= 0 && dy1 < 0) return +1;
59       else if (dy1 == 0 && dy2 == 0) {
60         if (dx1 >= 0 && dx2 < 0) return -1;
61         else if (dx2 >= 0 && dx1 < 0) return +1;
62         else return 0;
63       } else return -collinear(pt, q1, q2);
64     }
