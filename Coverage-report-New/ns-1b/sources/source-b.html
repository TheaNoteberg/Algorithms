57   // of the polygon (swapping the endpoints p1 and p2 of the line

59   public static Pt[] cut(Pt[] poly, Pt p1, Pt p2) {
60     int n = poly.length;
61     List<Pt> res = new ArrayList<>();
62     for (int i = 0, j = n - 1; i < n; j = i++) {
63       int d1 = orientation(p1.x, p1.y, p2.x, p2.y, poly[j].x, poly[j].y);
64       int d2 = orientation(p1.x, p1.y, p2.x, p2.y, poly[i].x, poly[i].y);
65       if (d1 >= 0) res.add(poly[j]);
66       if (d1 * d2 < 0)
67         res.add(intersect(p1.x, p1.y, p2.x, p2.y, poly[j].x, poly[j].y, poly[i].x, poly[i].y));
68     }
69     return res.toArray(new Pt[res.size()]);
70   }

72   private static Pt intersect(
73       double x1, double y1, double x2, double y2, double x3, double y3, double x4, double y4) {
74     double a1 = y2 - y1, a2 = y4 - y3, b1 = x1 - x2, b2 = x3 - x4;
75     double c1 = -(x1 * y2 - x2 * y1), c2 = -(x3 * y4 - x4 * y3);
76     double det = a1 * b2 - a2 * b1;
77     double x = -(c1 * b2 - c2 * b1) / det, y = -(a1 * c2 - a2 * c1) / det;
78     return new Pt(x, y);
79   }
