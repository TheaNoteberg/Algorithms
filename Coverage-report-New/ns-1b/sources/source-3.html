21   // function may be outside its domain of [-1, +1] which would return
22   // the value Double.NaN which we do not want.
23   private static double arccosSafe(double x) {
24     if (x >= +1.0) return 0;
25     if (x <= -1.0) return PI;
26     return acos(x);
27   }

29   public static double circleCircleIntersectionArea(Point2D c1, double r1, Point2D c2, double r2) {
30 
31     double r = r1, R = r2;
32     Point2D c = c1, C = c2;
33 
34     if (r1 > r2) {
35       r = r2;
36       R = r1;
37       c = c2;
38       C = c1;
39     }
40 
41     double dist = c1.distance(c2);
42     Point2D[] intersections = circleCircleIntersection(c1, r1, c2, r2);
43 
44     // Smaller circle is contained within larger circle
45     if (intersections == null) {
46       return PI * r * r;
47 
48       // Circles do not intersect
49     } else if (intersections.length == 0) {
50       return 0;
51 
52       // One intersection point
53     } else if (intersections.length == 1) {
54 
55       // Check if the smaller circle is inside larger circle
56       if (dist < R) {
57         return PI * r * r;
58       } else {
59         return 0;
60       }
61 
62       // Circles intersect at two distinct points
63     } else {
64 
65       // http://stackoverflow.com/questions/4247889/area-of-intersection-between-two-circles
66       double d = dist;
67       Double part1 = r * r * arccosSafe((d * d + r * r - R * R) / (2 * d * r));
68       Double part2 = R * R * arccosSafe((d * d + R * R - r * r) / (2 * d * R));
69       Double part3 = 0.5 * sqrt((-d + r + R) * (d + r - R) * (d - r + R) * (d + r + R));
70 
71       return part1 + part2 - part3;
72     }
73   }

75   // Finds the intersection points of two circles. If the smaller circle is contained
76   // within the larger one this method returns null. If there is no intersection point
77   // an empty array is returned. If there is a unique intersection point that point alone
78   // is returned, otherwise two distinct points are returned where the intersection is.
79   public static Point2D[] circleCircleIntersection(Point2D c1, double r1, Point2D c2, double r2) {
80 
81     // r is the smaller radius and R is bigger radius
82     double r, R;
83 
84     // c is the center of the small circle
85     // C is the center of the big circle
86     Point2D c, C;

88     // Determine which is the bigger/smaller circle
89     if (r1 < r2) {
90       r = r1;
91       R = r2;
92       c = c1;
93       C = c2;
94     } else {
95       r = r2;
96       R = r1;
97       c = c2;
98       C = c1;
99     }
100

102 
103     // There are an infinite number of solutions
104     if (dist < EPS && abs(r - R) < EPS) return null;
105 
106     // No intersection (small circle contained within big circle)
107     if (r + dist < R) return null;
108 
109     // No intersection (circles are disjoint)
110     if (r + R < dist) return new Point2D[] {};
111 
112     // Let (cx, cy) be the center of the small circle
113     // Let (Cx, Cy) be the center of the larger circle
114     double cx = c.getX();
115     double Cx = C.getX();
116     double cy = c.getY();
117     double Cy = C.getY();
118 
119     // Compute the vector from the big circle to the little circle
120     double vx = cx - Cx;
121     double vy = cy - Cy;
122 
123     // Scale the vector by R and offset the vector so that the head of the vector
124     // is positioned on the circumference of the big circle ready to be rotated
125     double x = (vx / dist) * R + Cx;
126     double y = (vy / dist) * R + Cy;
127     Point2D point = new Point2D.Double(x, y);
128 
129     // Unique intersection point on circumference of both circles
130     if (abs(r + R - dist) < EPS || abs(R - (r + dist)) < EPS) return new Point2D[] {point};
131 
132     // Find the angle via cos law
133     double angle = arccosSafe((r * r - dist * dist - R * R) / (-2.0 * dist * R));
134 
135     // Two unique intersection points
136     Point2D pt1 = rotatePoint(C, point, angle);
137     Point2D pt2 = rotatePoint(C, point, -angle);
138 
139     return new Point2D[] {pt1, pt2};
140   }

142   // Rotate point 'pt' a certain number of radians clockwise
143   // relative to some fixed point 'fp'. Note that the angle
144   // should be specified in radians, not degrees.
145   private static Point2D rotatePoint(Point2D fp, Point2D pt, double angle) {
146 
147     double fpx = fp.getX();
148     double fpy = fp.getY();
149     double ptx = pt.getX();
150     double pty = pt.getY();
151 
152     // Compute the vector <x, y> from the fixed point
153     // to the point of rotation.
154     double x = ptx - fpx;
155     double y = pty - fpy;
156 
157     // Apply the clockwise rotation matrix to the vector <x, y>
158     // |  cos(theta) sin(theta) ||x|   |  xcos(theta) + ysin(theta) |
159     // | -sin(theta) cos(theta) ||y| = | -xsin(theta) + ycos(theta) |
160     double xRotated = x * cos(angle) + y * sin(angle);
161     double yRotated = y * cos(angle) - x * sin(angle);
162 
163     // The rotation matrix rotated the vector about the origin, so we
164     // need to offset it by the point (fpx, fpy) to get the right answer
165     return new Point2D.Double(fpx + xRotated, fpy + yRotated);
166   }
