27   // Determine is four points (ax,ay,az), (bx,by,bz), (cx,cy,cz),
28   // (dx,dy,dz) all lie in the same plane in 3D space
29   public static boolean coplanar(
30       double ax,
31       double ay,
32       double az,
33       double bx,
34       double by,
35       double bz,
36       double cx,
37       double cy,
38       double cz,
39       double dx,
40       double dy,
41       double dz) {
42 
43     // The problem of testing if four points are coplanar can be converted
44     // into a problem a checking if two vectors are orthogonal which is something
45     // the dot product is excellent at (if the dot product of two vectors is zero
46     // then they are orthogonal!)
47     //
48     // First, consider the three vectors spanning outwards from say point 'a':
49     // v1 = b-a, v2 = c-a, and v3 = d-a. If we take the cross product of any
50     // of these vectors (say v1 and v2) then we can obtain a fourth vector (v4)
51     // normal to the plane in which v1 and v2 live, so if the point 'd' lies in
52     // the same plane then the dot product of v3 and v4 should be 0
53 
54     Vector v1 = new Vector(bx - ax, by - ay, bz - az);
55     Vector v2 = new Vector(cx - ax, cy - ay, cz - az);
56     Vector v3 = new Vector(dx - ax, dy - ay, dz - az);
57 
58     // Find the vector normal (perpendicular) to v1 and v2
59     Vector v4 = cross(v1, v2);
60 
61     // Check whether to dot product of v3 and v4 is zero
62     return abs(dot(v3, v4)) < EPS;
63   }
64 
65   // Cross product of two vectors
66   private static Vector cross(Vector v1, Vector v2) {
67     double v3x = v1.y * v2.z - v1.z * v2.y;
68     double v3y = v1.z * v2.x - v1.x * v2.z;
69     double v3z = v1.x * v2.y - v1.y * v2.x;
70     return new Vector(v3x, v3y, v3z);
71   }
72 
73   // 3D vector dot product
74   private static double dot(Vector v1, Vector v2) {
75     return (v1.x * v2.x) + (v1.y * v2.y) + (v1.z * v2.z);
76   }
