16   // of the convex polygon defined by those points. The input to this
17   // method is an array of N+1 points where points[0] = points[N]
18   public static double convexPolygonArea(Point2D[] points) {
19 
20     int N = points.length - 1;
21 
22     if (N < 3 || points[0] != points[N])
23       throw new IllegalArgumentException(
24           "Make sure N >= 3 and that the points array has n+1 points and points[0] = points[N]");
25 
26     // Let the first point be the fixed point
27     Point2D fp = points[0];
28     double area = 0, fpx = fp.getX(), fpy = fp.getY();
29 
30     for (int i = 1; i < N; i++) {
31 
32       Point2D pt1 = points[i];
33       Point2D pt2 = points[i + 1];
34 
35       // Find the components of the vectors <a, c>
36       // and <b, d> which are the vectors from the
37       // fixed point to the points 'pt1' and 'pt2'
38       double a = pt1.getX() - fpx;
39       double b = pt2.getX() - fpx;
40       double c = pt1.getY() - fpy;
41       double d = pt2.getY() - fpy;
42 
43       // Sum the signed area of the parallelograms formed by
44       // the vectors using formula for the determinant of
45       // two vectors in R2 which is: ad - bc.
46       area += a * d - b * c;
47     }
48 
49     // Return half the area (since we were summing parallelograms)
50     return Math.abs(area) / 2.0;
51   }
