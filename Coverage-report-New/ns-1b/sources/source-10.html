11 public class LineSegmentToGeneralForm {

13   // Given a line segment as (x1,y1), (x2,y2) this method
14   // returns the general form of a line ax + by + c = 0
15   // NOTE: this is not the same as: ax + by = c
16   public static double[] segmentToGeneralForm(double x1, double y1, double x2, double y2) {
17 
18     // Starting with y = mx we can replace y with (y-y1), m with (y2-y1)/(x2-x1)
19     // and x with (x-x1), so we obtain (y-y1) = [(y2-y1)/(x2-x1)](x-x1) and simplify to:
20     // x(y1-y2) + y(x2-x1) + (x1y2-y1x2) = 0 and hence we conclude that a = (y1-y2)
21     // b = (x2-x1) and c = (x1y2-y1x2)
22 
23     // a = abc[0], b = abc[1], c = abc[2]
24     double[] abc = new double[3];
25 
26     abc[0] = y1 - y2;
27     abc[1] = x2 - x1;
28     abc[2] = x1 * y2 - y1 * x2; // or y1*x2 - x1*y2; for ax + by = c
29 
30     return abc;
31   }
