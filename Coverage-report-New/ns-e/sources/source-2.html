1 /**
2  * This file contains an implementation of an integer only queue which is extremely quick and
3  * lightweight. In terms of performance it can outperform java.util.ArrayDeque (Java's fastest queue
4  * implementation) by a factor of 40+! See the benchmark test below for proof. However, the downside
5  * is you need to know an upper bound on the number of elements that will be inside the queue at any
6  * given time for this queue to work.
7  *
8  * @author William Fiset, william.alexandre.fiset@gmail.com, liujingkun, liujkon@gmail.com
9  */
10 package com.williamfiset.algorithms.datastructures.queue;

12 public class IntQueue implements Queue<Integer> {
13 
14   private int[] data;
15   private int front, end;
16   private int size;
17 
18   // maxSize is the maximum number of items
19   // that can be in the queue at any given time
20   public IntQueue(int maxSize) {
21     front = end = size = 0;
22     data = new int[maxSize];
23   }
24 
25   // Return true/false on whether the queue is empty
26   public boolean isEmpty() {
27     return size == 0;
28   }
29 
30   // Return the number of elements inside the queue
31   public int size() {
32     return size;
33   }
34 
35   @Override
36   public Integer peek() {
37     if (isEmpty()) {
38       throw new RuntimeException("Queue is empty");
39     }
40     front = front % data.length;
41     return data[front];
42   }
43 
44   public boolean isFull() {
45     return size == data.length;
46   }
47 
48   // Add an element to the queue
49   @Override
50   public void offer(Integer value) {
51     if (isFull()) {
52       throw new RuntimeException("Queue too small!");
53     }
54     data[end++] = value;
55     size++;
56     end = end % data.length;
57   }
58 
59   // Make sure you check is the queue is not empty before calling poll!
60   @Override
61   public Integer poll() {
62     if (size == 0) {
63       throw new RuntimeException("Queue is empty");
64     }
65     size--;
66     front = front % data.length;
67     return data[front++];
68   }
