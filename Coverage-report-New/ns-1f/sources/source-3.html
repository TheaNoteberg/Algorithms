37   public void solve() {
38     // next[i] indicates the next unused edge index in the adjacency list for node i. This is part
39     // of the Shimon Even and Alon Itai optimization of pruning deads ends as part of the DFS phase.
40     int[] next = new int[n];
41 
42     while (bfs()) {
43       Arrays.fill(next, 0);
44       // Find max flow by adding all augmenting path flows.
45       for (long f = dfs(s, next, INF); f != 0; f = dfs(s, next, INF)) {
46         maxFlow += f;
47       }
48     }
49 
50     for (int i = 0; i < n; i++) if (level[i] != -1) minCut[i] = true;
51   }

53   // Do a BFS from source to sink and compute the depth/level of each node
54   // which is the minimum number of edges from that node to the source.
55   private boolean bfs() {
56     Arrays.fill(level, -1);
57     level[s] = 0;
58     Deque<Integer> q = new ArrayDeque<>(n);
59     q.offer(s);
60     while (!q.isEmpty()) {
61       int node = q.poll();
62       for (Edge edge : graph[node]) {
63         long cap = edge.remainingCapacity();
64         if (cap > 0 && level[edge.to] == -1) {
65           level[edge.to] = level[node] + 1;
66           q.offer(edge.to);
67         }
68       }
69     }
70     return level[t] != -1;
71   }

73   private long dfs(int at, int[] next, long flow) {
74     if (at == t) return flow;
75     final int numEdges = graph[at].size();
76 
77     for (; next[at] < numEdges; next[at]++) {
78       Edge edge = graph[at].get(next[at]);
79       long cap = edge.remainingCapacity();
80       if (cap > 0 && level[edge.to] == level[at] + 1) {
81 
82         long bottleNeck = dfs(edge.to, next, min(flow, cap));
83         if (bottleNeck > 0) {
84           edge.augment(bottleNeck);
85           return bottleNeck;
86         }
87       }
88     }
89     return 0;
90   }
