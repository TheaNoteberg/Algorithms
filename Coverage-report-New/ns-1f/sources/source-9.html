33 
34     // Sum up the bottlenecks on each augmenting path to find the max flow and min cost.
35     List<Edge> path;
36     while ((path = getAugmentingPath()).size() != 0) {
37 
38       // Find bottle neck edge value along path.
39       long bottleNeck = Long.MAX_VALUE;
40       for (Edge edge : path) bottleNeck = min(bottleNeck, edge.remainingCapacity());
41 
42       // Retrace path while augmenting the flow
43       for (Edge edge : path) {
44         edge.augment(bottleNeck);
45         minCost += bottleNeck * edge.originalCost;
46       }
47       maxFlow += bottleNeck;
48     }

50     // TODO(williamfiset): Compute mincut.

53   /**
54    * Use the Bellman-Ford algorithm (which work with negative edge weights) to find an augmenting
55    * path through the flow network.
56    */
57   private List<Edge> getAugmentingPath() {
58     long[] dist = new long[n];
59     Arrays.fill(dist, INF);
60     dist[s] = 0;
61 
62     Edge[] prev = new Edge[n];
63 
64     // For each vertex, relax all the edges in the graph, O(VE)
65     for (int i = 0; i < n - 1; i++) {
66       for (int from = 0; from < n; from++) {
67         for (Edge edge : graph[from]) {
68           if (edge.remainingCapacity() > 0 && dist[from] + edge.cost < dist[edge.to]) {
69             dist[edge.to] = dist[from] + edge.cost;
70             prev[edge.to] = edge;
71           }
72         }
73       }
74     }
75 
76     // Retrace augmenting path from sink back to the source.
77     LinkedList<Edge> path = new LinkedList<>();
78     for (Edge edge = prev[t]; edge != null; edge = prev[edge.from]) path.addFirst(edge);
79     return path;
80   }
