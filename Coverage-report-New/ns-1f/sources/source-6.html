30   // Performs the Ford-Fulkerson method applying a depth first search as
31   // a means of finding an augmenting path.
32   @Override
33   public void solve() {
34 
35     // Find max flow by adding all augmenting path flows.
36     for (long f = dfs(s, INF); f != 0; f = dfs(s, INF)) {
37       markAllNodesAsUnvisited();
38       maxFlow += f;
39     }
40 
41     // Find min cut.
42     for (int i = 0; i < n; i++) if (visited(i)) minCut[i] = true;
43   }

45   private long dfs(int node, long flow) {
46     // At sink node, return augmented path flow.
47     if (node == t) return flow;
48 
49     List<Edge> edges = graph[node];
50     visit(node);
51 
52     for (Edge edge : edges) {
53       long rcap = edge.remainingCapacity();
54       if (rcap > 0 && !visited(edge.to)) {
55         long bottleNeck = dfs(edge.to, min(flow, rcap));
56 
57         // Augment flow with bottle neck value
58         if (bottleNeck > 0) {
59           edge.augment(bottleNeck);
60           return bottleNeck;
61         }
62       }
63     }
64     return 0;
65   }
