44   private static int augment(List<List<Integer>> graph, int[] visited, int[] next, int at) {
45 
46     // Node already visited in this augmenting path
47     if (visited[at] == visitToken) return 0;
48     visited[at] = visitToken;
49 
50     for (int node : graph.get(at)) {
51       // If the value of oppositeNode is FREE then it has not yet been
52       // matched, otherwise the value refers to the index of the node
53       // used to reach the oppositeNode.
54       int oppositeNode = next[node];
55 
56       if (oppositeNode == FREE) {
57         // Record which node we came from and return
58         // 1 to indicate a path was found
59         next[node] = at;
60         return 1;
61       }
62 
63       // We were able to find an alternating path
64       if (augment(graph, visited, next, oppositeNode) != 0) {
65 
66         // Record which node we came from and return
67         // 1 to indicate a path was found
68         next[node] = at;
69         return 1;
70       }
71     }
72 
73     // No path found :/
74     return 0;
75   }
