3  * an array. This range adder is especially useful for offline algorithms which know all range
4  * updates ahead of time.

23   // Initialize an instance of a LazyRangeAdder on some input values
24   public LazyRangeAdder(int[] array) {
25     this.array = array;
26     this.n = array.length;
27 
28     differenceArray = new int[n + 1];
29     differenceArray[0] = array[0];
30     for (int i = 1; i < n; i++) {
31       differenceArray[i] = array[i] - array[i - 1];
32     }
33   }
34 
35   // Add `x` to the range [l, r] inclusive
36   public void add(int l, int r, int x) {
37     differenceArray[l] += x;

39   }
40 
41   // IMPORTANT: Make certain to call this method once all the additions
42   // have been made with add(l, r, x)
43   public void done() {
44     for (int i = 0; i < n; i++) {
45       if (i == 0) {
46         array[i] = differenceArray[i];
47       } else {
48         array[i] = differenceArray[i] + array[i - 1];
49       }
50     }
51   }
