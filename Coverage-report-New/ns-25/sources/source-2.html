60   public static boolean nextCombination(int[] selection, int N, int r) {
61     if (r > N) throw new IllegalArgumentException("r must be <= N");
62     int i = r - 1;

64     selection[i]++;
65     for (int j = i + 1; j < r; j++) selection[j] = selection[i] + j - i;
66     return true;
