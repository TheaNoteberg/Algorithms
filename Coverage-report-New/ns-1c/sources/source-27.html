54   // Solves the traveling salesman problem and caches solution.
55   public void solve() {
56 
57     if (ranSolver) return;
58 
59     final int END_STATE = (1 << N) - 1;
60     Double[][] memo = new Double[N][1 << N];
61 
62     // Add all outgoing edges from the starting node to memo table.
63     for (int end = 0; end < N; end++) {
64       if (end == start) continue;
65       memo[end][(1 << start) | (1 << end)] = distance[start][end];
66     }
67 
68     for (int r = 3; r <= N; r++) {
69       for (int subset : combinations(r, N)) {
70         if (notIn(start, subset)) continue;
71         for (int next = 0; next < N; next++) {
72           if (next == start || notIn(next, subset)) continue;
73           int subsetWithoutNext = subset ^ (1 << next);
74           double minDist = Double.POSITIVE_INFINITY;
75           for (int end = 0; end < N; end++) {
76             if (end == start || end == next || notIn(end, subset)) continue;
77             double newDistance = memo[end][subsetWithoutNext] + distance[end][next];
78             if (newDistance < minDist) {
79               minDist = newDistance;
80             }
81           }
82           memo[next][subset] = minDist;
83         }
84       }
85     }
86 
87     // Connect tour back to starting node and minimize cost.
88     for (int i = 0; i < N; i++) {
89       if (i == start) continue;
90       double tourCost = memo[i][END_STATE] + distance[i][start];
91       if (tourCost < minTourCost) {
92         minTourCost = tourCost;
93       }
94     }
95 
96     int lastIndex = start;
97     int state = END_STATE;
98     tour.add(start);
99 
100     // Reconstruct TSP path from memo table.
101     for (int i = 1; i < N; i++) {
102 
103       int bestIndex = -1;
104       double bestDist = Double.POSITIVE_INFINITY;
105       for (int j = 0; j < N; j++) {
106         if (j == start || notIn(j, state)) continue;
107         double newDist = memo[j][state] + distance[j][lastIndex];
108         if (newDist < bestDist) {
109           bestIndex = j;
110           bestDist = newDist;
111         }
112       }
113 
114       tour.add(bestIndex);
115       state = state ^ (1 << bestIndex);
116       lastIndex = bestIndex;
117     }
118 
119     tour.add(start);
120     Collections.reverse(tour);
121 
122     ranSolver = true;
123   }

137   // To find all the combinations of size r we need to recurse until we have
138   // selected r elements (aka r = 0), otherwise if r != 0 then we still need to select
139   // an element which is found after the position of our last selected element
140   private static void combinations(int set, int at, int r, int n, List<Integer> subsets) {
141 
142     // Return early if there are more elements left to select than what is available.
143     int elementsLeftToPick = n - at;
144     if (elementsLeftToPick < r) return;
145 
146     // We selected 'r' elements so we found a valid subset!
147     if (r == 0) {
148       subsets.add(set);
149     } else {
150       for (int i = at; i < n; i++) {
151         // Try including this element
152         set ^= (1 << i);
153 
154         combinations(set, i + 1, r - 1, n, subsets);
155 
156         // Backtrack and try the instance where we did not include this element
157         set ^= (1 << i);
158       }
159     }
160   }
