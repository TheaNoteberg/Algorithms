17   // indexes of the nodes in topological order
18   public static int[] topologicalSort(Double[][] adj) {
19 
20     // Setup
21     int n = adj.length;
22     boolean[] visited = new boolean[n];
23     int[] order = new int[n];
24     int index = n - 1;
25 
26     // Visit each node
27     for (int u = 0; u < n; u++) if (!visited[u]) index = visit(adj, visited, order, index, u);
28 
29     // Return topological sort
30     return order;
31   }

33   private static int visit(Double[][] adj, boolean[] visited, int[] order, int index, int u) {
34 
35     if (visited[u]) return index;
36     visited[u] = true;
37 
38     // Visit all neighbors
39     for (int v = 0; v < adj.length; v++)
40       if (adj[u][v] != null && !visited[v]) index = visit(adj, visited, order, index, v);
41 
42     // Place this node at the head of the list
43     order[index--] = u;
44 
45     return index;
46   }

52   public static double[] dagShortestPath(Double[][] adj, int start) {
53 
54     // Set up array used to maintain minimum distances from start
55     int n = adj.length;

57     java.util.Arrays.fill(dist, Double.POSITIVE_INFINITY);

61     for (int u : topologicalSort(adj))
62       for (int v = 0; v < n; v++) {
63         if (adj[u][v] != null) {
64           double newDist = dist[u] + adj[u][v];
65           if (newDist < dist[v]) dist[v] = newDist;
66         }
67       }
68 
69     // Return minimum distance
70     return dist;
71   }
