64   private void relaxEdgesAtNode(int currentNodeIndex) {
65     visited[currentNodeIndex] = true;
66 
67     // edges will never be null if the createEmptyGraph method was used to build the graph.
68     List<Edge> edges = graph.get(currentNodeIndex);
69 
70     for (Edge edge : edges) {
71       int destNodeIndex = edge.to;
72 
73       // Skip edges pointing to already visited nodes.
74       if (visited[destNodeIndex]) continue;
75 
76       if (ipq.contains(destNodeIndex)) {
77         // Try and improve the cheapest edge at destNodeIndex with the current edge in the IPQ.
78         ipq.decrease(destNodeIndex, edge);
79       } else {
80         // Insert edge for the first time.
81         ipq.insert(destNodeIndex, edge);
82       }
83     }
84   }

87   private void solve() {
88     if (solved) return;
89     solved = true;
90 
91     int m = n - 1, edgeCount = 0;
92     visited = new boolean[n];
93     mstEdges = new Edge[m];
94 
95     // The degree of the d-ary heap supporting the IPQ can greatly impact performance, especially
96     // on dense graphs. The base 2 logarithm of n is a decent value based on my quick experiments
97     // (even better than E/V in many cases).
98     int degree = (int) Math.ceil(Math.log(n) / Math.log(2));
99     ipq = new MinIndexedDHeap<>(max(2, degree), n);
100 
101     // Add initial set of edges to the priority queue starting at node 0.
102     relaxEdgesAtNode(0);
103 
104     while (!ipq.isEmpty() && edgeCount != m) {
105       int destNodeIndex = ipq.peekMinKeyIndex(); // equivalently: edge.to
106       Edge edge = ipq.pollMinValue();
107 
108       mstEdges[edgeCount++] = edge;
109       minCostSum += edge.cost;
110 
111       relaxEdgesAtNode(destNodeIndex);
112     }
113 
114     // Verify MST spans entire graph.
115     mstExists = (edgeCount == m);
116   }

388   private static class MinIndexedDHeap<T extends Comparable<T>> {
389 
390     // Current number of elements in the heap.
391     private int sz;
392 
393     // Maximum number of elements in the heap.
394     private final int N;
395 
396     // The degree of every node in the heap.
397     private final int D;
398 
399     // Lookup arrays to track the child/parent indexes of each node.
400     private final int[] child, parent;
401 
402     // The Position Map (pm) maps Key Indexes (ki) to where the position of that
403     // key is represented in the priority queue in the domain [0, sz).
404     public final int[] pm;
405 
406     // The Inverse Map (im) stores the indexes of the keys in the range
407     // [0, sz) which make up the priority queue. It should be noted that
408     // 'im' and 'pm' are inverses of each other, so: pm[im[i]] = im[pm[i]] = i
409     public final int[] im;
410 
411     // The values associated with the keys. It is very important  to note
412     // that this array is indexed by the key indexes (aka 'ki').
413     public final Object[] values;
414 
415     // Initializes a D-ary heap with a maximum capacity of maxSize.
416     public MinIndexedDHeap(int degree, int maxSize) {
417       if (maxSize <= 0) throw new IllegalArgumentException("maxSize <= 0");
418 
419       D = max(2, degree);
420       N = max(D + 1, maxSize);
421 
422       im = new int[N];
423       pm = new int[N];
424       child = new int[N];
425       parent = new int[N];
426       values = new Object[N];
427 
428       for (int i = 0; i < N; i++) {
429         parent[i] = (i - 1) / D;
430         child[i] = i * D + 1;
431         pm[i] = im[i] = -1;
432       }
433     }
434

436       return sz;
437     }
438 
439     public boolean isEmpty() {
440       return sz == 0;
441     }

444       keyInBoundsOrThrow(ki);
445       return pm[ki] != -1;
446     }

449       isNotEmptyOrThrow();

461       isNotEmptyOrThrow();

471     public void insert(int ki, T value) {
472       if (contains(ki)) throw new IllegalArgumentException("index already exists; received: " + ki);
473       valueNotNullOrThrow(value);
474       pm[ki] = sz;
475       im[sz] = ki;
476       values[ki] = value;
477       swim(sz++);
478     }

480     @SuppressWarnings("unchecked")

488       keyExistsOrThrow(ki);
489       final int i = pm[ki];
490       swap(i, --sz);
491       sink(i);
492       swim(i);

494       values[ki] = null;

502       keyExistsAndValueNotNullOrThrow(ki, value);
503       final int i = pm[ki];
504       T oldValue = (T) values[ki];

512     public void decrease(int ki, T value) {
513       keyExistsAndValueNotNullOrThrow(ki, value);
514       if (less(value, values[ki])) {
515         values[ki] = value;
516         swim(pm[ki]);
517       }
518     }

521     public void increase(int ki, T value) {
522       keyExistsAndValueNotNullOrThrow(ki, value);
523       if (less(values[ki], value)) {
524         values[ki] = value;
525         sink(pm[ki]);
526       }

531     private void sink(int i) {
532       for (int j = minChild(i); j != -1; ) {
533         swap(i, j);
534         i = j;
535         j = minChild(i);
536       }
537     }
538 
539     private void swim(int i) {
540       while (less(i, parent[i])) {
541         swap(i, parent[i]);
542         i = parent[i];
543       }
544     }
545 
546     // From the parent node at index i find the minimum child below it
547     private int minChild(int i) {
548       int index = -1, from = child[i], to = min(sz, from + D);
549       for (int j = from; j < to; j++) if (less(j, i)) index = i = j;
550       return index;
551     }
552 
553     private void swap(int i, int j) {
554       pm[im[j]] = i;
555       pm[im[i]] = j;
556       int tmp = im[i];
557       im[i] = im[j];
558       im[j] = tmp;
559     }
560 
561     // Tests if the value of node i < node j
562     @SuppressWarnings("unchecked")
563     private boolean less(int i, int j) {
564       return ((Comparable<? super T>) values[im[i]]).compareTo((T) values[im[j]]) < 0;
565     }

568     private boolean less(Object obj1, Object obj2) {

570     }

612       int from = child[i], to = min(sz, from + D);
613       for (int j = from; j < to; j++) {
