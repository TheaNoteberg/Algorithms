11 // This file contains an implementation of an integer only stack which is
12 // extremely quick and lightweight. In terms of performance it can outperform
13 // java.util.ArrayDeque (Java's fastest stack implementation) by a factor of 50!
14 // However, the downside is you need to know an upper bound on the number of
15 // elements that will be inside the stack at any given time for it to work correctly.
16 class IntStack {
17 
18   private int[] ar;
19   private int pos = 0, sz;
20 
21   // max_sz is the maximum number of items
22   // that can be in the queue at any given time
23   public IntStack(int max_sz) {
24     ar = new int[(sz = max_sz)];
25   }
26 
27   public boolean isEmpty() {
28     return pos == 0;
29   }
30 
31   // Returns the element at the top of the stack
32   public int peek() {
33     return ar[pos - 1];
34   }
35 
36   // Add an element to the top of the stack
37   public void push(int value) {
38     ar[pos++] = value;
39   }
40 
41   // Make sure you check that the stack is not empty before calling pop!
42   public int pop() {
43     return ar[--pos];
44   }
45 }

63

65     boolean[] visited = new boolean[n];
66     IntStack stack = new IntStack(n);
67 
68     // Start by visiting the starting node
69     stack.push(start);
70 
71     while (!stack.isEmpty()) {
72       int node = stack.pop();
73       if (!visited[node]) {
74 
75         count++;
76         visited[node] = true;
77         List<Edge> edges = graph.get(node);
78 
79         if (edges != null) {
80           for (Edge edge : edges) {
81             if (!visited[edge.to]) {
82               stack.push(edge.to);
83             }
84           }
85         }
86       }
87     }
88
