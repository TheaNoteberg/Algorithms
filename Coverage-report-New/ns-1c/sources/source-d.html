77 class UnionFind {
78 
79   // The number of elements in this union find
80   private int size;
81 
82   // Used to track the sizes of each of the components
83   private int[] sz;
84 
85   // id[i] points to the parent of i, if id[i] = i then i is a root node
86   private int[] id;
87 
88   // Tracks the number of components in the union find
89   private int numComponents;
90 
91   public UnionFind(int size) {
92 
93     if (size <= 0) throw new IllegalArgumentException("Size <= 0 is not allowed");
94 
95     this.size = numComponents = size;
96     sz = new int[size];
97     id = new int[size];
98 
99     for (int i = 0; i < size; i++) {
100       id[i] = i; // Link to itself (self root)
101       sz[i] = 1; // Each component is originally of size one
102     }
103   }
104 
105   // Find which component/set 'p' belongs to, takes amortized constant time.
106   public int find(int p) {
107 
108     // Find the root of the component/set
109     int root = p;
110     while (root != id[root]) root = id[root];
111 
112     // Compress the path leading back to the root.
113     // Doing this operation is called "path compression"
114     // and is what gives us amortized constant time complexity.
115     while (p != root) {
116       int next = id[p];
117       id[p] = root;
118       p = next;
119     }
120 
121     return root;
122   }
123 
124   // Return whether or not the elements 'p' and
125   // 'q' are in the same components/set.
126   public boolean connected(int p, int q) {
127     return find(p) == find(q);
128   }
129 
130   // Return the size of the components/set 'p' belongs to
131   public int componentSize(int p) {
132     return sz[find(p)];
133   }
134 
135   // Return the number of elements in this UnionFind/Disjoint set
136   public int size() {

138   }
139

141   public int components() {

143   }

145   // Unify the components/sets containing elements 'p' and 'q'
146   public void unify(int p, int q) {
147 
148     int root1 = find(p);
149     int root2 = find(q);
150 
151     // These elements are already in the same group!
152     if (root1 == root2) return;
153 
154     // Merge smaller component/set into the larger one.
155     if (sz[root1] < sz[root2]) {
156       sz[root2] += sz[root1];
157       id[root1] = root2;
158     } else {
159       sz[root1] += sz[root2];
160       id[root2] = root1;
161     }
162 
163     // Since the roots found are different we know that the
164     // number of components/sets has decreased by one
165     numComponents--;
166   }
