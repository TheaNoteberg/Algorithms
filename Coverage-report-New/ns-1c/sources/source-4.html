43   public List<Integer> reconstructShortestPath(int end) {
44     if (!solved) solve();
45     LinkedList<Integer> path = new LinkedList<>();
46     if (dist[end] == Double.POSITIVE_INFINITY) return path;
47     for (int at = end; prev[at] != null; at = prev[at]) {
48       // Return null since there are an infinite number of shortest paths.
49       if (prev[at] == -1) return null;

56   public void solve() {
57     if (solved) return;
58 
59     // Initialize the distance to all nodes to be infinity
60     // except for the start node which is zero.
61     dist = new double[n];
62     java.util.Arrays.fill(dist, Double.POSITIVE_INFINITY);
63     dist[start] = 0;
64 
65     // Initialize prev array which will allows for shortest path
66     // reconstruction after the algorithm has terminated.
67     prev = new Integer[n];
68 
69     // For each vertex, apply relaxation for all the edges
70     for (int k = 0; k < n - 1; k++)
71       for (int i = 0; i < n; i++)
72         for (int j = 0; j < n; j++)
73           if (dist[i] + matrix[i][j] < dist[j]) {
74             dist[j] = dist[i] + matrix[i][j];
75             prev[j] = i;
76           }
77 
78     // Run algorithm a second time to detect which nodes are part
79     // of a negative cycle. A negative cycle has occurred if we
80     // can find a better path beyond the optimal solution.
81     for (int k = 0; k < n - 1; k++)
82       for (int i = 0; i < n; i++)
83         for (int j = 0; j < n; j++)
84           if (dist[i] + matrix[i][j] < dist[j]) {
85             dist[j] = Double.NEGATIVE_INFINITY;
86             prev[j] = -1;
87           }
88
