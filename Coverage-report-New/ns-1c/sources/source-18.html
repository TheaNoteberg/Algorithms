27   // to easily determine the distance to other nodes
28   static final int DEPTH_TOKEN = -1;

34   // Compute the eccentricity from a given node. The eccentricity
35   // is the distance to the furthest node(s).
36   private static int eccentricity(int nodeID, Map<Integer, List<Edge>> graph) {
37 
38     VISITED_TOKEN++;
39 
40     queue.offer(nodeID);
41     queue.offer(DEPTH_TOKEN);
42     visited.put(nodeID, VISITED_TOKEN);
43 
44     int depth = 0;
45 
46     // Do BFS to count the
47     while (true) {
48 
49       Integer id = queue.poll();
50 
51       // If we encounter a depth token this means that we
52       // have finished the current frontier and are about
53       // to start the new layer (some of which may already
54       // be in the queue) or have reached the end.
55       if (id == DEPTH_TOKEN) {
56 
57         // No more nodes to process
58         if (queue.isEmpty()) break;
59 
60         // Add another DEPTH_TOKEN
61         queue.offer(DEPTH_TOKEN);
62 
63         // Increase the max depth for each DEPTH_TOKEN seen
64         depth++;
65 
66       } else {
67 
68         List<Edge> edges = graph.get(id);
69         if (edges != null) {
70           for (Edge edge : edges) {
71             if (visited.get(edge.to) != VISITED_TOKEN) {
72               visited.put(edge.to, VISITED_TOKEN);
73               queue.offer(edge.to);
74             }
75           }
76         }
77       }
78     }
79 
80     return depth;
81   }

83   // Compute the diameter of an arbitrary graph
84   // NOTE: The input graph should be undirected
85   public static int graphDiameter(Map<Integer, List<Edge>> graph) {
86 
87     if (graph == null) return 0;
88 
89     int diameter = 0;
90     int radius = Integer.MAX_VALUE;
91 
92     // The diameter of a graph is the maximum of all the eccentricity values from all nodes.
93     // The radius on the other hand is the minimum of all the eccentricity values from all nodes.
94     for (Integer nodeID : graph.keySet()) {
95       int eccentricity = eccentricity(nodeID, graph);
96       diameter = Math.max(diameter, eccentricity);
97       radius = Math.min(radius, eccentricity);
98     }
99 
100     // return radius;
101     return diameter;
102   }
