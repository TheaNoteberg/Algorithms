38   /**
39    * Reconstructs the path (of nodes) from 'start' to 'end' inclusive. If the edges are unweighted
40    * then this method returns the shortest path from 'start' to 'end'
41    *
42    * @return An array of nodes indexes of the shortest path from 'start' to 'end'. If 'start' and
43    *     'end' are not connected then an empty array is returned.
44    */
45   public List<Integer> reconstructPath(int start, int end) {
46     bfs(start);
47     List<Integer> path = new ArrayList<>();
48     for (Integer at = end; at != null; at = prev[at]) path.add(at);
49     Collections.reverse(path);
50     if (path.get(0) == start) return path;
51     path.clear();
52     return path;
53   }

55   // Perform a breadth first search on a graph a starting node 'start'.
56   private void bfs(int start) {
57     prev = new Integer[n];
58     boolean[] visited = new boolean[n];
59     Deque<Integer> queue = new ArrayDeque<>(n);
60 
61     // Start by visiting the 'start' node and add it to the queue.
62     queue.offer(start);
63     visited[start] = true;
64 
65     // Continue until the BFS is done.
66     while (!queue.isEmpty()) {
67       int node = queue.poll();
68       List<Edge> edges = graph.get(node);
69 
70       // Loop through all edges attached to this node. Mark nodes as visited once they're
71       // in the queue. This will prevent having duplicate nodes in the queue and speedup the BFS.
72       for (Edge edge : edges) {
73         if (!visited[edge.to]) {
74           visited[edge.to] = true;
75           prev[edge.to] = node;
76           queue.offer(edge.to);
77         }
78       }
79     }
80   }
