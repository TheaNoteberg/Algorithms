30   // Returns the indexes for all articulation points in the graph even if the
31   // graph is not fully connected.
32   public boolean[] findArticulationPoints() {
33     if (solved) return isArticulationPoint;
34 
35     id = 0;
36     low = new int[n]; // Low link values
37     ids = new int[n]; // Nodes ids
38     visited = new boolean[n];
39     isArticulationPoint = new boolean[n];
40 
41     for (int i = 0; i < n; i++) {
42       if (!visited[i]) {
43         rootNodeOutcomingEdgeCount = 0;
44         dfs(i, i, -1);
45         isArticulationPoint[i] = (rootNodeOutcomingEdgeCount > 1);
46       }
47     }
48 
49     solved = true;
50     return isArticulationPoint;
51   }

53   private void dfs(int root, int at, int parent) {
54 
55     if (parent == root) rootNodeOutcomingEdgeCount++;
56 
57     visited[at] = true;
58     low[at] = ids[at] = id++;
59 
60     List<Integer> edges = graph.get(at);
61     for (Integer to : edges) {
62       if (to == parent) continue;
63       if (!visited[to]) {
64         dfs(root, to, at);
65         low[at] = min(low[at], low[to]);
66         if (ids[at] <= low[to]) {
67           isArticulationPoint[at] = true;
68         }
69       } else {
70         low[at] = min(low[at], ids[to]);
71       }
72     }
73   }
