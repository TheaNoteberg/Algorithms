4  * <p>Kahn's algorithm finds a topological ordering by iteratively removing nodes in the graph which
5  * have no incoming edges. When a node is removed from the graph, it is added to the topological

26   // Given a an acyclic graph `g` represented as a adjacency list, return a
27   // topological ordering on the nodes of the graph.
28   public int[] kahns(List<List<Integer>> g) {
29     int n = g.size();
30 
31     // Calculate the in-degree of each node.
32     int[] inDegree = new int[n];
33     for (List<Integer> edges : g) {
34       for (int to : edges) {
35         inDegree[to]++;
36       }
37     }
38 
39     // q always contains the set nodes with no incoming edges.
40     Queue<Integer> q = new ArrayDeque<>();
41 
42     // Find all start nodes.
43     for (int i = 0; i < n; i++) {
44       if (inDegree[i] == 0) {
45         q.offer(i);
46       }
47     }
48 
49     int index = 0;
50     int[] order = new int[n];
51     while (!q.isEmpty()) {
52       int at = q.poll();
53       order[index++] = at;
54       for (int to : g.get(at)) {
55         inDegree[to]--;
56         if (inDegree[to] == 0) {
57           q.offer(to);
58         }
59       }
60     }
61     if (index != n) {
62       throw new IllegalArgumentException("Graph is not acyclic! Detected a cycle.");
63     }
64     return order;
65   }
