32   // Returns a list of pairs of nodes indicating which nodes form bridges.
33   // The returned list is always of even length and indexes (2*i, 2*i+1) form a
34   // pair. For example, nodes are indexes (0, 1) are a pair, (2, 3) are another
35   // pair, etc...
36   public List<Integer> findBridges() {
37     if (solved) return bridges;
38 
39     id = 0;
40     low = new int[n]; // Low link values
41     ids = new int[n]; // Nodes ids
42     visited = new boolean[n];
43 
44     bridges = new ArrayList<>();
45 
46     // Finds all bridges even if the graph is not one single connected component.
47     for (int i = 0; i < n; i++) {
48       if (visited[i]) continue;
49 
50       Deque<Integer> stack = new ArrayDeque<>();
51       Deque<Integer> parentStack = new ArrayDeque<>();
52       stack.push(i);
53       parentStack.push(-1);
54 
55       while (!stack.isEmpty()) {
56         int at = stack.pop();
57 
58         if (at == CALLBACK_TOKEN) {
59           at = stack.pop();
60           int to = stack.pop();
61           low[at] = min(low[at], low[to]);
62           if (ids[at] < low[to]) {
63             bridges.add(at);
64             bridges.add(to);
65           }
66           continue;
67         }
68 
69         int parent = parentStack.pop();
70         if (!visited[at]) {
71           low[at] = ids[at] = ++id;
72           visited[at] = true;
73 
74           List<Integer> edges = graph.get(at);
75           for (Integer to : edges) {
76             if (to == parent) continue;
77             if (!visited[to]) {
78               stack.push(to);
79               stack.push(at);
80               stack.push(CALLBACK_TOKEN);
81               stack.push(to);
82               parentStack.push(at);
83             } else {
84               low[at] = min(low[at], ids[to]);
85             }
86           }
87         }
88       }
89     }
90 
91     solved = true;
92     return bridges;
93   }
