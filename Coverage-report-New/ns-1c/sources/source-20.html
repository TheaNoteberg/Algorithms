30   // Given an N*N undirected adjacency matrix, that is a
31   // graph with matrix[i][j] = matrix[j][i] for all i,j this method
32   // finds the minimum spanning tree cost using Prim's algorithm
33   public static Long prims(Integer[][] graph) {
34 
35     int n = graph.length;
36     long sum = 0, visitedNodes = 1;
37     PriorityQueue<Edge> pq = new PriorityQueue<>();
38 
39     boolean[] connected = new boolean[n];
40     connected[0] = true;
41 
42     for (int i = 1; i < n; i++) if (graph[0][i] != null) pq.offer(new Edge(i, graph[0][i]));
43 
44     // Loop while the MST is not complete
45     while (visitedNodes != n && !pq.isEmpty()) {
46 
47       Edge edge = pq.poll();
48 
49       if (!connected[edge.to]) {
50 
51         // Update minimum distances
52         for (int i = 0; i < n; i++)
53           if (!connected[i] && graph[edge.to][i] != null) pq.offer(new Edge(i, graph[edge.to][i]));
54 
55         connected[edge.to] = true;
56         sum += edge.cost;
57         visitedNodes++;
58       }
59     }
60 
61     // Make sure MST spans the whole graph
62     if (visitedNodes != n) return null;
63     return sum;
64   }
