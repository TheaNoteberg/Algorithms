50   private void solve() {
51     sccCount = 0;
52     sccs = new int[n];
53     visited = new boolean[n];
54     postOrderTraversal = new ArrayList<>();
55 
56     for (int i = 0; i < n; i++) {
57       dfs1(i);
58     }
59 
60     Arrays.fill(visited, false);
61     createTransposeGraph();
62 
63     // Reverse the post order traversal to make iterating through it
64     // in the next step more intuitive.
65     Collections.reverse(postOrderTraversal);
66 
67     for (int node : postOrderTraversal) {
68       if (!visited[node]) {
69         dfs2(node);
70         sccCount++;
71       }
72     }
73 
74     solved = true;
75   }

77   // Traverse the original graph and add nodes to the post order traversal on the callback.
78   private void dfs1(int from) {
79     if (visited[from]) {
80       return;
81     }
82     visited[from] = true;
83     for (int to : graph.get(from)) {
84       dfs1(to);
85     }
86     postOrderTraversal.add(from);
87   }

90   private void dfs2(int from) {
91     if (visited[from]) {
92       return;
93     }
94     visited[from] = true;
95     for (int to : transposeGraph.get(from)) {
96       dfs2(to);
97     }
98     sccs[from] = sccCount;
99   }

233   private static void runKosaraju(List<List<Integer>> graph) {
234     int n = graph.size();
235     Kosaraju solver = new Kosaraju(graph);
236     int[] sccs = solver.getSccs();
237     Map<Integer, List<Integer>> multimap = new HashMap<>();
238     for (int i = 0; i < n; i++) {
239       if (!multimap.containsKey(sccs[i])) multimap.put(sccs[i], new ArrayList<>());
240       multimap.get(sccs[i]).add(i);
241     }
242 
243     System.out.printf("Number of Strongly Connected Components: %d\n", solver.sccCount());
244     for (List<Integer> scc : multimap.values()) {
245       System.out.println("Nodes: " + scc + " form a Strongly Connected Component.");
