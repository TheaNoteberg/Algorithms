10 // A custom implementation of a circular integer only queue which is
11 // extremely quick and lightweight. In terms of performance it can outperform
12 // java.util.ArrayDeque (Java's fastest queue implementation) by a factor of 40+!
13 // However, the downside is you need to know an upper bound on the number of elements
14 // that will be inside the queue at any given time for this queue to work.
15 class IntQueue {
16 
17   private int[] ar;
18   private int front, end, sz;
19 
20   // max_sz is the maximum number of items
21   // that can be in the queue at any given time
22   public IntQueue(int max_sz) {
23     front = end = 0;
24     this.sz = max_sz + 1;
25     ar = new int[sz];
26   }
27 
28   public boolean isEmpty() {
29     return front == end;
30   }
31 
32   public int peek() {
33     return ar[front];
34   }
35 
36   // Add an element to the queue
37   public void enqueue(int value) {
38     ar[end] = value;
39     if (++end == sz) end = 0;
40     if (end == front) throw new RuntimeException("Queue too small!");
41   }
42 
43   // Make sure you check is the queue is not empty before calling dequeue!
44   public int dequeue() {
45     int ret_val = ar[front];
46     if (++front == sz) front = 0;
47     return ret_val;
48   }

63   // Perform a breadth first search on a graph with n nodes
64   // from a starting point to count the number of nodes
65   // in a given component.
66   static int bfs(Map<Integer, List<Edge>> graph, int start, int n) {
67 
68     int count = 0;
69     boolean[] visited = new boolean[n];
70     IntQueue queue = new IntQueue(n + 1);
71 
72     // For each breadth first search layer gets separated by a DEPTH_TOKEN
73     // to easily augment this method for additional functionality
74     int DEPTH_TOKEN = -1;
75 
76     // Start by visiting the starting node
77     queue.enqueue(start);
78     queue.enqueue(DEPTH_TOKEN);
79     visited[start] = true;
80 
81     // Continue until the BFS is done
82     while (true) {
83 
84       Integer node = queue.dequeue();
85 
86       // If we encounter a depth token this means that we
87       // have finished the current frontier and are about
88       // to start the new layer (some of which may already
89       // be in the queue) or have reached the end.
90       if (node == DEPTH_TOKEN) {
91 
92         // No more nodes to process
93         if (queue.isEmpty()) break;
94 
95         // Add another DEPTH_TOKEN
96         queue.enqueue(DEPTH_TOKEN);
97 
98       } else {
99 
100         count++;
101 
102         List<Edge> edges = graph.get(node);
103         if (edges != null) {
104 
105           // Loop through all edges attached to this node. Mark nodes as
106           // visited once they're in the queue. This will prevent having
107           // duplicate nodes in the queue and speedup the BFS.
108           for (Edge edge : edges) {
109             if (!visited[edge.to]) {
110               visited[edge.to] = true;
111               queue.enqueue(edge.to);
112             }
113           }
114         }
115       }
116     }
117 
118     return count;
119   }
