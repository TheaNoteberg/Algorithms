24   public static int bfs(List<List<Integer>> graph, int start, int n) {

32   private static int bfs(boolean[] visited, Queue<Integer> queue, List<List<Integer>> graph) {
33 
34     int at = queue.poll();
35 
36     if (at == DEPTH_TOKEN) {
37       queue.offer(DEPTH_TOKEN);
38       return 1;
39     }
40 
41     // This node is already visited.
42     if (visited[at]) return 0;
43 
44     // Visit this node.
45     visited[at] = true;
46 
47     // Add all neighbors to queue.
48     List<Integer> neighbors = graph.get(at);
49     if (neighbors != null) for (int next : neighbors) if (!visited[next]) queue.add(next);
50 
51     int depth = 0;
52 
53     while (true) {
54       // Stop when the queue is empty (i.e there's only one depth token remaining)
55       if (queue.size() == 1 && queue.peek() == DEPTH_TOKEN) break;
56 
57       // The depth is the sum of all DEPTH_TOKENS encountered.
58       depth += bfs(visited, queue, graph);
59     }
60 
61     return depth;
62   }
