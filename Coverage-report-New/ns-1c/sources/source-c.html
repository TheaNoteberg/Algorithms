294       if (solved) return;
295       MatchingCost[] dp = new MatchingCost[1 << n];
296       int[] history = new int[1 << n];
297 
298       MatchingCost matchingCost = f(FULL_STATE, dp, history);
299       minWeightCost = matchingCost.cost;
300 
301       reconstructMatching(history);
302       solved = true;
303     }

305     private MatchingCost f(int state, MatchingCost[] dp, int[] history) {
306       if (dp[state] != null) {
307         return dp[state];
308       }
309       if (state == 0) {
310         return new MatchingCost();
311       }
312       int p1, p2;
313       // Seek to find active bit position (p1)
314       for (p1 = 0; p1 < n; p1++) {
315         if ((state & (1 << p1)) > 0) {
316           break;
317         }
318       }
319 
320       int bestState = -1;
321       MatchingCost bestMatchingCost = MatchingCost.createInfiniteValueMatchingCost();
322 
323       for (p2 = p1 + 1; p2 < n; p2++) {
324         // Position `p2` is on. Try matching the pair (p1, p2) together.
325         if ((state & (1 << p2)) > 0) {
326           int reducedState = state ^ (1 << p1) ^ (1 << p2);
327 
328           MatchingCost matchCost = new MatchingCost(f(reducedState, dp, history));
329           matchCost.updateMatchingCost(cost[p1][p2]);
330 
331           if (matchCost.isBetterMatchingCost(bestMatchingCost)) {
332             bestMatchingCost = matchCost;
333             bestState = reducedState;
334           }
335         }
336       }
337 
338       history[state] = bestState;
339       return dp[state] = bestMatchingCost;
340     }

342     // Populates the `matching` array with a sorted deterministic matching sorted by lowest node
343     // index. For example, if the perfect matching consists of the pairs (3, 4), (1, 5), (0, 2).
344     // The matching is sorted such that the pairs appear in the ordering: (0, 2), (1, 5), (3, 4).
345     // Furthermore, it is guaranteed that for any pair (a, b) that a < b.
346     private void reconstructMatching(int[] history) {
347       // A map between pairs of nodes that were matched together.
348       int[] map = new int[n];
349       int[] leftNodes = new int[n / 2];
350 
351       int matchingSize = 0;
352 
353       // Reconstruct the matching of pairs of nodes working backwards through computed states.
354       for (int i = 0, state = FULL_STATE; state != 0; state = history[state]) {
355         // Isolate the pair used by xoring the state with the state used to generate it.
356         int pairUsed = state ^ history[state];
357 
358         int leftNode = getBitPosition(Integer.lowestOneBit(pairUsed));
359         int rightNode = getBitPosition(Integer.highestOneBit(pairUsed));
360 
361         leftNodes[i++] = leftNode;
362         map[leftNode] = rightNode;
363 
364         if (cost[leftNode][rightNode] != null) {
365           matchingSize++;
366         }
367       }
368 
369       // Sort the left nodes in ascending order.
370       java.util.Arrays.sort(leftNodes);
371 
372       matchingSize = matchingSize * 2;
373       matching = new int[matchingSize];
374 
375       for (int i = 0, j = 0; i < n / 2; i++) {
376         int leftNode = leftNodes[i];
377         int rightNode = map[leftNodes[i]];
378         // Ignore the artificial node when there is an odd number of nodes.
379         if (isOdd && (leftNode == artificialNodeId || rightNode == artificialNodeId)) {
380           continue;
381         }
382         // Only match edges which actually exist
383         if (cost[leftNode][rightNode] != null) {
384           matching[2 * j] = leftNode;
385           matching[2 * j + 1] = rightNode;
386           j++;
387         }
388       }
389     }
