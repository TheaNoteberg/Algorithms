224     PriorityQueue<DNode> pq = new PriorityQueue<>();
225     pq.offer(new DNode(start, 0.0));
226

228     boolean[] visited = new boolean[n];
229 
230     // In the event that you wish to rebuild the shortest path
231     // you can do so using the prev array and starting at some node 'end'
232     // and finding the previous node using prev[end] and the previous node
233     // after that prev[prev[end]] etc... working all the way back until
234     // the index of start is found. Simply uncomment where the prev array is used.

236

244       if (node.value > dists[node.id]) continue;
