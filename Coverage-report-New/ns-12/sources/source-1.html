48   // All functions must be associative, e.g: a * (b * c) = (a * b) * c for some operation '*'
49   private BinaryOperator<Long> sumFn = (a, b) -> a + b;
50   private BinaryOperator<Long> minFn = (a, b) -> Math.min(a, b);

53   private BinaryOperator<Long> gcdFn =
54       (a, b) -> {
55         long gcd = a;
56         while (b != 0) {
57           gcd = b;
58           b = a % b;
59           a = gcd;
60         }
61         return Math.abs(gcd);
62       };

70   private void init(long[] v) {
71     n = v.length;
72 
73     // Tip: to get the floor of the logarithm base 2 in Java you can also do:
74     // Integer.numberOfTrailingZeros(Integer.highestOneBit(n)).
75     P = (int) (Math.log(n) / Math.log(2));
76     dp = new long[P + 1][n];
77     it = new int[P + 1][n];
78 
79     for (int i = 0; i < n; i++) {
80       dp[0][i] = v[i];
81       it[0][i] = i;
82     }
83 
84     log2 = new int[n + 1];
85     for (int i = 2; i <= n; i++) {
86       log2[i] = log2[i / 2] + 1;
87     }
88 
89     // Build sparse table combining the values of the previous intervals.
90     for (int i = 1; i <= P; i++) {
91       for (int j = 0; j + (1 << i) <= n; j++) {
92         long leftInterval = dp[i - 1][j];
93         long rightInterval = dp[i - 1][j + (1 << (i - 1))];
94         if (op == Operation.MIN) {
95           dp[i][j] = minFn.apply(leftInterval, rightInterval);
96           // Propagate the index of the best value
97           if (leftInterval <= rightInterval) {
98             it[i][j] = it[i - 1][j];
99           } else {
100             it[i][j] = it[i - 1][j + (1 << (i - 1))];
101           }
102         } else if (op == Operation.MAX) {
103           dp[i][j] = maxFn.apply(leftInterval, rightInterval);
104           // Propagate the index of the best value
105           if (leftInterval >= rightInterval) {
106             it[i][j] = it[i - 1][j];
107           } else {
108             it[i][j] = it[i - 1][j + (1 << (i - 1))];
109           }
110         } else if (op == Operation.SUM) {
111           dp[i][j] = sumFn.apply(leftInterval, rightInterval);
112         } else if (op == Operation.MULT) {
113           dp[i][j] = multFn.apply(leftInterval, rightInterval);
114         } else if (op == Operation.GCD) {
115           dp[i][j] = gcdFn.apply(leftInterval, rightInterval);
116         }
117       }
118     }
119     // Uncomment for debugging
120     // printTable();
121   }

162   private int minQueryIndex(int l, int r) {
163     int len = r - l + 1;
164     int p = log2[len];
165     long leftInterval = dp[p][l];
166     long rightInterval = dp[p][r - (1 << p) + 1];
167     if (leftInterval <= rightInterval) {
168       return it[p][l];
169     } else {
170       return it[p][r - (1 << p) + 1];

186   // Do sum query [l, r] in O(log2(n)).
187   //
188   // Perform a cascading query which shrinks the left endpoint while summing over all the intervals
189   // which are powers of 2 between [l, r].
190   //
191   // WARNING: This method can easily produces values that overflow.
192   //
193   // NOTE: You can achieve a faster time complexity and use less memory with a simple prefix sum
194   // array. This method is here more as a proof of concept than for its usefulness.
195   private long sumQuery(int l, int r) {
196     long sum = 0;
197     for (int p = log2[r - l + 1]; l <= r; p = log2[r - l + 1]) {
198       sum += dp[p][l];
199       l += (1 << p);
200     }
201     return sum;
202   }
203 
204   private long multQuery(int l, int r) {
205     long result = 1;
206     for (int p = log2[r - l + 1]; l <= r; p = log2[r - l + 1]) {
207       result *= dp[p][l];
208       l += (1 << p);
209     }
210     return result;
211   }
212 
213   // Do either a min, max or gcd query on the interval [l, r] in O(1).
214   //
215   // We can get O(1) query by finding the smallest power of 2 that fits within the interval length
216   // which we'll call k. Then we can query the intervals [l, l+k] and [r-k+1, r] (which likely
217   // overlap) and apply the function again. Some functions (like min and max) don't care about
218   // overlapping intervals so this trick works, but for a function like sum this would return the
219   // wrong result since it is not an idempotent binary function.
220   private long query(int l, int r, BinaryOperator<Long> fn) {
221     int len = r - l + 1;
222     int p = log2[len];
223     return fn.apply(dp[p][l], dp[p][r - (1 << p) + 1]);
224   }
