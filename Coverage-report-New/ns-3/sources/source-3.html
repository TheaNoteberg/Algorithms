182     Node y;
183     while (z.parent.color == RED) {

223     Node y = x.right;

247     Node z;
248     if (key == null || (z = (search(key, root))) == NIL) return false;
249     Node x;

252

279   private void deleteFix(Node x) {
280     while (x != root && x.getColor() == BLACK) {
281       if (x == x.getParent().getLeft()) {
282         Node w = x.getParent().getRight();
283         if (w.getColor() == RED) {
284           w.setColor(BLACK);
285           x.getParent().setColor(RED);
286           leftRotate(x.parent);
287           w = x.getParent().getRight();
288         }
289         if (w.getLeft().getColor() == BLACK && w.getRight().getColor() == BLACK) {
290           w.setColor(RED);
291           x = x.getParent();
292           continue;
293         } else if (w.getRight().getColor() == BLACK) {
294           w.getLeft().setColor(BLACK);
295           w.setColor(RED);
296           rightRotate(w);
297           w = x.getParent().getRight();
298         }
299         if (w.getRight().getColor() == RED) {
300           w.setColor(x.getParent().getColor());
301           x.getParent().setColor(BLACK);
302           w.getRight().setColor(BLACK);
303           leftRotate(x.getParent());

305         }
306       } else {
307         Node w = (x.getParent().getLeft());
308         if (w.color == RED) {

311           rightRotate(x.getParent());

351     if (curr == NIL) return NIL;
352     else if (curr.value.equals(val)) return curr;
353     else if (curr.value.compareTo(val) < 0) return search(val, curr.right);
354     else return search(val, curr.left);
