217   // Removes a value from the AVL tree. If the value we're trying to remove
218   // does not exist in the tree then the 'TOKEN' value is returned. Otherwise,
219   // the new (or old) root node is returned.
220   private Node remove(Node node, T elem) {
221 
222     // Return 'TOKEN' to indicate value to remove was not found.
223     if (node == null) return TOKEN;
224 
225     int cmp = elem.compareTo(node.value);
226 
227     // Dig into left subtree, the value we're looking
228     // for is smaller than the current value.
229     if (cmp < 0) {
230       Node newLeftNode = remove(node.left, elem);
231       if (newLeftNode == TOKEN) return TOKEN;
232       node.left = newLeftNode;
233 
234       // Dig into right subtree, the value we're looking
235       // for is greater than the current value.
236     } else if (cmp > 0) {
237       Node newRightNode = remove(node.right, elem);
238       if (newRightNode == TOKEN) return TOKEN;
239       node.right = newRightNode;
240 
241       // Found the node we wish to remove.
242     } else {
243 
244       // This is the case with only a right subtree or no subtree at all.
245       // In this situation just swap the node we wish to remove
246       // with its right child.
247       if (node.left == null) {
248         return node.right;
249 
250         // This is the case with only a left subtree or
251         // no subtree at all. In this situation just
252         // swap the node we wish to remove with its left child.
253       } else if (node.right == null) {
254         return node.left;
255 
256         // When removing a node from a binary tree with two links the
257         // successor of the node being removed can either be the largest
258         // value in the left subtree or the smallest value in the right
259         // subtree. As a heuristic, I will remove from the subtree with
260         // the most nodes in hopes that this may help with balancing.
261       } else {
262 
263         // Choose to remove from left subtree
264         if (node.left.height > node.right.height) {
265 
266           // Swap the value of the successor into the node.
267           T successorValue = findMax(node.left);
268           node.value = successorValue;
269 
270           // Find the largest node in the left subtree.
271           Node replacement = remove(node.left, successorValue);
272           if (replacement == TOKEN) return TOKEN;
273           node.left = replacement;
274 
275         } else {
276 
277           // Swap the value of the successor into the node.
278           T successorValue = findMin(node.right);
279           node.value = successorValue;
280 
281           // Go into the right subtree and remove the leftmost node we
282           // found and swapped data with. This prevents us from having
283           // two nodes in our tree with the same value.
284           Node replacement = remove(node.right, successorValue);
285           if (replacement == TOKEN) return TOKEN;
286           node.right = replacement;
287         }
288       }
289     }
290 
291     // Update balance factor and height values.
292     update(node);
293 
294     // Re-balance tree.
295     return balance(node);
296   }
