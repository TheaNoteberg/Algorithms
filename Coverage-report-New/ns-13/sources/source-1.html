35   // The number of elements in the original input array.
36   private int n;
37 
38   // The maximum power of 2 needed. This value is floor(log2(n))
39   private int P;
40 
41   // Fast log base 2 logarithm lookup table, 1 <= i <= n
42   private int[] log2;
43 
44   // The sparse table values.
45   private long[][] dp;
46 
47   // Index Table (IT) associated with the values in the sparse table. This table
48   // is only useful when we want to query the index of the min (or max) element
49   // in the range [l, r] rather than the value itself. The index table doesnâ€™t
50   // make sense for most other range query types like gcd or sum.
51   private int[][] it;
52 
53   public MinSparseTable(long[] values) {
54     n = values.length;
55     P = (int) (Math.log(n) / Math.log(2));
56     dp = new long[P + 1][n];
57     it = new int[P + 1][n];
58 
59     for (int i = 0; i < n; i++) {
60       dp[0][i] = values[i];
61       it[0][i] = i;
62     }
63 
64     log2 = new int[n + 1];
65     for (int i = 2; i <= n; i++) {
66       log2[i] = log2[i / 2] + 1;
67     }
68 
69     // Build sparse table combining the values of the previous intervals.
70     for (int p = 1; p <= P; p++) {
71       for (int i = 0; i + (1 << p) <= n; i++) {
72         long leftInterval = dp[p - 1][i];
73         long rightInterval = dp[p - 1][i + (1 << (p - 1))];
74         dp[p][i] = Math.min(leftInterval, rightInterval);
75 
76         // Propagate the index of the best value
77         if (leftInterval <= rightInterval) {
78           it[p][i] = it[p - 1][i];
79         } else {
80           it[p][i] = it[p - 1][i + (1 << (p - 1))];
81         }
82       }
83     }
84   }

86   // Do a min query on the interval [l, r] in O(1).
87   //
88   // We can get O(1) query by finding the smallest power of 2 that fits within
89   // the interval length which we'll call k. Then we can query the intervals
90   // [l, l+k] and [r-k+1, r] (which likely overlap) and apply the function
91   // again. Some functions (like min and max) don't care about overlapping
92   // intervals so this trick works, but for a function like sum this would
93   // return the wrong result since it is not an idempotent binary function
94   // (aka an overlap friendly function).
95   private long queryMin(int l, int r) {
96     int length = r - l + 1;
97     int p = log2[length];
98     int k = 1 << p; // 2 to the power of p
99     return Math.min(dp[p][l], dp[p][r - k + 1]);
100   }

102   // Returns the index of the minimum element in the range [l, r].
103   public int queryMinIndex(int l, int r) {
104     int length = r - l + 1;
105     int p = log2[length];
106     int k = 1 << p; // 2 to the power of p
107     long leftInterval = dp[p][l];
108     long rightInterval = dp[p][r - k + 1];
109     if (leftInterval <= rightInterval) {
110       return it[p][l];
111     } else {
112       return it[p][r - k + 1];
113     }
114   }
