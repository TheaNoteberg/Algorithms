76   private T[] findMinRecursive(int dim, KDNode<T> curr, int axis) {
77     // If nothing is found, return nothing
78     if (curr == null) return null;
79     // If the axis and dimension match, follow typical search procedure
80     if (dim == axis) {
81       if (curr.left == null) return curr.point;
82       return findMinRecursive(dim, curr.left, (axis + 1) % k);
83     }
84     // If there are no children, return what you have
85     if (curr.left == null && curr.right == null) return curr.point;
86     // If there is at least one child, search the children of the current node
87     T[] leftSubTree = findMinRecursive(dim, curr.left, (axis + 1) % k);
88     T[] rightSubTree = findMinRecursive(dim, curr.right, (axis + 1) % k);
