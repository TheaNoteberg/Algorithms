28   static long[][] squareMatrixMult(long[][] m1, long[][] m2) {
29 
30     final int N = m1.length;
31     long[][] newMatrix = new long[N][N];
32 
33     for (int i = 0; i < N; i++)
34       for (int j = 0; j < N; j++)
35         for (int k = 0; k < N; k++)
36           // Overflow can happen here, watch out!
37           newMatrix[i][j] = newMatrix[i][j] + m1[i][k] * m2[k][j];
38

42   // Raise a matrix to the pth power. If p is negative
43   // return null and if p is zero return the identity.
44   // NOTE: Make sure the matrix is a square matrix and
45   // also watch out for overflow as the numbers climb quickly!
46   static long[][] matrixPower(long[][] matrix, long p) {
47 
48     if (p < 0) return null;
49 
50     final int N = matrix.length;
51     long[][] newMatrix = null;
52 
53     // Return identity matrix
54     if (p == 0) {
55       newMatrix = new long[N][N];
56       for (int i = 0; i < N; i++) newMatrix[i][i] = 1L;
57     } else {
58 
59       long[][] P = matrixDeepCopy(matrix);
60 
61       while (p > 0) {
62 
63         if ((p & 1L) == 1L) {
64           if (newMatrix == null) newMatrix = matrixDeepCopy(P);
65           else newMatrix = squareMatrixMult(newMatrix, P);
66         }
67 
68         // Repeatedly square P every loop, O(n^3)
69         P = squareMatrixMult(P, P);
70         p >>= 1L;
71       }
72     }
73

75   }

77   // Construct the transformation matrix
78   // http://fusharblog.com/solving-linear-recurrence-for-programming-contest/
79   static long[][] createTransformationMatrix(long[] coeffs, int size) {
80 
81     long T[][] = new long[size][size];
82     for (int i = 0; i + 1 < size; i++) T[i][i + 1] = 1L;
83     for (int i = 0; i < size - 1; i++) T[size - 2][i] = coeffs[coeffs.length - i - 1];
84     T[size - 1][size - 1] = T[size - 2][size - 1] = 1L;
85     return T;

89    * Solve for the nth term in a linear recurrence of the following form f(n) = k + c_1*f(n-1) +
90    * c_2*f(n-2) + ... + c_m*f(n-m) in O(m^3log(n)) time
91    *
92    * @param coefficients - contains the coefficients for the recurrence, so [c_1, c_2, c_3, ... ,
93    *     c_m]
94    * @param f_0 - The value of the function at f(0). This is usually 1 or k.
95    * @param k - The constant k added to the recurrence
96    * @param n - The nth term of the recurrence you wish to find
97    *     <p>NOTE1: The numbers produced by this method can get VERY LARGE quickly so watch out for
98    *     overflow because there is a very high probability it will occur.
99    *     <p>NOTE2: Make sure the dimension of coefficients array is no bigger than it needs to be.
100    *     For instance do not add any additional zero coefficient terms at the end of the
101    *     coefficients array as this is throw off the recurrence.
102    *     <p>EXAMPLE: If your recurrence is f(n) = 2 + 3*f(n-1) + 5f(n-4) with f(0) = 2 and you want
103    *     to find f(100) call the function like: solveRecurrence([3, 0, 0, 5], 2, 2, 100)
104    */
105   static long solveRecurrence(long[] coefficients, long f_0, long k, long n) {
106 
107     if (n < 0) throw new IllegalArgumentException("n should probably be >= 0");
108     long[] initialValues = computeInitialValues(coefficients, f_0, k);
109 
110     // We already know the value
111     if (n < initialValues.length) return initialValues[(int) n];
112 
113     // Add 1 to account for the extra constant 'k' in the recurrence: f(n) = k + c_1*f(n-1) + ...
114     final int size = initialValues.length + 1;
115 
116     long[][] T = createTransformationMatrix(coefficients, size);
117     long[][] result = matrixPower(T, n);
118 
119     // Find answer by multiplying resultant matrix with multiplication
120     // vector, that is the initial values appended with the constant k
121     long ans = 0L;
122     for (int j = 0; j < size; j++) {
123       if (j == size - 1) {
124         ans = ans + result[0][j] * k;
125       } else {
126         ans = ans + result[0][j] * initialValues[j];
127       }
128     }
129 
130     return ans;
131   }

134    * You may not always know what the initial values for your recurrence relation are, so sometimes
135    * it's useful to brute force them using dynamic programming.
136    *
137    * <p>Given the constants [c_1, c_2, c_3, ...] and the constant 'k' in the recurrence f(n) = k +
138    * c_1*f(n-1) + c_2*f(n-2) + ... + c_m*f(n-m) this function computes and returns the initial
139    * values of the function: [f(0), f(1), f(2), ...]
140    *
141    * @param coeffs - The coefficients on your linear recurrence
142    * @param f_0 - The value of the function at f(0), this is usually 1 or k
143    * @param k - The constant value in the linear recurrence
144    *     <p>NOTE: This method assumes that f(n) = 0 when n < 0
145    */
146   static long[] computeInitialValues(long[] coeffs, long f_0, long k) {
147 
148     final int N = coeffs.length;
149     long[] DP = new long[N];
150     DP[0] = f_0;
151 
152     for (int n = 1; n < N; n++) {
153       for (int i = 1; i <= n; i++) DP[n] += DP[n - i] * coeffs[i - 1];
154       DP[n] += k;
155     }
156 
157     return DP;
158   }
