1 /**
2  * SkipList is a data structure that is useful for dealing with dynamic sorted data. In particular
3  * it gives O(log(n)) average complexity of insertion, removal, and find operations. This
4  * implementation has been augmented with a method for determining the index of an element in the
5  * SkipList. Finding the index of an element is also O(log(n)) on average. The complexities are
6  * average complexities since this algorithm is dependent on randomisation to achieve nice balanced
7  * properties. To make this efficient, instantiate the SkipList with a height equal to, or just
8  * greater than, log(n) where n is the number of elements that will be in the list. On average, this
9  * data structure uses O(n) space. Worst case space is O(nlog(n)), and worst case for all other
10  * operations is O(n)

32   public SkipList(int height, int minValue, int maxValue) {
33     // Height goes from 0 to height-1
34     this.height = height;
35     head = new Node(minValue);
36     tail = new Node(maxValue);
37     Node currLeft = head;
38     Node currRight = tail;
39     // Setup links between all levels of head and tail nodes
40     for (int i = 1; i < height; i++) {
41       setHeadTail(height - i, currLeft, currRight);
42       // Create and setup down node
43       currLeft.down = new Node(currLeft.value);
44       currLeft.down.up = currLeft;
45       currRight.down = new Node(currRight.value);
46       currRight.down.up = currRight;
47       currLeft = currLeft.down;
48       currRight = currRight.down;
49     }
50     // Set last level
51     setHeadTail(0, currLeft, currRight);
52   }

81       if (node.down != null) return search(num, node.down);

87   /** Inserts the number into the list. */
88   public boolean insert(int num) {
89     if (num < head.value || num > tail.value) return false;
90     Node node = search(num);
91     if (node.value == num) return false;
92     int nodeHeight = 0;
93     // 0.5 probability of having height 1, 0.25 for height 2
94     while (rand.nextBoolean() && nodeHeight < (height - 1)) {
95       nodeHeight++;
96     }
97     insert(node, new Node(num), null, nodeHeight, node.index + 1);
98     return true;
99   }

102   private void insert(Node startNode, Node insertNode, Node lower, int insertHeight, int distance) {
103     if (startNode.height <= insertHeight) {
104       insertNode.left = startNode;
105       insertNode.right = startNode.right;
106       // If not at lowest level
107       if (lower != null) {
108         lower.up = insertNode;
109         insertNode.down = lower;
110       }
111       // If at lowest level, update ranks of following
112       else if (startNode.height == 0) {
113         increaseRank(insertNode.right);
114       }
115       startNode.right.left = insertNode;
116       startNode.right = insertNode;
117       insertNode.height = startNode.height;
118       insertNode.index = distance;
119       Node curr = startNode;
120       while (curr.up == null && curr.left != null) {
121         curr = curr.left;
122       }
123       if (curr.up != null) {
124         curr = curr.up;
125         insert(curr, new Node(insertNode.value), insertNode, insertHeight, distance);
126       }
