34   // Computes the polynomial product modulo p
35   public static long[] multiply(long[] x, long[] y) {
36 
37     // If the coefficients are negative place them in the range of [0, p)
38     for (int i = 0; i < x.length; i++) if (x[i] < 0) x[i] += p;
39     for (int i = 0; i < y.length; i++) if (y[i] < 0) y[i] += p;
40 
41     int zLength = x.length + y.length - 1;
42     int logN = 32 - Integer.numberOfLeadingZeros(zLength - 1);
43     long[] xx = transform(x, logN, false);
44     long[] yy = transform(y, logN, false);
45     long[] zz = new long[1 << logN];
46     for (int i = 0; i < zz.length; i++) zz[i] = mult(xx[i], yy[i]);
47     long[] nZ = transform(zz, logN, true);
48     long[] z = new long[zLength];
49     long nInverse = p - ((p - 1) >>> logN);
50     for (int i = 0; i < z.length; i++) {
51 
52       z[i] = mult(nInverse, nZ[i]);
53 
54       // Allow for negative coefficients. If you know the answer cannot be
55       // greater than 2^31-1 subtract p to obtain the negative coefficient.
56       if (z[i] >= Integer.MAX_VALUE) z[i] -= p;
57     }
58     return z;
59   }

61   private static long mult(long x, long y) {
62     long z = x * y;
63     if (z < 0) {
64       z = z % p + q;
65       return z < 0 ? z + p : z;
66     }
67     if (z < (1L << 56) && x > (1 << 28) && y > (1 << 28)) {
68       z = z % p + q;
69       return z < p ? z : z - p;
70     }
71     return z % p;
72   }

74   private static long[] transform(long[] v, int logN, boolean inverse) {
75     int n = 1 << logN;
76     long[] w = new long[n];
77     for (int i = 0; i < v.length; i++) w[Integer.reverse(i) >>> 32 - logN] = v[i];
78     for (int i = 0; i < logN; i++) {
79       int jMax = 1 << i;
80       int kStep = 2 << i;
81       int index = 0;
82       int step = 1 << exp - i - 1;
83       if (inverse) {
84         index = 1 << exp;
85         step = -step;
86       }
87       for (int j = 0; j < jMax; j++) {
88         long zeta = powers[index];
89         index += step;
90         for (int k = j; k < n; k += kStep) {
91           int kk = jMax | k;
92           long x = w[k];
93           long y = mult(zeta, w[kk]);
94           long z = x + y;
95           w[k] = z < p ? z : z - p;
96           z = x - y;
97           w[kk] = z < 0 ? z + p : z;
98         }
99       }
100     }
101     return w;
102   }
