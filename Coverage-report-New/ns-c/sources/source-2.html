127   // Perform bottom up node swim, O(log(n))
128   private void swim(int k) {
129 
130     // Grab the index of the next parent node WRT to k
131     int parent = (k - 1) / 2;
132 
133     // Keep swimming while we have not reached the
134     // root and while we're less than our parent.
135     while (k > 0 && less(k, parent)) {
136 
137       // Exchange k with the parent
138       swap(parent, k);
139       k = parent;
140 
141       // Grab the index of the next parent node WRT to k
142       parent = (k - 1) / 2;
143     }
144   }

146   // Top down node sink, O(log(n))
147   private void sink(int k) {
148     int heapSize = size();
149 
150     while (true) {
151 
152       int left = 2 * k + 1; // Left  node
153       int right = 2 * k + 2; // Right node
154       int smallest = left; // Assume left is the smallest node of the two children
155 
156       // Find which is smaller left or right
157       // If right is smaller set smallest to be right
158       if (right < heapSize && less(right, left)) smallest = right;
159 
160       // Stop if we're outside the bounds of the tree
161       // or stop early if we cannot sink k anymore
162       if (left >= heapSize || less(k, smallest)) break;
163 
164       // Move down the tree following the smallest node
165       swap(smallest, k);
166       k = smallest;
167     }
168   }

202   private T removeAt(int i) {
203 
204     if (isEmpty()) return null;
205 
206     int indexOfLastElem = size() - 1;
207     T removed_data = heap.get(i);
