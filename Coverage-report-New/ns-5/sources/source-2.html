20   // Doing 'n & 0x7F' is the same as modding by 64, but faster
21   private static final long MOD64_MASK = 0x7F;
22 
23   // Doing 'n >> 6' is the same as dividing by 64, but faster
24   private static final long DIV64_SHIFT = 6;

26   // Create a bloom filter with a various bitsets of different sizes
27   public BloomFilter(int[] bitSetSizes) {
28     N_SETS = bitSetSizes.length;
29     SET_SIZES = bitSetSizes.clone();
30     bitsets = new long[N_SETS][];
31     for (int i = 0; i < N_SETS; i++) {
32       bitsets[i] = new long[SET_SIZES[i]];
33     }
34   }
35 
36   // Add a hash value to one of the bitsets in the bloom filter
37   public void add(int setIndex, long hash) {
38     hash = hash % SET_SIZES[setIndex];
39     int block = (int) (hash >> DIV64_SHIFT);
40     bitsets[setIndex][block] |= (1L << (hash & MOD64_MASK));
41   }
42 
43   // Adds a group of related hash values to the bloom filter.
44   // These hash values should be the hash values that were applied
45   // to all the various hash functions on the same key.
46   public void add(long[] hashes) {
47     for (int i = 0; i < N_SETS; i++) {
48       add(i, hashes[i]);
49     }
50   }
51 
52   // Checks if a particular key is found within the bloom filter
53   public boolean contains(long[] hashes) {
54     for (int i = 0; i < hashes.length; i++) {
55       int block = (int) (hashes[i] >> DIV64_SHIFT);
56       long MASK = 1L << (hashes[i] & MOD64_MASK);
57       if ((bitsets[i][block] & MASK) != MASK) return false;
58     }
59     return true;
60   }
