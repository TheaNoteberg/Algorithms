1 /**
2  * The StringSet is a probabilistic string set data structure implemented using a bloom filter and
3  * positional rolling hashing techniques. This data structure can be very fast (outperforming Java
4  * HashSet) if you're doing queries involving all substrings of a particular string. Despite being
5  * probabilistic, this DS is very safe to use because the probability of a false positive can be set
6  * to as low as you wish it to be.
7  *
8  * @author William Alexandre Fiset, william.alexandre.fiset@gmail.com
9  */

14   // Our alphabet size is 95 because there are only 95 printable ASCII characters
15   // which are in the range between [32, 127). We also need to add +1 to our alphabet
16   // size because we're going to redefine the first ASCII character (the space character)
17   // to be 1 instead of 0 to avoid collisions where the string ' ' hashes to the
18   // same value as '   ' since 0*95^0 = 0*95^0 + 0*95^1 + 0*95^2
19   private static final int ALPHABET_SZ = 95 + 1;
20   private static final int[] ALPHABET = new int[127];
21 
22   private final int N_HASHES;
23   private final long POWERS[][];
24   private final int[] MODS, MOD_INVERSES;
25   private final long[] rollingHashes;
26   private final BloomFilter bloomFilter;

28   // More primes: 1009, 1013, 1019, 10007, 10009, 10037, 100003, 100019, 100043, 1000003, 1000033,
29   // 1000037,

45   // mods - The mod values to use for the bloom filter, they should probably be prime numbers
46   // maxLen - The maximum length string we will need to deal with
47   public StringSet(int[] mods, int maxLen) {
48 
49     MODS = mods.clone();
50     N_HASHES = mods.length;
51     MOD_INVERSES = new int[N_HASHES];
52     POWERS = new long[N_HASHES][maxLen];
53     rollingHashes = new long[N_HASHES];
54     bloomFilter = new BloomFilter(mods);
55 
56     java.math.BigInteger bigAlpha = new java.math.BigInteger(String.valueOf(ALPHABET_SZ));
57 
58     // Assuming all mods are primes each mod value will have a modular inverse
59     for (int i = 0; i < N_HASHES; i++) {
60       java.math.BigInteger mod = new java.math.BigInteger(String.valueOf(MODS[i]));
61       MOD_INVERSES[i] = bigAlpha.modInverse(mod).intValue();
62     }
63 
64     // Precompute powers of the alphabet size mod all the mod values
65     for (int i = 0; i < N_HASHES; i++) {
66       POWERS[i][0] = 1L;
67       for (int j = 1; j < maxLen; j++) {
68         POWERS[i][j] = (POWERS[i][j - 1] * ALPHABET_SZ) % mods[i];
69       }
70     }
71   }

73   // Returns a shallow copy of the current rolling hash value.
74   // Make sure you clone the array if you want to cache it.
75   public long[] computeHash(String str) {
76 
77     java.util.Arrays.fill(rollingHashes, 0L);
78 
79     for (int i = 0; i < str.length(); i++) {
80       for (int k = 0; k < N_HASHES; k++) {
81         int rightChar = ALPHABET[str.charAt(i)];
82         rollingHashes[k] = addRight(rollingHashes[k], rightChar, k);
83       }
84     }
85 
86     return rollingHashes; // rollingHashes.clone();
87   }

98   // Add all sequences of length 'sz' to the bloom filter
99   public void add(String str, int sz) {
100 
101     add(str);
102 
103     for (int i = sz; i < str.length() - sz; i++) {
104       for (int k = 0; k < N_HASHES; k++) {
105 
106         int rightChar = ALPHABET[str.charAt(i)];
107         int leftChar = ALPHABET[str.charAt(i - sz)];
108 
109         // Add the right character
110         rollingHashes[k] = addRight(rollingHashes[k], rightChar, k);
111 
112         // Remove the leftmost character
113         rollingHashes[k] = removeLeft(rollingHashes[k], leftChar, k, sz);
114       }
115       bloomFilter.add(rollingHashes);
116     }
117   }

119   // Adds all the substrings of 'data' into the
120   // bloom filter using positional hashing
121   public void addAllSubstrings(String str) {

124     int[] values = new int[N];
125

127 
128     for (int i = 0; i < N; i++) {

131       java.util.Arrays.fill(rollingHashes, 0L);
132 
133       for (int j = i; j < N; j++) {
134 
135         // Compute the next rolling hash value for each hash
136         // function with a different modulus value
137         for (int k = 0; k < N_HASHES; k++) {
138           rollingHashes[k] = addRight(rollingHashes[k], values[j], k);
139         }
140

143       }

167   // Given a rolling hash x_n*A^n + x_n-1*A^(n-1) + ... + x_2*A^2 + x_1*A^1 + x_0*A^0
168   // where x_i is a string character value and 'A' is the alphabet size we
169   // want to remove the first term 'x_n*A^n' from our rolling hash.
170   //
171   // firstValue - This is x_n, the first character of this string
172   public long removeLeft(long rollingHash, int firstValue, int modIndex, int len) {
173     rollingHash = (rollingHash - firstValue * POWERS[modIndex][len - 1]) % MODS[modIndex];
174     return (rollingHash + MODS[modIndex]) % MODS[modIndex];
175   }

182     rollingHash = (((rollingHash - lastValue) % MODS[modIndex]) + MODS[modIndex]) % MODS[modIndex];
183     return (rollingHash * MOD_INVERSES[modIndex]) % MODS[modIndex];
184   }
