265    * Builds a segment tree by starting with the leaf nodes and combining segment values on callback.
266    *
267    * @param i the index of the segment in the segment tree
268    * @param tl the left index (inclusive) of the segment range
269    * @param tr the right index (inclusive) of the segment range
270    * @param values the initial values array
271    */
272   private void buildSegmentTree(int i, int tl, int tr, long[] values) {
273     if (tl == tr) {
274       st[i] = new Segment(i, values[tl], values[tl], values[tl], tl, tr);
275       return;
276     }
277     int tm = (tl + tr) / 2;
278     buildSegmentTree(2 * i + 1, tl, tm, values);
279     buildSegmentTree(2 * i + 2, tm + 1, tr, values);
280 
281     Long segmentValue = combinationFn.apply(st[2 * i + 1].value, st[2 * i + 2].value);
282     Long minValue = Math.min(st[2 * i + 1].min, st[2 * i + 2].min);
283     Long maxValue = Math.max(st[2 * i + 1].max, st[2 * i + 2].max);
284     Segment segment = new Segment(i, segmentValue, minValue, maxValue, tl, tr);
285 
286     st[i] = segment;
287   }

300    * Returns the range query value of the range [l, r]
301    *
302    * @param i the index of the current segment in the tree
303    * @param tl the left endpoint (inclusive) of the current segment
304    * @param tr the right endpoint (inclusive) of the current segment
305    * @param l the target left endpoint (inclusive) for the range query
306    * @param r the target right endpoint (inclusive) for the range query
307    */
308   private Long rangeQuery1(int i, int tl, int tr, int l, int r) {
309     // Different segment tree types have different base cases
310     if (l > r) {
311       return null;
312     }
313     propagate1(i, tl, tr);
314     if (tl == l && tr == r) {
315       return st[i].value;
316     }
317     int tm = (tl + tr) / 2;
318     // Instead of checking if [tl, tm] overlaps [l, r] and [tm+1, tr] overlaps
319     // [l, r], simply recurse on both segments and let the base case return the
320     // default value for invalid intervals.
321     return combinationFn.apply(
322         rangeQuery1(2 * i + 1, tl, tm, l, Math.min(tm, r)),
323         rangeQuery1(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r));
324   }

326   // Apply the delta value to the current node and push it to the child segments
327   private void propagate1(int i, int tl, int tr) {
328     if (st[i].lazy != null) {
329       // Only used for min/max mul queries
330       st[i].min = st[i].min * st[i].lazy;
331       st[i].max = st[i].max * st[i].lazy;
332 
333       // Apply the delta to the current segment.
334       st[i].value = ruf.apply(st[i], st[i].lazy);
335       // Push the delta to left/right segments for non-leaf nodes
336       propagateLazy1(i, tl, tr, st[i].lazy);
337       st[i].lazy = null;
338     }
339   }

341   private void propagateLazy1(int i, int tl, int tr, long delta) {
342     // Ignore leaf segments
343     if (tl == tr) return;
344     st[2 * i + 1].lazy = lruf.apply(st[2 * i + 1], delta);
345     st[2 * i + 2].lazy = lruf.apply(st[2 * i + 2], delta);
346   }

352   private void rangeUpdate1(int i, int tl, int tr, int l, int r, long x) {
353     propagate1(i, tl, tr);
354     if (l > r) {
355       return;
356     }
357 
358     if (tl == l && tr == r) {
359       // Only used for min/max mul queries
360       st[i].min = st[i].min * x;
361       st[i].max = st[i].max * x;
362 
363       st[i].value = ruf.apply(st[i], x);
364       propagateLazy1(i, tl, tr, x);
365     } else {
366       int tm = (tl + tr) / 2;
367       // Instead of checking if [tl, tm] overlaps [l, r] and [tm+1, tr] overlaps
368       // [l, r], simply recurse on both segments and let the base case disregard
369       // invalid intervals.
370       rangeUpdate1(2 * i + 1, tl, tm, l, Math.min(tm, r), x);
371       rangeUpdate1(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r, x);
372 
373       st[i].value = combinationFn.apply(st[2 * i + 1].value, st[2 * i + 2].value);
374       st[i].max = Math.max(st[2 * i + 1].max, st[2 * i + 2].max);
375       st[i].min = Math.min(st[2 * i + 1].min, st[2 * i + 2].min);
376     }
377   }
