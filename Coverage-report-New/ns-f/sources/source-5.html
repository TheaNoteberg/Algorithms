27     if (a == null && b == null) return null;
28     if (a == null) return b;

63    * Builds a segment tree by starting with the leaf nodes and combining segment values on callback.
64    *
65    * @param i the index of the segment in the segment tree
66    * @param tl the left index (inclusive) of the segment range
67    * @param tr the right index (inclusive) of the segment range
68    * @param values the initial values array
69    */
70   private void buildSegmentTree(int i, int tl, int tr, long[] values) {
71     if (tl == tr) {
72       t[i] = values[tl];
73       return;
74     }
75     int tm = (tl + tr) / 2;
76     buildSegmentTree(2 * i + 1, tl, tm, values);
77     buildSegmentTree(2 * i + 2, tm + 1, tr, values);
78 
79     t[i] = maxFunction(t[2 * i + 1], t[2 * i + 2]);
80   }

99    * @param r the target right endpoint (inclusive) for the range query

101   private Long rangeQuery1(int i, int tl, int tr, int l, int r) {
102     if (l > r) {
103       return null;
104     }
105     propagate1(i, tl, tr);
106     if (tl == l && tr == r) {
107       return t[i];
108     }
109     int tm = (tl + tr) / 2;
110     // Instead of checking if [tl, tm] overlaps [l, r] and [tm+1, tr] overlaps
111     // [l, r], simply recurse on both segments and let the base case return the
112     // default value for invalid intervals.
113     return maxFunction(
114         rangeQuery1(2 * i + 1, tl, tm, l, Math.min(tm, r)),
115         rangeQuery1(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r));
116   }

122   private void propagateLazy(int i, int tl, int tr, long delta) {
123     // Ignore leaf segments
124     if (tl == tr) return;
125     // TODO(william): should this also used the minSegmentUpdateFn
126     lazy[2 * i + 1] = sumFunction(lazy[2 * i + 1], delta);

130   private void propagate1(int i, int tl, int tr) {
131     // Check for default value because you don't want to assign to the lazy
132     // value if it's the default value.
133     if (lazy[i] != null) {
134       // The minimum value increases by the delta over the whole range.
135       t[i] = minSegmentUpdateFn(t[i], /*unused*/ 0, /*unused*/ 0, lazy[i]);
136       // Push delta to left/right segments for non-leaf nodes
137       propagateLazy(i, tl, tr, lazy[i]);
138       lazy[i] = null;
139     }
140   }

143     propagate1(i, tl, tr);
144     if (l > r) {
145       return;
146     }
147

151       // TODO(william): confirm if this is needed if we already propagated?
152       lazy[i] = null;
153     } else {
