263    * Builds a segment tree by starting with the leaf nodes and combining segment values on callback.
264    *
265    * @param i the index of the segment in the segment tree
266    * @param tl the left index (inclusive) of the segment range
267    * @param tr the right index (inclusive) of the segment range
268    * @param values the initial values array
269    */
270   private void buildSegmentTree(int i, int tl, int tr, long[] values) {
271     if (tl == tr) {
272       t[i] = values[tl];
273       return;
274     }
275     int tm = (tl + tr) / 2;
276     buildSegmentTree(2 * i + 1, tl, tm, values);
277     buildSegmentTree(2 * i + 2, tm + 1, tr, values);
278 
279     t[i] = combinationFn.apply(t[2 * i + 1], t[2 * i + 2]);
280   }

292   /**
293    * Returns the range query value of the range [l, r]
294    *
295    * @param i the index of the current segment in the tree
296    * @param tl the left endpoint (inclusive) of the current segment
297    * @param tr the right endpoint (inclusive) of the current segment
298    * @param l the target left endpoint (inclusive) for the range query
299    * @param r the target right endpoint (inclusive) for the range query
300    */
301   private Long rangeQuery1(int i, int tl, int tr, int l, int r) {
302     // Different segment tree types have different base cases
303     if (l > r) {
304       return null;
305     }
306     propagate1(i, tl, tr);
307     if (tl == l && tr == r) {
308       return t[i];
309     }
310     int tm = (tl + tr) / 2;
311     // Instead of checking if [tl, tm] overlaps [l, r] and [tm+1, tr] overlaps
312     // [l, r], simply recurse on both segments and let the base case return the
313     // default value for invalid intervals.
314     return combinationFn.apply(
315         rangeQuery1(2 * i + 1, tl, tm, l, Math.min(tm, r)),
316         rangeQuery1(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r));
317   }

319   // Apply the lazy delta value to the current node and push it to the child segments
320   private void propagate1(int i, int tl, int tr) {
321     // No lazy value to propagate
322     if (lazy[i] == null) {
323       return;
324     }
325     // Apply the lazy delta to the current segment.
326     t[i] = ruf.apply(t[i], tl, tr, lazy[i]);
327     // Push the lazy delta to left/right segments for non-leaf nodes
328     propagateLazy1(i, tl, tr, lazy[i]);
329     lazy[i] = null;
330   }
331 
332   private void propagateLazy1(int i, int tl, int tr, long delta) {
333     // Ignore leaf segments
334     if (tl == tr) return;
335     lazy[2 * i + 1] = lruf.apply(lazy[2 * i + 1], tl, tr, delta);
336     lazy[2 * i + 2] = lruf.apply(lazy[2 * i + 2], tl, tr, delta);
337   }

343   private void rangeUpdate1(int i, int tl, int tr, int l, int r, long x) {
344     propagate1(i, tl, tr);
345     if (l > r) {
346       return;
347     }
348 
349     if (tl == l && tr == r) {
350       t[i] = ruf.apply(t[i], tl, tr, x);
351       propagateLazy1(i, tl, tr, x);
352     } else {
353       int tm = (tl + tr) / 2;
354       // Instead of checking if [tl, tm] overlaps [l, r] and [tm+1, tr] overlaps
355       // [l, r], simply recurse on both segments and let the base case disregard
356       // invalid intervals.
357       rangeUpdate1(2 * i + 1, tl, tm, l, Math.min(tm, r), x);
358       rangeUpdate1(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r, x);
359 
360       t[i] = combinationFn.apply(t[2 * i + 1], t[2 * i + 2]);
361     }
362   }
