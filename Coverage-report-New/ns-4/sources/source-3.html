12  * Standard Splay Tree Implementation, supports generic data(must implement Comparable)
13  *
14  * <p>The Basic Concept of SplayTree is to keep frequently used nodes close to the root of the tree
15  * It performs basic operations such as insertion,search,delete,findMin,findMax in O(log n)
16  * amortized time Having frequently-used nodes near to the root can be useful in implementing many
17  * algorithms. e.g: Implementing caches, garbage collection algorithms etc Primary disadvantage of
18  * the splay tree can be the fact that its height can go linear. This causes the worst case running
19  * times to go O(n) However, the amortized costs of this worst case situation is logarithmic, O(log
20  * n)
21  *
22  * @author Ashiqur Rahman,https://github.com/ashiqursuperfly
23  */

198   private BinaryTree<T> splayUtil(BinaryTree<T> root, T key) {
199     if (root == null || root.getData() == key) return root;
200 
201     if (root.getData().compareTo(key) > 0) {
202       if (root.getLeft() == null) return root;
203 
204       if (root.getLeft().getData().compareTo(key) > 0) {
205 
206         root.getLeft().setLeft(splayUtil(root.getLeft().getLeft(), key));
207 
208         root = rightRotate(root);
209       } else if (root.getLeft().getData().compareTo(key) < 0) {
210 
211         root.getLeft().setRight(splayUtil(root.getLeft().getRight(), key));
212 
213         if (root.getLeft().getRight() != null) root.setLeft(leftRotate(root.getLeft()));
214       }
215       return (root.getLeft() == null) ? root : rightRotate(root);
216     } else {
217       if (root.getRight() == null) return root;
218 
219       if (root.getRight().getData().compareTo(key) > 0) {
220         root.getRight().setLeft(splayUtil(root.getRight().getLeft(), key));
221         if (root.getRight().getLeft() != null) root.setRight(rightRotate(root.getRight()));
222       } else if (root.getRight().getData().compareTo(key) < 0) // Zag-Zag (Right Right)
223       {
224         root.getRight().setRight(splayUtil(root.getRight().getRight(), key));
225         root = leftRotate(root);
226       }
227 
228       return (root.getRight() == null) ? root : leftRotate(root);
229     }
230   }
