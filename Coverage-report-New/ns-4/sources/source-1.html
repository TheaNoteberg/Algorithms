93

99     // for is smaller than the current value
100     if (cmp < 0) {

102 
103       // Dig into right subtree, the value we're looking
104       // for is greater than the current value
105     } else if (cmp > 0) {
106       node.right = remove(node.right, elem);
107 
108       // Found the node we wish to remove
109     } else {

112       // no subtree at all. In this situation just
113       // swap the node we wish to remove with its right child.
114       if (node.left == null) {

116 
117         // This is the case with only a left subtree or
118         // no subtree at all. In this situation just
119         // swap the node we wish to remove with its left child.
120       } else if (node.right == null) {

123

125         // successor of the node being removed can either be the largest
126         // value in the left subtree or the smallest value in the right
127         // subtree. In this implementation I have decided to find the
128         // smallest value in the right subtree which can be found by
129         // traversing as far left as possible in the right subtree.
130       } else {
