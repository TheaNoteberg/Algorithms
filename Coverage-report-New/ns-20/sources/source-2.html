143   private static class DinicsSolver extends NetworkFlowSolverBase {
144 
145     private int[] level;
146 
147     /**
148      * Creates an instance of a flow network solver. Use the {@link #addEdge} method to add edges to
149      * the graph.
150      *
151      * @param n - The number of nodes in the graph including source and sink nodes.
152      * @param s - The index of the source node, 0 <= s < n
153      * @param t - The index of the sink node, 0 <= t < n, t != s
154      */
155     public DinicsSolver(int n, int s, int t) {
156       super(n, s, t);
157       level = new int[n];
158     }
159 
160     @Override
161     public void solve() {
162       // next[i] indicates the next edge index to take in the adjacency list for node i. This is
163       // part
164       // of the Shimon Even and Alon Itai optimization of pruning deads ends as part of the DFS
165       // phase.
166       int[] next = new int[n];
167 
168       while (bfs()) {
169         Arrays.fill(next, 0);
170         // Find max flow by adding all augmenting path flows.
171         for (long f = dfs(s, next, INF); f != 0; f = dfs(s, next, INF)) {
172           maxFlow += f;
173         }
174       }
175     }
176 
177     // Do a BFS from source to sink and compute the depth/level of each node
178     // which is the minimum number of edges from that node to the source.
179     private boolean bfs() {
180       Arrays.fill(level, -1);
181       Deque<Integer> q = new ArrayDeque<>(n);
182       q.offer(s);
183       level[s] = 0;
184       while (!q.isEmpty()) {
185         int node = q.poll();
186         for (Edge edge : graph[node]) {
187           long cap = edge.remainingCapacity();
188           if (cap > 0 && level[edge.to] == -1) {
189             level[edge.to] = level[node] + 1;
190             q.offer(edge.to);
191           }
192         }
193       }
194       // Return whether we were able to reach the sink node.
195       return level[t] != -1;
196     }
197 
198     private long dfs(int at, int[] next, long flow) {
199       if (at == t) return flow;
200       final int numEdges = graph[at].size();
201 
202       for (; next[at] < numEdges; next[at]++) {
203         Edge edge = graph[at].get(next[at]);
204         long cap = edge.remainingCapacity();
205         if (cap > 0 && level[edge.to] == level[at] + 1) {
206 
207           long bottleNeck = dfs(edge.to, next, min(flow, cap));
208           if (bottleNeck > 0) {
209             edge.augment(bottleNeck);
210             return bottleNeck;
211           }
212         }
213       }
214       return 0;
215     }
216   }
