69   // Do dfs to construct rooted tree.
70   private static TreeNode buildTree(List<List<Integer>> graph, TreeNode node) {
71     for (int childId : graph.get(node.id())) {
72       // Ignore adding an edge pointing back to parent.
73       if (node.parent() != null && childId == node.parent().id()) {
74         continue;
75       }
76 
77       TreeNode child = new TreeNode(childId, node);
78       node.addChildren(child);
79 
80       buildTree(graph, child);
81     }
82     return node;
83   }
