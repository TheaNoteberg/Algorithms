216     nodeOrder[tourIndex] = node;

223   public TreeNode lca(int index1, int index2) {
224     int l = Math.min(last[index1], last[index2]);
225     int r = Math.max(last[index1], last[index2]);

227     return nodeOrder[i];

231   private static class MinSparseTable {

234     private int n;

237     private int P;

252     private void init(long[] v) {
253       n = v.length;
254       P = (int) (Math.log(n) / Math.log(2));
255       dp = new long[P + 1][n];
256       it = new int[P + 1][n];
257 
258       for (int i = 0; i < n; i++) {
259         dp[0][i] = v[i];
260         it[0][i] = i;
261       }
262 
263       log2 = new int[n + 1];
264       for (int i = 2; i <= n; i++) {
265         log2[i] = log2[i / 2] + 1;
266       }
267 
268       // Build sparse table combining the values of the previous intervals.
269       for (int p = 1; p <= P; p++) {
270         for (int i = 0; i + (1 << p) <= n; i++) {
271           long leftInterval = dp[p - 1][i];
272           long rightInterval = dp[p - 1][i + (1 << (p - 1))];
273           dp[p][i] = Math.min(leftInterval, rightInterval);
274 
275           // Propagate the index of the best value
276           if (leftInterval <= rightInterval) {
277             it[p][i] = it[p - 1][i];
278           } else {
279             it[p][i] = it[p - 1][i + (1 << (p - 1))];
280           }
281         }
282       }
283     }

285     // Returns the index of the minimum element in the range [l, r].
286     public int queryIndex(int l, int r) {
287       int len = r - l + 1;
288       int p = log2[r - l + 1];
289       long leftInterval = dp[p][l];
290       long rightInterval = dp[p][r - (1 << p) + 1];
291       if (leftInterval <= rightInterval) {
292         return it[p][l];
293       } else {
294         return it[p][r - (1 << p) + 1];
295       }
296     }
