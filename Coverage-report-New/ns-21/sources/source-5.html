28   private static DfsResult dfs(
29       List<List<Integer>> graph, boolean[] visited, int[] prev, int at, int parent) {
30 
31     // Already visited this node
32     if (visited[at]) return new DfsResult(0, parent);
33 
34     // Visit this node
35     visited[at] = true;
36 
37     // Remember where we came from to rebuild path later on.
38     prev[at] = parent;
39 
40     int bestDist = 0, index = -1;
41     List<Integer> edges = graph.get(at);
42 
43     for (int to : edges) {
44       DfsResult result = dfs(graph, visited, prev, to, at);
45       int dist = result.distance + 1;
46       if (dist > bestDist) {
47         bestDist = dist;
48         index = result.index;
49       }
50     }
51 
52     return new DfsResult(bestDist, index);

55   public static List<Integer> findTreeCenters(List<List<Integer>> graph) {
56     List<Integer> centers = new ArrayList<>();
57     if (graph == null) return centers;
58 
59     int n = graph.size();
60     boolean[] visited = new boolean[n];
61     int[] prev = new int[n];
62 
63     // Do DFS to find furthest node from the start
64     DfsResult result = dfs(graph, visited, prev, 0, -1);
65     int furthestNode1 = result.index;
66 
67     // Singleton
68     if (furthestNode1 == -1) {
69       centers.add(0);
70       return centers;
71     }
72 
73     // Do another DFS, but this time from the furthest node.
74     Arrays.fill(visited, false);
75     Arrays.fill(prev, 0);
76 
77     result = dfs(graph, visited, prev, furthestNode1, -1);
78     int furthestNode2 = result.index;
79 
80     List<Integer> path = new LinkedList<>();
81     for (int i = furthestNode2; i != -1; i = prev[i]) {
82       path.add(i);
83     }
84 
85     if (path.size() % 2 == 0) {
86       centers.add(path.get(path.size() / 2 - 1));
87     }
88     centers.add(path.get(path.size() / 2));
89     return centers;
90   }
