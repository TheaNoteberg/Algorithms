17   public static List<Integer> findTreeCenters(List<List<Integer>> tree) {
18     final int n = tree.size();
19     int[] degree = new int[n];
20 
21     // Find all leaf nodes
22     List<Integer> leaves = new ArrayList<>();
23     for (int i = 0; i < n; i++) {
24       List<Integer> edges = tree.get(i);
25       degree[i] = edges.size();
26       if (degree[i] <= 1) {
27         leaves.add(i);
28         degree[i] = 0;
29       }
30     }
31 
32     int processedLeafs = leaves.size();
33 
34     // Remove leaf nodes and decrease the degree of each node adding new leaf nodes progressively
35     // until only the centers remain.
36     while (processedLeafs < n) {
37       List<Integer> newLeaves = new ArrayList<>();
38       for (int node : leaves) {
39         for (int neighbor : tree.get(node)) {
40           if (--degree[neighbor] == 1) {
41             newLeaves.add(neighbor);
42           }
43         }
44         degree[node] = 0;
45       }
46       processedLeafs += newLeaves.size();
47       leaves = newLeaves;
48     }
49 
50     return leaves;
51   }
