32   private static List<Integer> findTreeCenters(List<List<Integer>> tree) {
33     final int n = tree.size();
34     int[] degrees = new int[n];
35 
36     // Find all leaf nodes
37     List<Integer> leaves = new ArrayList<>();
38     for (int i = 0; i < n; i++) {
39       List<Integer> edges = tree.get(i);
40       degrees[i] = edges.size();
41       if (degrees[i] <= 1) leaves.add(i);
42     }
43 
44     int processedLeafs = leaves.size();
45 
46     // Remove leaf nodes and decrease the degree of
47     // each node adding new leaf nodes progressively
48     // until only the centers remain.
49     while (processedLeafs < n) {
50       List<Integer> newLeaves = new ArrayList<>();
51       for (int node : leaves)
52         for (int neighbor : tree.get(node)) if (--degrees[neighbor] == 1) newLeaves.add(neighbor);
53       processedLeafs += newLeaves.size();
54       leaves = newLeaves;
55     }
56 
57     return leaves;
58   }

60   // Encodes a tree as a string such that any isomorphic tree
61   // also has the same encoding.
62   // TODO(william): make this method private and test only with the treesAreIsomorphic method
63   public static String encodeTree(List<List<Integer>> tree) {
64     if (tree == null || tree.size() == 0) return "";
65     if (tree.size() == 1) return "()";
66     final int n = tree.size();
67 
68     int root = findTreeCenters(tree).get(0);
69 
70     int[] degree = new int[n];
71     int[] parent = new int[n];
72     boolean[] visited = new boolean[n];
73     List<Integer> leafs = new ArrayList<>();
74 
75     Queue<Integer> q = new ArrayDeque<>();
76     visited[root] = true;
77     parent[root] = -1; // unused.
78     q.offer(root);
79 
80     // Do a BFS to find all the leaf nodes
81     while (!q.isEmpty()) {
82       int at = q.poll();
83       List<Integer> edges = tree.get(at);
84       degree[at] = edges.size();
85       for (int next : edges) {
86         if (!visited[next]) {
87           visited[next] = true;
88           parent[next] = at;
89           q.offer(next);
90         }
91       }
92       if (degree[at] == 1) leafs.add(at);
93     }
94 
95     List<Integer> newLeafs = new ArrayList<>();
96     String[] map = new String[n];
97     for (int i = 0; i < n; i++) {
98       visited[i] = false;
99       map[i] = "()";
100     }
101 
102     int treeSize = n;
103     while (treeSize > 2) {
104       for (int leaf : leafs) {
105 
106         // Find parent of leaf node and check if the parent
107         // is a candidate for the next cycle of leaf nodes
108         visited[leaf] = true;
109         int p = parent[leaf];
110         if (--degree[p] == 1) newLeafs.add(p);
111 
112         treeSize--;
113       }
114 
115       // Update parent labels
116       for (int p : newLeafs) {
117 
118         List<String> labels = new ArrayList<>();
119         for (int child : tree.get(p))
120           // Recall edges are bidirectional so we don't want to
121           // access the parent's parent here.
122           if (visited[child]) labels.add(map[child]);
123 
124         String parentInnerParentheses = map[p].substring(1, map[p].length() - 1);
125         labels.add(parentInnerParentheses);
126 
127         Collections.sort(labels);
128         map[p] = "(".concat(String.join("", labels)).concat(")");
129       }
130 
131       leafs.clear();
132       leafs.addAll(newLeafs);
133       newLeafs.clear();
134     }
135 
136     // Only one node remains and it holds the canonical form
137     String l1 = map[leafs.get(0)];
138     if (treeSize == 1) return l1;
139 
140     // Two nodes remain and we need to combine their labels
141     String l2 = map[leafs.get(1)];
142     return ((l1.compareTo(l2) < 0) ? (l1 + l2) : (l2 + l1));
143   }
