29   public static int f(int k, int r) {
30     // Compute the optimal value of ending at the top LEFT and the top RIGHT of
31     // the gallery and return the minimum.
32     return min(f(k, r, LEFT), f(k, r, RIGHT));

35   // f(k,r,c) Computes the minimum value you can save by closing off `k` rooms
36   // in a gallery with `r` levels starting on side `c`.
37   //
38   // k = The number of rooms the curator needs to close
39   // r = The gallery row index
40   // c = The column, either LEFT (= 0) or RIGHT (= 1)
41   public static int f(int k, int r, int c) {
42     // We finished closing all K rooms
43     if (k == 0) {
44       return 0;
45     }
46     if (r < 0) {
47       return INF;
48     }
49     // Return the value of this subproblem, if it's already been computed.
50     // if (dp[k][r][c] != null) {
51     //   return dp[k][r][c];
52     // }
53     // Get the value of the current room at row `r` and column `c`.
54     int roomValue = gallery[r][c];
55     return dp[k][r][c] =
56         min(
57             // Close the current room, and take the best value from the partial
58             // state directly below the current room.
59             f(k - 1, r - 1, c) + roomValue,
60             // Don't include the current room. Instead, take the last best value from
61             // the previously calculated partial state which includes `k` rooms closed.
62             f(k, r - 1));
63   }
