173   // exists inside the hash-table then the value is updated.
174   public V insert(K key, V val) {
175     if (key == null) throw new IllegalArgumentException("Null key");
176     if (usedBuckets >= threshold) resizeTable();
177 
178     setupProbing(key);
179     final int offset = normalizeIndex(key.hashCode());
180 
181     for (int i = offset, j = -1, x = 1; ; i = normalizeIndex(offset + probe(x++))) {
182 
183       // The current slot was previously deleted
184       if (keys[i] == TOMBSTONE) {
185         if (j == -1) j = i;
186 
187         // The current cell already contains a key
188       } else if (keys[i] != null) {
189         // The key we're trying to insert already exists in the hash-table,
190         // so update its value with the most recent value
191         if (keys[i].equals(key)) {
192 
193           V oldValue = values[i];
194           if (j == -1) {
195             values[i] = val;
196           } else {
197             keys[i] = TOMBSTONE;
198             values[i] = null;
199             keys[j] = key;
200             values[j] = val;
201           }
202           modificationCount++;

204         }
205 
206         // Current cell is null so an insertion/update can occur
207       } else {

209         if (j == -1) {
210           usedBuckets++;

216           // the new element at i where the null element is insert

230   // Returns true/false on whether a given key exists within the hash-table
231   public boolean hasKey(K key) {
232     if (key == null) throw new IllegalArgumentException("Null key");
233 
234     setupProbing(key);
235     final int offset = normalizeIndex(key.hashCode());
236 
237     // Start at the original hash value and probe until we find a spot where our key
238     // is or hit a null element in which case our element does not exist.
239     for (int i = offset, j = -1, x = 1; ; i = normalizeIndex(offset + probe(x++))) {
240 
241       // Ignore deleted cells, but record where the first index
242       // of a deleted cell is found to perform lazy relocation later.
243       if (keys[i] == TOMBSTONE) {
244 
245         if (j == -1) j = i;
246 
247         // We hit a non-null key, perhaps it's the one we're looking for.
248       } else if (keys[i] != null) {
249 
250         // The key we want is in the hash-table!
251         if (keys[i].equals(key)) {
252 
253           // If j != -1 this means we previously encountered a deleted cell.
254           // We can perform an optimization by swapping the entries in cells
255           // i and j so that the next time we search for this key it will be
256           // found faster. This is called lazy deletion/relocation.
257           if (j != -1) {
258             // Swap the key-value pairs of positions i and j.
259             keys[j] = keys[i];
260             values[j] = values[i];
261             keys[i] = TOMBSTONE;
262             values[i] = null;
263           }
264           return true;
265         }
266 
267         // Key was not found in the hash-table :/
268       } else return false;
269     }
