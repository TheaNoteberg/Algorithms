2  * This file shows you how to use a suffix array to determine if a pattern exists within a text.
3  * This implementation has the advantage that once the suffix array is built queries can be very
4  * fast.
5  *

56     private void construct() {
57       int i, p, r;
58       for (i = 0; i < N; ++i) c[rank[i] = T[i]]++;
59       for (i = 1; i < ALPHABET_SZ; ++i) c[i] += c[i - 1];
60       for (i = N - 1; i >= 0; --i) sa[--c[T[i]]] = i;
61       for (p = 1; p < N; p <<= 1) {

63         for (i = 0; i < N; ++i) if (sa[i] >= p) sa2[r++] = sa[i] - p;
64         Arrays.fill(c, 0, ALPHABET_SZ, 0);
65         for (i = 0; i < N; ++i) c[rank[i]]++;
66         for (i = 1; i < ALPHABET_SZ; ++i) c[i] += c[i - 1];
67         for (i = N - 1; i >= 0; --i) sa[--c[rank[sa2[i]]]] = sa2[i];
68         for (sa2[sa[0]] = r = 0, i = 1; i < N; ++i) {
