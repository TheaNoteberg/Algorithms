237     // Construct a suffix array in O(nlog^2(n))
238     @Override
239     protected void construct() {
240       sa = new int[N];
241 
242       // Maintain suffix ranks in both a matrix with two rows containing the
243       // current and last rank information as well as some sortable rank objects
244       int[][] suffixRanks = new int[2][N];
245       SuffixRankTuple[] ranks = new SuffixRankTuple[N];
246 
247       // Assign a numerical value to each character in the text
248       for (int i = 0; i < N; i++) {
249         suffixRanks[0][i] = T[i];
250         ranks[i] = new SuffixRankTuple();
251       }
252 
253       // O(log(n))
254       for (int pos = 1; pos < N; pos *= 2) {
255         for (int i = 0; i < N; i++) {
256           SuffixRankTuple suffixRank = ranks[i];
257           suffixRank.firstHalf = suffixRanks[0][i];
258           suffixRank.secondHalf = i + pos < N ? suffixRanks[0][i + pos] : -1;
259           suffixRank.originalIndex = i;
260         }
261 
262         // O(nlog(n))
263         java.util.Arrays.sort(ranks);
264 
265         int newRank = 0;
266         suffixRanks[1][ranks[0].originalIndex] = 0;
267 
268         for (int i = 1; i < N; i++) {
269           SuffixRankTuple lastSuffixRank = ranks[i - 1];
270           SuffixRankTuple currSuffixRank = ranks[i];
271 
272           // If the first half differs from the second half
273           if (currSuffixRank.firstHalf != lastSuffixRank.firstHalf
274               || currSuffixRank.secondHalf != lastSuffixRank.secondHalf) newRank++;
275 
276           suffixRanks[1][currSuffixRank.originalIndex] = newRank;
277         }
278 
279         // Place top row (current row) to be the last row
280         suffixRanks[0] = suffixRanks[1];
281 
282         // Optimization to stop early
283         if (newRank == N - 1) break;
284       }
285 
286       // Fill suffix array
287       for (int i = 0; i < N; i++) {
288         sa[i] = ranks[i].originalIndex;
289         ranks[i] = null;
290       }
291 
292       // Cleanup
293       suffixRanks[0] = suffixRanks[1] = null;
294       suffixRanks = null;
295       ranks = null;
296     }

505     public SlidingWindowMinimum(int[] values) {
506       if (values == null) throw new IllegalArgumentException();
507       this.values = values;

511     // Advances the front of the window by one unit
512     public void advance() {
513       // Remove all the worse values in the back of the deque
514       while (!deque.isEmpty() && values[deque.peekLast()] > values[hi]) deque.removeLast();
515 
516       // Add the next index to the back of the deque
517       deque.addLast(hi);
518 
519       // Increase the window size
520       hi++;
521     }
522 
523     // Retracks the back of the window by one unit
524     public void shrink() {
525       // Decrease window size by pushing it forward
526       lo++;
527 
528       // Remove elements in the front of the queue whom are no longer
529       // valid in the reduced window.
530       while (!deque.isEmpty() && deque.peekFirst() < lo) deque.removeFirst();
531     }

533     // Query the current minimum value in the window
534     public int getMin() {
535       if (lo >= hi) throw new IllegalStateException("Make sure lo < hi");
536       return values[deque.peekFirst()];

540   static class CompactMinSegmentTree {
541     private int n;
542 
543     // Let UNIQUE be a value which does NOT and will NOT appear in the segment tree.
544     private int UNIQUE = 93136074;
545 
546     // Segment tree values
547     private int[] tree;
548 
549     public CompactMinSegmentTree(int size) {
550       tree = new int[2 * (n = size)];
551       java.util.Arrays.fill(tree, UNIQUE);
552     }
553 
554     public CompactMinSegmentTree(int[] values) {
555       this(values.length);
556       for (int i = 0; i < n; i++) modify(i, values[i]);
557     }
558 
559     // The segment tree function used for queries.
560     private int function(int a, int b) {
561       if (a == UNIQUE) return b;
562       else if (b == UNIQUE) return a;
563       return (a < b) ? a : b; // minimum value over a range
564     }
565 
566     // Adjust point i by a value, O(log(n))
567     public void modify(int i, int value) {
568       tree[i + n] = function(tree[i + n], value);
569       for (i += n; i > 1; i >>= 1) {
570         tree[i >> 1] = function(tree[i], tree[i ^ 1]);
571       }
572     }
573 
574     // Query interval [l, r), O(log(n))
575     public int query(int l, int r) {
576       int res = UNIQUE;
577       for (l += n, r += n; l < r; l >>= 1, r >>= 1) {
578         if ((l & 1) != 0) res = function(res, tree[l++]);
579         if ((r & 1) != 0) res = function(res, tree[--r]);
580       }
581       if (res == UNIQUE) {
582         throw new IllegalStateException("UNIQUE should not be the return value.");
583       }
584       return res;
585     }
