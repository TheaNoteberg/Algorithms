13   // Our alphabet size is 95 because there are only 95 printable ASCII characters
14   // which are in the range between [32, 127). We also need to add +1 to our alphabet
15   // size because we're going to redefine the first ASCII character (the space character)
16   // to be 1 instead of 0 to avoid collisions where the string ' ' hashes to the
17   // same value as '   ' since 0*95^0 = 0*95^0 + 0*95^1 + 0*95^2
18   private static final long ALPHABET_BASE = 95 + 1;
19   private static final long[] ALPHABET = new long[127];
20   private static final BigInteger BIG_ALPHA = new BigInteger(String.valueOf(ALPHABET_BASE));

23   // 1000037,

25   // 1000000021, 1000000033
26   private static final long[] MODS = {10_000_019, 10_000_079, 10_000_103};
27   private static final int N_HASHES = MODS.length;

34     for (int i = 32, n = 1; i < ALPHABET.length; i++, n++) ALPHABET[i] = n;

71   // Given a text and a pattern Rabin-Karp finds all occurrences
72   // of the pattern in the text in O(n+m) time.
73   public static List<Integer> rabinKarp(String text, String pattern) {
74 
75     List<Integer> matches = new ArrayList<>();
76     if (text == null || pattern == null) return matches;
77 
78     // Find pattern length (PL) and text length (TL)
79     final int PL = pattern.length(), TL = text.length();
80     if (PL > TL) return matches;
81 
82     // Compute the initial hash values
83     long[] patternHash = computeHash(pattern);
84     long[] rollingHash = computeHash(text.substring(0, PL));
85 
86     final BigInteger BIG_PL = new BigInteger(String.valueOf(PL));
87     final long[] POWERS = new long[N_HASHES];
88     for (int i = 0; i < N_HASHES; i++)
89       POWERS[i] = BIG_ALPHA.modPow(BIG_PL, BIG_MODS[i]).longValue();
90 
91     for (int i = PL - 1; ; ) {
92 
93       if (Arrays.equals(patternHash, rollingHash)) {
94         matches.add(i - PL + 1);
95       }
96       if (++i == TL) return matches;
97 
98       char firstValue = text.charAt(i - PL);
99       char lastValue = text.charAt(i);
100 
101       // Update rolling hash
102       for (int j = 0; j < patternHash.length; j++) {
103         rollingHash[j] = addRight(rollingHash[j], lastValue, j);
104         rollingHash[j] = removeLeft(rollingHash[j], POWERS[j], firstValue, j);
105       }
106     }
107   }

147   // This function adds a character to the end of the rolling hash
148   private static long addRight(long rollingHash, char lastValue, int modIndex) {
149     rollingHash = (rollingHash * ALPHABET_BASE + ALPHABET[lastValue]) % MODS[modIndex];
150     return (rollingHash + MODS[modIndex]) % MODS[modIndex];
151   }

153   // Given a rolling hash x_n*A^n + x_n-1*A^(n-1) + ... + x_2*A^2 + x_1*A^1 + x_0*A^0
154   // where x_i is a character value and 'A' is the alphabet size this method removes
155   // the x_0*A^0 term and divides by A (multiplies by modular inverse)
156   private static long removeRight(long rollingHash, char lastValue, int modIndex) {
157     rollingHash =
158         (((rollingHash - ALPHABET[lastValue]) % MODS[modIndex]) + MODS[modIndex]) % MODS[modIndex];
159     return (rollingHash * MOD_INVERSES[modIndex]) % MODS[modIndex];
160   }

162   // Given a rolling hash x_n*A^n + x_n-1*A^(n-1) + ... + x_2*A^2 + x_1*A^1 + x_0*A^0
163   // where x_i is a character value and 'A' is the alphabet size this method adds
164   // an additional term 'x_n+1*A^(n+1)' to the rolling hash.
165   //
166   // firstValue - This is x_n+1, the first character of this string
167   // alphabetBasePower - A^(n+1)
168   private static long addLeft(
169       long rollingHash, long alphabetBasePower, char firstValue, int modIndex) {
170     rollingHash = (ALPHABET[firstValue] * alphabetBasePower + rollingHash) % MODS[modIndex];
171     return (rollingHash + MODS[modIndex]) % MODS[modIndex];
172   }

174   // Given a rolling hash x_n*A^n + x_n-1*A^(n-1) + ... + x_2*A^2 + x_1*A^1 + x_0*A^0
175   // where x_i is a character value and 'A' is the alphabet size this method removes
176   // the first term 'x_n*A^n' from the rolling hash.
177   //
178   // firstValue - This is x_n, the first character of this string
179   // alphabetBasePower - A^n
180   private static long removeLeft(
181       long rollingHash, long alphabetBasePower, char firstValue, int modIndex) {
182     rollingHash = (rollingHash - ALPHABET[firstValue] * alphabetBasePower) % MODS[modIndex];
183     return (rollingHash + MODS[modIndex]) % MODS[modIndex];
184   }

186   // Computes the hashes for a particular string
187   public static long[] computeHash(String str) {
188 
189     long[] rollingHashes = new long[N_HASHES];
190     for (int k = 0; k < N_HASHES; k++)
191       for (int i = 0; i < str.length(); i++)
192         rollingHashes[k] = addRight(rollingHashes[k], str.charAt(i), k);
193     return rollingHashes;
