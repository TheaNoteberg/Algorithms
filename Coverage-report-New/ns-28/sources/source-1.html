13   public static int leastCyclicRotation(String s) {

17     int k = 0;
18     for (int j = 1; j < s.length(); j++) {
19       char sj = s.charAt(j);
20       int i = f[j - k - 1];
21       while (i != -1 && sj != s.charAt(k + i + 1)) {
22         if (sj < s.charAt(k + i + 1)) k = j - i - 1;
23         i = f[i];
24       }
25       if (sj != s.charAt(k + i + 1)) {
26         if (sj < s.charAt(k)) k = j;
27         f[j - k] = -1;
28       } else f[j - k] = i + 1;
29     }
30     return k;
31   }
