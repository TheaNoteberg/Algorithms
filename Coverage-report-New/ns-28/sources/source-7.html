9   // Manacher's algorithm finds the length of the longest palindrome
10   // centered at a specific index. Since even length palindromes have
11   // a center in between two characters we expand the string to insert
12   // those centers, for example "abba" becomes "^#a#b#b#a#$" where the
13   // '#' sign represents the center of an even length string and '^' & '$'
14   // are the front and the back of the string respectively. The output
15   // of this function gives the diameter of each palindrome centered
16   // at each character in this expanded string, for instance:
17   // manachers("abba") -> [0, 0, 1, 0, 1, 4, 1, 0, 1, 0, 0]
18   public static int[] manachers(char[] str) {
19     char[] arr = preProcess(str);
20     int n = arr.length, c = 0, r = 0;
21     int[] p = new int[n];
22     for (int i = 1; i < n - 1; i++) {
23       int invI = 2 * c - i;
24       p[i] = r > i ? Math.min(r - i, p[invI]) : 0;
25       while (arr[i + 1 + p[i]] == arr[i - 1 - p[i]]) p[i]++;
26       if (i + p[i] > r) {
27         c = i;
28         r = i + p[i];
29       }
30     }
31     return p;
32   }

35   // We do this to account for even length palindromes, so we can
36   // assign them a unique center, for example: "abba" -> "^#a#b#b#a#$"
37   private static char[] preProcess(char[] str) {
38     char[] arr = new char[str.length * 2 + 3];

43     }
44     arr[arr.length - 2] = '#';
