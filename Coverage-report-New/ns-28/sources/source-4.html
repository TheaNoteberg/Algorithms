54     private void construct() {
55       int i, p, r;
56       for (i = 0; i < N; ++i) c[rank[i] = T[i]]++;
57       for (i = 1; i < ALPHABET_SZ; ++i) c[i] += c[i - 1];
58       for (i = N - 1; i >= 0; --i) sa[--c[T[i]]] = i;
59       for (p = 1; p < N; p <<= 1) {
60         for (r = 0, i = N - p; i < N; ++i) sa2[r++] = i;
61         for (i = 0; i < N; ++i) if (sa[i] >= p) sa2[r++] = sa[i] - p;
62         Arrays.fill(c, 0, ALPHABET_SZ, 0);
63         for (i = 0; i < N; ++i) c[rank[i]]++;
64         for (i = 1; i < ALPHABET_SZ; ++i) c[i] += c[i - 1];
65         for (i = N - 1; i >= 0; --i) sa[--c[rank[sa2[i]]]] = sa2[i];
66         for (sa2[sa[0]] = r = 0, i = 1; i < N; ++i) {
67           if (!(rank[sa[i - 1]] == rank[sa[i]]
68               && sa[i - 1] + p < N
69               && sa[i] + p < N
70               && rank[sa[i - 1] + p] == rank[sa[i] + p])) r++;
71           sa2[sa[i]] = r;
72         }
73         tmp = rank;
74         rank = sa2;
75         sa2 = tmp;
76         if (r == N - 1) break;
77         ALPHABET_SZ = r + 1;
78       }

83       lcp = new int[N];
84       int[] inv = new int[N];
85       for (int i = 0; i < N; i++) inv[sa[i]] = i;
86       for (int i = 0, len = 0; i < N; i++) {

90           lcp[inv[i] - 1] = len;
91           if (len > 0) len--;
92         }
93       }
