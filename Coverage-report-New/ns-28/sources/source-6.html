58     private void construct() {
59       int i, p, r;
60       for (i = 0; i < N; ++i) c[rank[i] = T[i]]++;
61       for (i = 1; i < ALPHABET_SZ; ++i) c[i] += c[i - 1];
62       for (i = N - 1; i >= 0; --i) sa[--c[T[i]]] = i;
63       for (p = 1; p < N; p <<= 1) {
64         for (r = 0, i = N - p; i < N; ++i) sa2[r++] = i;
65         for (i = 0; i < N; ++i) if (sa[i] >= p) sa2[r++] = sa[i] - p;
66         Arrays.fill(c, 0, ALPHABET_SZ, 0);
67         for (i = 0; i < N; ++i) c[rank[i]]++;
68         for (i = 1; i < ALPHABET_SZ; ++i) c[i] += c[i - 1];
69         for (i = N - 1; i >= 0; --i) sa[--c[rank[sa2[i]]]] = sa2[i];
