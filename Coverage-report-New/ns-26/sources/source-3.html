22   // Perform a ternary search on the interval low to high.
23   // Remember that your function must be a continuous unimodal
24   // function, this means a function which decreases then increases (U shape)
25   public static double ternarySearch(double low, double high, DoubleFunction<Double> function) {
26     Double best = null;
27     while (true) {
28       double mid1 = (2 * low + high) / 3, mid2 = (low + 2 * high) / 3;
29       double res1 = function.apply(mid1), res2 = function.apply(mid2);
30       if (res1 > res2) low = mid1;
31       else high = mid2;
32       if (best != null && Math.abs(best - mid1) < EPS) break;
33       best = mid1;
34     }
35     return best;
36   }
