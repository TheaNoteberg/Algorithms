29   // partiton array in such a way that all the elements whose value is equal to
30   // pivot are grouped together
31   private static int[] partition3(int[] a, int l, int r) {
32     int j, k;
33     if (r - l <= 1) {
34       if (a[r] < a[l]) {
35         swap(a, l, r);
36       }
37       j = l;
38       k = r;
39       int[] m = {j, k};
40       return m;
41     }
42     int mid = l;
43     int p = a[r];
44     while (mid <= r) {
45       if (a[mid] < p) {
46         swap(a, l, mid);
47         l++;
48         mid++;
49       } else if (a[mid] == p) {
50         mid++;
51       } else {
52         swap(a, mid, r);
53         r--;
54       }
55     }
56     j = l - 1;
57     k = mid;
58     int[] m = {j, k};
59     return m;
60   }

62   // Sort interval [lo, hi] inplace recursively
63   // This chooses random pivot value thus improving time complexity
64   private static void randomizedQuickSort(int[] a, int l, int r) {
65     if (l >= r) {
66       return;
67     }
68     int k = random.nextInt(r - l + 1) + l;
69     int t = a[l];
70     a[l] = a[k];
71     a[k] = t;
72     // use partition3
73     int[] m = partition3(a, l, r);
74     randomizedQuickSort(a, l, m[0]);
75     randomizedQuickSort(a, m[1], r);
76   }
