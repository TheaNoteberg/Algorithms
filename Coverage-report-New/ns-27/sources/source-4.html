21   private static void heapsort(int[] ar) {
22     if (ar == null) return;
23     int n = ar.length;
24 
25     // Heapify, converts array into binary heap O(n), see:
26     // http://www.cs.umd.edu/~meesh/351/mount/lectures/lect14-heapsort-analysis-part.pdf
27     for (int i = Math.max(0, (n / 2) - 1); i >= 0; i--) {
28       sink(ar, n, i);
29     }
30 
31     // Sorting bit
32     for (int i = n - 1; i >= 0; i--) {
33       swap(ar, 0, i);
34       sink(ar, i, 0);
35     }
36   }
37 
38   private static void sink(int[] ar, int n, int i) {
39     while (true) {
40       int left = 2 * i + 1; // Left  node
41       int right = 2 * i + 2; // Right node
42       int largest = i;
43 
44       // Right child is larger than parent
45       if (right < n && ar[right] > ar[largest]) largest = right;
46 
47       // Left child is larger than parent
48       if (left < n && ar[left] > ar[largest]) largest = left;
49 
50       // Move down the tree following the largest node
51       if (largest != i) {
52         swap(ar, largest, i);
53         i = largest;
54       } else break;
55     }
56   }
57 
58   private static void swap(int[] ar, int i, int j) {
59     int tmp = ar[i];
60     ar[i] = ar[j];
61     ar[j] = tmp;
62   }
