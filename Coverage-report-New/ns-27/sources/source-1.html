19   // Sort the array using bubble sort. The idea behind
20   // bubble sort is to look for adjacent indexes which
21   // are out of place and interchange their elements
22   // until the entire array is sorted.
23   private static void bubbleSort(int[] ar) {
24     if (ar == null) {
25       return;
26     }
27 
28     boolean sorted;
29     do {
30       sorted = true;
31       for (int i = 1; i < ar.length; i++) {
32         if (ar[i] < ar[i - 1]) {
33           swap(ar, i - 1, i);
34           sorted = false;
35         }
36       }
37     } while (!sorted);
38   }
39 
40   private static void swap(int[] ar, int i, int j) {
41     int tmp = ar[i];
42     ar[i] = ar[j];
43     ar[j] = tmp;
44   }
