20     for (int i = 0; i < values.length; i++) {
21       if (values[i] < minValue) minValue = values[i];
22       if (values[i] > maxValue) maxValue = values[i];

27   // Performs a bucket sort of an array in which all the elements are
28   // bounded in the range [minValue, maxValue]. For bucket sort to give linear
29   // performance the elements need to be uniformly distributed
30   private static void bucketSort(int[] ar, int minValue, int maxValue) {
31     if (ar == null || ar.length == 0 || minValue == maxValue) return;
32 
33     // N is number elements and M is the range of values
34     final int N = ar.length, M = maxValue - minValue + 1, numBuckets = M / N + 1;
35     List<List<Integer>> buckets = new ArrayList<>(numBuckets);
36     for (int i = 0; i < numBuckets; i++) buckets.add(new ArrayList<>());
37 
38     // Place each element in a bucket
39     for (int i = 0; i < N; i++) {
40       int bi = (ar[i] - minValue) / M;
41       List<Integer> bucket = buckets.get(bi);
42       bucket.add(ar[i]);
43     }
44 
45     // Sort buckets and stitch together answer
46     for (int bi = 0, j = 0; bi < numBuckets; bi++) {
47       List<Integer> bucket = buckets.get(bi);
48       if (bucket != null) {
49         Collections.sort(bucket);
50         for (int k = 0; k < bucket.size(); k++) {
51           ar[j++] = bucket.get(k);
52         }
53       }
54     }
55   }
