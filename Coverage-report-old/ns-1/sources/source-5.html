38   static long f(int w, int ribbon) {
39     // We're trying to create a mountain scene for which we don't have enough ribbon material for.
40     if (ribbon < 0) {
41       return 0;
42     }
43     // When the width value exceeds the frame width, we know weâ€™ve finished creating a unique
