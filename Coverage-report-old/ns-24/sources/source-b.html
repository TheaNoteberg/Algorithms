27   // This function performs the extended euclidean algorithm on two numbers a and b.
28   // The function returns the gcd(a,b) as well as the numbers x and y such
29   // that ax + by = gcd(a,b). This calculation is important in number theory
30   // and can be used for several things such as finding modular inverses and
31   // solutions to linear Diophantine equations.
32   private static long[] egcd(long a, long b) {
33     if (b == 0) return new long[] {a < 0 ? -a : a, 1L, 0L};
34     long[] v = egcd(b, a % b);
35     long tmp = v[1] - v[2] * (a / b);
36     v[1] = v[2];
37     v[2] = tmp;
38     return v;
39   }

53   // Computes a^n modulo mod very efficiently in O(lg(n)) time.
54   // This function supports negative exponent values and a negative
55   // base, however the modulus must be positive.
56   public static long modPow(long a, long n, long mod) {
57 
58     if (mod <= 0) throw new ArithmeticException("mod must be > 0");
59     if (a > MAX || mod > MAX)
60       throw new IllegalArgumentException("Long overflow is upon you, mod or base is too high!");
61     if (a < MIN || mod < MIN)
62       throw new IllegalArgumentException("Long overflow is upon you, mod or base is too low!");
63 
64     // To handle negative exponents we can use the modular
65     // inverse of a to our advantage since: a^-n mod m = (a^-1)^n mod m
66     if (n < 0) {
67       if (gcd(a, mod) != 1)
68         throw new ArithmeticException("If n < 0 then must have gcd(a, mod) = 1");
69       return modPow(modInv(a, mod), -n, mod);
70     }
71 
72     if (n == 0L) return 1L;
73     long p = a, r = 1L;
74 
75     for (long i = 0; n != 0; i++) {
76       long mask = 1L << i;
77       if ((n & mask) == mask) {
78         r = (((r * p) % mod) + mod) % mod;
79         n -= mask;
80       }
81       p = ((p * p) % mod + mod) % mod;
82     }
83 
84     return ((r % mod) + mod) % mod;
85   }
