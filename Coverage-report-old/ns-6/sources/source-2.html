58   public void add(int elem) {
59     if (len + 1 >= capacity) {
60       if (capacity == 0) capacity = 1;
61       else capacity *= 2; // double the size
62       arr = java.util.Arrays.copyOf(arr, capacity); // pads with extra 0/null elements
63     }
64     arr[len++] = elem;
65   }
