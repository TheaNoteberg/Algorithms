14 public class BinarySearch {

16   // Comparing double values directly is bad practice.
17   // Using a small epsilon value is the preferred approach
18   private static final double EPS = 0.00000001;

20   public static double binarySearch(
21       double lo, double hi, double target, DoubleFunction<Double> function) {
22 
23     if (hi <= lo) throw new IllegalArgumentException("hi should be greater than lo");
24 
25     double mid;
26     do {
27 
28       // Find the middle point
29       mid = (hi + lo) / 2.0;
30 
31       // Compute the value of our function for the middle point
32       // Note that f can be any function not just the square root function
33       double value = function.apply(mid);
34 
35       if (value > target) {
36         hi = mid;
37       } else {
38         lo = mid;
39       }
40 
41     } while ((hi - lo) > EPS);
42 
43     return mid;
44   }

52     // the square root of 875 is bounded in the region: [0, 875].

56     // successively closer values of 875 we get better and better
