129   // Update a node's height and balance factor.
130   private void update(Node node) {
131     int leftNodeHeight = (node.left == null) ? -1 : node.left.height;
132     int rightNodeHeight = (node.right == null) ? -1 : node.right.height;
133

221   // Removes a value from the AVL tree.
222   private Node remove(Node node, T elem) {
223     if (node == null) return null;
224 
225     int cmp = elem.compareTo(node.value);
226 
227     // Dig into left subtree, the value we're looking
228     // for is smaller than the current value.
229     if (cmp < 0) {
230       node.left = remove(node.left, elem);
231 
232       // Dig into right subtree, the value we're looking
233       // for is greater than the current value.
234     } else if (cmp > 0) {
235       node.right = remove(node.right, elem);
236 
237       // Found the node we wish to remove.
238     } else {
239 
240       // This is the case with only a right subtree or no subtree at all.
241       // In this situation just swap the node we wish to remove
242       // with its right child.
243       if (node.left == null) {
244         return node.right;
245 
246         // This is the case with only a left subtree or
247         // no subtree at all. In this situation just
248         // swap the node we wish to remove with its left child.
249       } else if (node.right == null) {
250         return node.left;
251 
252         // When removing a node from a binary tree with two links the
253         // successor of the node being removed can either be the largest
254         // value in the left subtree or the smallest value in the right
255         // subtree. As a heuristic, I will remove from the subtree with
256         // the greatest hieght in hopes that this may help with balancing.
257       } else {
258 
259         // Choose to remove from left subtree
260         if (node.left.height > node.right.height) {
261 
262           // Swap the value of the successor into the node.
263           T successorValue = findMax(node.left);
264           node.value = successorValue;
265 
266           // Find the largest node in the left subtree.
267           node.left = remove(node.left, successorValue);
268 
269         } else {
270 
271           // Swap the value of the successor into the node.
272           T successorValue = findMin(node.right);
273           node.value = successorValue;
274 
275           // Go into the right subtree and remove the leftmost node we
276           // found and swapped data with. This prevents us from having
277           // two nodes in our tree with the same value.
278           node.right = remove(node.right, successorValue);
279         }
280       }
281     }
282 
283     // Update balance factor and height values.
284     update(node);
285 
286     // Re-balance tree.
287     return balance(node);
288   }
