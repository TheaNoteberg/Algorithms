114    * Builds a segment tree by starting with the leaf nodes and combining segment values on callback.
115    *
116    * @param i the index of the segment in the segment tree
117    * @param tl the left index (inclusive) of the segment range
118    * @param tr the right index (inclusive) of the segment range
119    * @param values the initial values array
120    */
121   private void buildSegmentTree(int i, int tl, int tr, long[] values) {
122     if (tl == tr) {
123       t[i] = values[tl];
124       return;
125     }
126     int tm = (tl + tr) / 2;
127     buildSegmentTree(2 * i + 1, tl, tm, values);
128     buildSegmentTree(2 * i + 2, tm + 1, tr, values);
129 
130     t[i] = combinationFn.apply(t[2 * i + 1], t[2 * i + 2]);

154    * Returns the range query value of the range [l, r]
155    *
156    * @param i the index of the current segment in the tree
157    * @param tl the left endpoint (inclusive) of the current segment
158    * @param tr the right endpoint (inclusive) of the current segment
159    * @param l the target left endpoint (inclusive) for the range query
160    * @param r the target right endpoint (inclusive) for the range query
161    */
162   private long rangeQuery(int i, int tl, int tr, int l, int r) {
163     if (l > r) {
164       // Different segment tree types have different base cases:
165       if (segmentCombinationFn == SegmentCombinationFn.SUM) {
166         return 0;
167       } else if (segmentCombinationFn == SegmentCombinationFn.MIN) {
168         return Long.MAX_VALUE;
169       } else if (segmentCombinationFn == SegmentCombinationFn.MAX) {
170         return Long.MIN_VALUE;
171       }
172     }
173     if (tl == l && tr == r) {
174       return t[i];
175     }
176     int tm = (tl + tr) / 2;
177     // Instead of checking if [tl, tm] overlaps [l, r] and [tm+1, tr] overlaps
178     // [l, r], simply recurse on both segments and let the base case return the
179     // default value for invalid intervals.
180     return combinationFn.apply(
181         rangeQuery(2 * i + 1, tl, tm, l, Math.min(tm, r)),
182         rangeQuery(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r));
183   }

185   /**
186    * Returns the range query value of the range [l, r]
187    *
188    * <p>An alternative implementation of the range query function that intelligently only digs into
189    * the branches of the segment tree which overlap with the query [l, r].
190    *
191    * <p>This version of the range query implementation has the advantage that it doesn't need to
192    * know the explicit base case value for each range query type.
193    *
194    * @param i the index of the current segment in the tree
195    * @param tl the left endpoint (inclusive) of the current segment
196    * @param tr the right endpoint (inclusive) of the current segment
197    * @param l the target left endpoint (inclusive) for the range query
198    * @param r the target right endpoint (inclusive) for the range query
199    */
200   private long rangeQuery2(int i, int tl, int tr, int l, int r) {
201     if (tl == l && tr == r) {
202       return t[i];
203     }
204     int tm = (tl + tr) / 2;
205     // Test how the left and right segments of the interval [tl, tr] overlap with the query [l, r]
206     boolean overlapsLeftSegment = (l <= tm);
207     boolean overlapsRightSegment = (r > tm);
208     if (overlapsLeftSegment && overlapsRightSegment) {
209       return combinationFn.apply(
210           rangeQuery2(2 * i + 1, tl, tm, l, Math.min(tm, r)),
211           rangeQuery2(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r));
212     } else if (overlapsLeftSegment) {
213       return rangeQuery2(2 * i + 1, tl, tm, l, Math.min(tm, r));
214     } else {
215       return rangeQuery2(2 * i + 2, tm + 1, tr, Math.max(l, tm + 1), r);
216     }
217   }

224   /**
225    * Update a point value to a new value and update all affected segments, O(log(n))
226    *
227    * <p>Do this by performing a binary search to find the interval containing the point, then update
228    * the leaf segment with the new value, and re-compute all affected segment values on the
229    * callback.
230    *
231    * @param i the index of the current segment in the tree
232    * @param pos the target position to update
233    * @param tl the left segment endpoint (inclusive)
234    * @param tr the right segment endpoint (inclusive)
235    * @param newValue the new value to update
236    */
237   private void pointUpdate(int i, int pos, int tl, int tr, long newValue) {
238     if (tl == tr) { // `tl == pos && tr == pos` might be clearer
239       t[i] = newValue;
240       return;
241     }
242     int tm = (tl + tr) / 2;
243     if (pos <= tm) {
244       // The point index `pos` is contained within the left segment [tl, tm]
245       pointUpdate(2 * i + 1, pos, tl, tm, newValue);
246     } else {
247       // The point index `pos` is contained within the right segment [tm+1, tr]
248       pointUpdate(2 * i + 2, pos, tm + 1, tr, newValue);
249     }
250     // Re-compute the segment value of the current segment on the callback
251     t[i] = combinationFn.apply(t[2 * i + 1], t[2 * i + 2]);
252   }
