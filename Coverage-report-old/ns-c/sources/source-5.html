2  * An implementation of an indexed min D-ary heap priority queue.
3  *
4  * <p>This implementation supports arbitrary keys with comparable values. To use arbitrary keys
5  * (such as strings or objects) first map all your keys to the integer domain [0, N) where N is the
6  * number of keys you have and then use the mapping with this indexed priority queue.
7  *
8  * <p>As convention, I denote 'ki' as the index value in the domain [0, N) associated with a key k,
9  * therefore: ki = map[k]
10  *
11  * @author William Fiset, william.alexandre.fiset@gmail.com
12  */

49   // Initializes a D-ary heap with a maximum capacity of maxSize.
50   public MinIndexedDHeap(int degree, int maxSize) {
51     if (maxSize <= 0) throw new IllegalArgumentException("maxSize <= 0");
52 
53     D = max(2, degree);
54     N = max(D + 1, maxSize);
55 
56     im = new int[N];
57     pm = new int[N];
58     child = new int[N];
59     parent = new int[N];
60     values = new Object[N];
61 
62     for (int i = 0; i < N; i++) {
63       parent[i] = (i - 1) / D;
64       child[i] = i * D + 1;
65       pm[i] = im[i] = -1;
66     }
67   }

165   private void sink(int i) {
166     for (int j = minChild(i); j != -1; ) {
167       swap(i, j);
168       i = j;
169       j = minChild(i);
170     }

173   private void swim(int i) {
174     while (less(i, parent[i])) {
175       swap(i, parent[i]);
176       i = parent[i];
177     }
178   }

180   // From the parent node at index i find the minimum child below it
181   private int minChild(int i) {
182     int index = -1, from = child[i], to = min(sz, from + D);
183     for (int j = from; j < to; j++) if (less(j, i)) index = i = j;
184     return index;
185   }

188     pm[im[j]] = i;
