63   // Generates the next ordered permutation in-place (skips repeated permutations).
64   // Calling this when the array is already at the highest permutation returns false.
65   // Recommended usage is to start with the smallest permutations and use a do while
66   // loop to generate each successive permutations (see main for example).
67   static <T extends Comparable<? super T>> boolean nextPermutation(T[] sequence) {
68     int first = getFirst(sequence);
69     if (first == -1) return false;
70     int toSwap = sequence.length - 1;
71     while (sequence[first].compareTo(sequence[toSwap]) >= 0) --toSwap;
72     swap(sequence, first++, toSwap);
73     toSwap = sequence.length - 1;
74     while (first < toSwap) swap(sequence, first++, toSwap--);
75     return true;
76   }
