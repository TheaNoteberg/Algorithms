154     private void relaxEdgesAtNode(int currentNodeIndex) {
155       visited[currentNodeIndex] = true;
156 
157       // edges will never be null if the createEmptyGraph method was used to build the graph.
158       List<Edge> edges = graph.get(currentNodeIndex);
159 
160       for (Edge edge : edges) {
161         int destNodeIndex = edge.to;
162 
163         // Skip edges pointing to already visited nodes.
164         if (visited[destNodeIndex]) continue;
165 
166         if (ipq.contains(destNodeIndex)) {
167           // Try and improve the cheapest edge at destNodeIndex with the current edge in the IPQ.
168           ipq.decrease(destNodeIndex, edge);
169         } else {
170           // Insert edge for the first time.
171           ipq.insert(destNodeIndex, edge);
172         }
173       }
174     }

176     // Computes the minimum spanning tree and minimum spanning tree cost.
177     private void solve() {
178       if (solved) return;
179       solved = true;
180 
181       int m = n - 1, edgeCount = 0;
182       visited = new boolean[n];
183       mstEdges = new Edge[m];
184 
185       // The degree of the d-ary heap supporting the IPQ can greatly impact performance, especially
186       // on dense graphs. The base 2 logarithm of n is a decent value based on my quick experiments
187       // (even better than E/V in many cases).
188       int degree = (int) Math.ceil(Math.log(n) / Math.log(2));
189       ipq = new MinIndexedDHeap<>(max(2, degree), n);
190 
191       // Add initial set of edges to the priority queue starting at node 0.
192       relaxEdgesAtNode(0);
193 
194       while (!ipq.isEmpty() && edgeCount != m) {
195         int destNodeIndex = ipq.peekMinKeyIndex(); // equivalently: edge.to
196         Edge edge = ipq.pollMinValue();
197 
198         mstEdges[edgeCount++] = edge;
199         minCostSum += edge.cost;
200 
201         relaxEdgesAtNode(destNodeIndex);
202       }
203 
204       // Verify MST spans entire graph.
205       mstExists = (edgeCount == m);
206     }

391   private static class MinIndexedDHeap<T extends Comparable<T>> {
392 
393     // Current number of elements in the heap.
394     private int sz;
395 
396     // Maximum number of elements in the heap.
397     private final int N;
398 
399     // The degree of every node in the heap.
400     private final int D;
401 
402     // Lookup arrays to track the child/parent indexes of each node.
403     private final int[] child, parent;
404 
405     // The Position Map (pm) maps Key Indexes (ki) to where the position of that
406     // key is represented in the priority queue in the domain [0, sz).
407     public final int[] pm;
408 
409     // The Inverse Map (im) stores the indexes of the keys in the range
410     // [0, sz) which make up the priority queue. It should be noted that
411     // 'im' and 'pm' are inverses of each other, so: pm[im[i]] = im[pm[i]] = i
412     public final int[] im;
413 
414     // The values associated with the keys. It is very important  to note
415     // that this array is indexed by the key indexes (aka 'ki').
416     public final Object[] values;
417 
418     // Initializes a D-ary heap with a maximum capacity of maxSize.
419     public MinIndexedDHeap(int degree, int maxSize) {
420       if (maxSize <= 0) throw new IllegalArgumentException("maxSize <= 0");
421 
422       D = max(2, degree);
423       N = max(D + 1, maxSize);
424 
425       im = new int[N];
426       pm = new int[N];
427       child = new int[N];
428       parent = new int[N];
429       values = new Object[N];
430 
431       for (int i = 0; i < N; i++) {
432         parent[i] = (i - 1) / D;
433         child[i] = i * D + 1;
434         pm[i] = im[i] = -1;
435       }
436     }
437 
438     public int size() {
439       return sz;
440     }
441 
442     public boolean isEmpty() {
443       return sz == 0;
444     }
445 
446     public boolean contains(int ki) {
447       keyInBoundsOrThrow(ki);
448       return pm[ki] != -1;
449     }

452       isNotEmptyOrThrow();

463     public T peekMinValue() {
464       isNotEmptyOrThrow();

474     public void insert(int ki, T value) {
475       if (contains(ki)) throw new IllegalArgumentException("index already exists; received: " + ki);
476       valueNotNullOrThrow(value);
477       pm[ki] = sz;
478       im[sz] = ki;

534     private void sink(int i) {
535       for (int j = minChild(i); j != -1; ) {
536         swap(i, j);
537         i = j;
538         j = minChild(i);
539       }
540     }
541 
542     private void swim(int i) {
543       while (less(i, parent[i])) {
544         swap(i, parent[i]);
545         i = parent[i];
546       }
547     }
548 
549     // From the parent node at index i find the minimum child below it
550     private int minChild(int i) {
551       int index = -1, from = child[i], to = min(sz, from + D);
552       for (int j = from; j < to; j++) if (less(j, i)) index = i = j;
553       return index;
554     }
555 
556     private void swap(int i, int j) {
557       pm[im[j]] = i;
558       pm[im[i]] = j;
559       int tmp = im[i];
560       im[i] = im[j];
561       im[j] = tmp;
562     }

568     }

573     }
