141     // Computes the minimum spanning tree and minimum spanning tree cost.
142     private void solve() {
143       if (solved) return;
144       solved = true;
145 
146       int m = n - 1, edgeCount = 0;
147       visited = new boolean[n];
148       mstEdges = new Edge[m];
149 
150       // The degree of the d-ary heap supporting the IPQ can greatly impact performance, especially
151       // on dense graphs. The base 2 logarithm of n is a decent value based on my quick experiments
152       // (even better than E/V in many cases).
153       int degree = (int) Math.ceil(Math.log(n) / Math.log(2));
154       ipq = new MinIndexedDHeap<>(max(2, degree), n);
155 
156       // Add initial set of edges to the indexed priority queue starting with node 0.
157       relaxEdgesAtNode(0);
158 
159       while (!ipq.isEmpty() && edgeCount != m) {
160         int destNodeIndex = ipq.peekMinKeyIndex(); // equivalently: edge.to
161         Edge edge = ipq.pollMinValue();
162 
163         mstEdges[edgeCount++] = edge;
164         minCostSum += edge.cost;
165 
166         relaxEdgesAtNode(destNodeIndex);
167       }
168 
169       // Verify MST spans entire graph.
170       mstExists = (edgeCount == m);
171     }
172 
173     private void relaxEdgesAtNode(int currentNodeIndex) {
174       visited[currentNodeIndex] = true;
175 
176       // edges will never be null if the createEmptyGraph method was used to build the graph.
177       List<Edge> edges = graph.get(currentNodeIndex);
178 
179       for (Edge edge : edges) {
180         int destNodeIndex = edge.to;
181 
182         // Skip edges pointing to already visited nodes.
183         if (visited[destNodeIndex]) continue;
184 
185         if (ipq.contains(destNodeIndex)) {
186           // Try and improve the cheapest edge at destNodeIndex with the current edge in the IPQ.
187           ipq.decrease(destNodeIndex, edge);
188         } else {
189           // Insert edge for the first time.
190           ipq.insert(destNodeIndex, edge);
191         }
192       }
193     }
194   }

198   private static class MinIndexedDHeap<T extends Comparable<T>> {
199 
200     // Current number of elements in the heap.
201     private int sz;
202 
203     // Maximum number of elements in the heap.
204     private final int N;
205 
206     // The degree of every node in the heap.
207     private final int D;
208 
209     // Lookup arrays to track the child/parent indexes of each node.
210     private final int[] child, parent;
211 
212     // The Position Map (pm) maps Key Indexes (ki) to where the position of that
213     // key is represented in the priority queue in the domain [0, sz).
214     public final int[] pm;
215 
216     // The Inverse Map (im) stores the indexes of the keys in the range
217     // [0, sz) which make up the priority queue. It should be noted that
218     // 'im' and 'pm' are inverses of each other, so: pm[im[i]] = im[pm[i]] = i
219     public final int[] im;
220 
221     // The values associated with the keys. It is very important  to note
222     // that this array is indexed by the key indexes (aka 'ki').
223     public final Object[] values;
224 
225     // Initializes a D-ary heap with a maximum capacity of maxSize.
226     public MinIndexedDHeap(int degree, int maxSize) {
227       if (maxSize <= 0) throw new IllegalArgumentException("maxSize <= 0");
228 
229       D = max(2, degree);
230       N = max(D + 1, maxSize);
231 
232       im = new int[N];
233       pm = new int[N];
234       child = new int[N];
235       parent = new int[N];
236       values = new Object[N];
237 
238       for (int i = 0; i < N; i++) {
239         parent[i] = (i - 1) / D;
240         child[i] = i * D + 1;
241         pm[i] = im[i] = -1;
242       }
243     }
244

246       return sz;
247     }

250       return sz == 0;
251     }

254       keyInBoundsOrThrow(ki);

271       isNotEmptyOrThrow();

273     }

281     public void insert(int ki, T value) {
282       if (contains(ki)) throw new IllegalArgumentException("index already exists; received: " + ki);
283       valueNotNullOrThrow(value);
284       pm[ki] = sz;
285       im[sz] = ki;
286       values[ki] = value;
287       swim(sz++);
288     }

294     }

296     @SuppressWarnings("unchecked")
297     public T delete(int ki) {
298       keyExistsOrThrow(ki);
299       final int i = pm[ki];
300       swap(i, --sz);
301       sink(i);
302       swim(i);
303       T value = (T) values[ki];
304       values[ki] = null;
305       pm[ki] = -1;
306       im[sz] = -1;
307       return value;
308     }

312       keyExistsAndValueNotNullOrThrow(ki, value);
313       final int i = pm[ki];
314       T oldValue = (T) values[ki];

323       keyExistsAndValueNotNullOrThrow(ki, value);
324       if (less(value, values[ki])) {

341     private void sink(int i) {
342       for (int j = minChild(i); j != -1; ) {
343         swap(i, j);
344         i = j;
345         j = minChild(i);
346       }
347     }
348 
349     private void swim(int i) {
350       while (less(i, parent[i])) {
351         swap(i, parent[i]);
352         i = parent[i];
353       }
354     }
355 
356     // From the parent node at index i find the minimum child below it
357     private int minChild(int i) {
358       int index = -1, from = child[i], to = min(sz, from + D);
359       for (int j = from; j < to; j++) if (less(j, i)) index = i = j;
360       return index;
361     }
362 
363     private void swap(int i, int j) {
364       pm[im[j]] = i;
365       pm[im[i]] = j;
366       int tmp = im[i];
367       im[i] = im[j];
368       im[j] = tmp;
369     }
