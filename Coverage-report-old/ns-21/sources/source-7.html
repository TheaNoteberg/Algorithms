56   public static boolean treesAreIsomorphic(List<List<Integer>> tree1, List<List<Integer>> tree2) {
57     if (tree1.isEmpty() || tree2.isEmpty()) {
58       throw new IllegalArgumentException("Empty tree input");
59     }
60 
61     List<Integer> centers1 = findTreeCenters(tree1);
62     List<Integer> centers2 = findTreeCenters(tree2);
63 
64     TreeNode rootedTree1 = rootTree(tree1, centers1.get(0));
65     String tree1Encoding = encode(rootedTree1);
66 
67     for (int center : centers2) {
68       TreeNode rootedTree2 = rootTree(tree2, center);
69       String tree2Encoding = encode(rootedTree2);
70 
71       if (tree1Encoding.equals(tree2Encoding)) {
72         return true;
73       }
74     }
75     return false;
76   }

78   private static List<Integer> findTreeCenters(List<List<Integer>> tree) {
79     int n = tree.size();
80 
81     int[] degree = new int[n];
82     List<Integer> leaves = new ArrayList<>();
83 
84     // Find the first outer layer of leaf nodes.
85     for (int i = 0; i < n; i++) {
86       List<Integer> edges = tree.get(i);
87       degree[i] = edges.size();
88       if (degree[i] <= 1) {
89         leaves.add(i);
90         degree[i] = 0;
91       }
92     }
93 
94     int processedLeafs = leaves.size();
95 
96     // Iteratively remove leaf nodes layer by layer until only the centers remain.
97     while (processedLeafs < n) {
98       List<Integer> newLeaves = new ArrayList<>();
99       for (int node : leaves) {
100         for (int neighbor : tree.get(node)) {
101           if (--degree[neighbor] == 1) {
102             newLeaves.add(neighbor);
103           }
104         }
105         degree[node] = 0;
106       }
107       processedLeafs += newLeaves.size();
108       leaves = newLeaves;
109     }
110 
111     return leaves;
112   }

119   // Do dfs to construct rooted tree.
120   private static TreeNode buildTree(List<List<Integer>> graph, TreeNode node) {
121     for (int neighbor : graph.get(node.id())) {
122       // Ignore adding an edge pointing back to parent.
123       if (node.parent() != null && neighbor == node.parent().id()) {
124         continue;
125       }
126 
127       TreeNode child = new TreeNode(neighbor, node);
128       node.addChildren(child);
129 
130       buildTree(graph, child);
131     }
132     return node;
133   }
