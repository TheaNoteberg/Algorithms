20   static Integer VISITED_TOKEN = 0;
21   static int[] visited = new int[MAX_NUM_NODES];

23   // Do a DFS and return a tuple (#nodes seen, index) to/of furthest node
24   private static int[] dfs(List<List<Integer>> graph, int at, int parent) {
25 
26     // Already visited this node
27     if (visited[at] == VISITED_TOKEN) return new int[] {0, parent};
28 
29     // Visit this node
30     visited[at] = VISITED_TOKEN;
31 
32     int bestDist = 0, index = -1;
33     List<Integer> edges = graph.get(at);
34 
35     if (edges != null) {
36       for (Integer to : edges) {
37         int[] tuple = dfs(graph, to, at);
38         int dist = tuple[0] + 1;
39         if (dist > bestDist) {
40           bestDist = dist;
41           index = tuple[1];
42         }
43       }
44     }
45 
46     return new int[] {bestDist, index};
47   }

49   // Given an undirected tree setup as an adjacency and a
50   // starting node list determine the diameter of this tree.
51   public static int treeDiameter(List<List<Integer>> graph, int start) {
52 
53     if (graph == null) return 0;
54 
55     // Do DFS to find furthest node from the start
56     VISITED_TOKEN++;
57     int furthestIndex = dfs(graph, start, -1)[1];
58 
59     // Singleton
60     if (furthestIndex == -1) return 0;
61 
62     // Do another DFS, but this time from the furthest
63     // node and record the distance to that node.
64     VISITED_TOKEN++;
65     int diameter = dfs(graph, furthestIndex, -1)[0] - 1;
66 
67     return diameter;
68   }
