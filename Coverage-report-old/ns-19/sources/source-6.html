19   /**
20    * @param capacity - The maximum capacity of the knapsack
21    * @param W - The weights of the items
22    * @param V - The values of the items
23    * @return The maximum achievable profit of selecting a subset of the elements such that the
24    *     capacity of the knapsack is not exceeded
25    */
26   public static int knapsack(int capacity, int[] W, int[] V) {
27 
28     if (W == null || V == null || W.length != V.length || capacity < 0)
29       throw new IllegalArgumentException("Invalid input");
30 
31     final int N = W.length;
32 
33     // Initialize a table where individual rows represent items
34     // and columns represent the weight of the knapsack
35     int[][] DP = new int[N + 1][capacity + 1];
36 
37     for (int i = 1; i <= N; i++) {
38 
39       // Get the value and weight of the item
40       int w = W[i - 1], v = V[i - 1];
41 
42       for (int sz = 1; sz <= capacity; sz++) {
43 
44         // Consider not picking this element
45         DP[i][sz] = DP[i - 1][sz];
46 
47         // Consider including the current element and
48         // see if this would be more profitable
49         if (sz >= w && DP[i - 1][sz - w] + v > DP[i][sz]) DP[i][sz] = DP[i - 1][sz - w] + v;
50       }
51     }
52 
53     int sz = capacity;
54     List<Integer> itemsSelected = new ArrayList<>();
55 
56     // Using the information inside the table we can backtrack and determine
57     // which items were selected during the dynamic programming phase. The idea
58     // is that if DP[i][sz] != DP[i-1][sz] then the item was selected
59     for (int i = N; i > 0; i--) {
60       if (DP[i][sz] != DP[i - 1][sz]) {
61         int itemIndex = i - 1;
62         itemsSelected.add(itemIndex);
63         sz -= W[itemIndex];
64       }
65     }
66 
67     // Return the items that were selected
68     // java.util.Collections.reverse(itemsSelected);
69     // return itemsSelected;
70 
71     // Return the maximum profit
72     return DP[N][capacity];
73   }
