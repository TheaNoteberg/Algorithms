43   private int f(Integer[][] dp, int i, int j) {
44     if (i == a.length && j == b.length) {
45       return 0;
46     }
47     if (i == a.length) {
48       return (b.length - j) * insertionCost;
49     }
50     if (j == b.length) {
51       return (a.length - i) * deletionCost;
52     }
53     if (dp[i][j] != null) {
54       return dp[i][j];
55     }
56     int substituteOrSkip = f(dp, i + 1, j + 1) + (a[i] == b[j] ? 0 : substitutionCost);
57     int delete = f(dp, i + 1, j) + deletionCost;
58     int insert = f(dp, i, j + 1) + insertionCost;
59     return dp[i][j] = min(substituteOrSkip, delete, insert);
60   }
