16   public static int coinChange(int[] coins, int amount) {
17 
18     if (coins == null) throw new IllegalArgumentException("Coins array is null");
19     if (coins.length == 0) throw new IllegalArgumentException("No coin values :/");
20 
21     final int N = coins.length;
22     // Initialize table and set first row to be infinity
23     int[][] dp = new int[N + 1][amount + 1];
24     java.util.Arrays.fill(dp[0], INF);
25     dp[1][0] = 0;
26 
27     // Iterate through all the coins
28     for (int i = 1; i <= N; i++) {
29 
30       int coinValue = coins[i - 1];
31       for (int j = 1; j <= amount; j++) {
32 
33         // Consider not selecting this coin
34         dp[i][j] = dp[i - 1][j];
35 
36         // Try selecting this coin if it's better
37         if (j - coinValue >= 0 && dp[i][j - coinValue] + 1 < dp[i][j]) {
38           dp[i][j] = dp[i][j - coinValue] + 1;
39         }
40       }
41     }
42 
43     // The amount we wanted to make cannot be made :/
44     if (dp[N][amount] == INF) return -1;
45 
46     // Return the minimum number of coins needed
47     return dp[N][amount];

87   private static int coinChangeRecursive(int amount, int[] coins, int[] dp) {
88 
89     // Base cases.
90     if (amount < 0) return -1;
91     if (amount == 0) return 0;
92     if (dp[amount] != 0) return dp[amount];
93 
94     int minCoins = INF;
95     for (int coinValue : coins) {
96 
97       int newAmount = amount - coinValue;
98       int value = coinChangeRecursive(newAmount, coins, dp);
99       if (value != -1 && value < minCoins) minCoins = value + 1;
100     }
101 
102     // If we weren't able to find some coins to make our
103     // amount then cache -1 as the answer.
104     return dp[amount] = (minCoins == INF) ? -1 : minCoins;
