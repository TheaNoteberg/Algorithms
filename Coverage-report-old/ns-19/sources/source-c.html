86   private double f(int state, Double[] dp, int[] history) {
87     if (dp[state] != null) {
88       return dp[state];
89     }
90     if (state == 0) {
91       return 0;
92     }
93     int p1, p2;
94     // Seek to find active bit position (p1)
95     for (p1 = 0; p1 < n; p1++) {
96       if ((state & (1 << p1)) > 0) {
97         break;
98       }
99     }
100     int bestState = -1;
101     double minimum = Double.MAX_VALUE;
102 
103     for (p2 = p1 + 1; p2 < n; p2++) {
104       // Position `p2` is on. Try matching the pair (p1, p2) together.
105       if ((state & (1 << p2)) > 0) {
106         int reducedState = state ^ (1 << p1) ^ (1 << p2);
107         double matchCost = f(reducedState, dp, history) + cost[p1][p2];
108         if (matchCost < minimum) {
109           minimum = matchCost;
110           bestState = reducedState;
111         }
112       }
113     }
114     history[state] = bestState;
115     return dp[state] = minimum;
116   }

118   public void solve() {
119     if (solved) return;
120 
121     // The DP state is encoded as a bitmask where the i'th bit is flipped on if the i'th node is
122     // included in the state. Encoding the state this way allows us to compactly represent selecting
123     // a subset of the nodes present in the matching. Furthermore, it allows using the '&' binary
124     // operator to compare states to see if they overlap and the '|' operator to combine states.
125     //
126     // dp[i] contains the optimal cost of the MWPM for the nodes captured in the binary
127     // representation of `i`. The dp table is always half empty because all states with an odd
128     // number of nodes do not have a MWPM.
129     Double[] dp = new Double[1 << n];
130 
131     // Memo table to save the history of the chosen states. This table is used to reconstruct the
132     // chosen pairs of nodes after the algorithm has executed.
133     int[] history = new int[1 << n];
134 
135     // All the states consisting of pairs of nodes are the building blocks of this algorithm.
136     // In every iteration, we try to add a pair of nodes to previous state to construct a larger
137     // matching.
138     final int numPairs = (n * (n - 1)) / 2;
139     int[] pairStates = new int[numPairs];
140     double[] pairCost = new double[numPairs];
141 
142     int k = 0;
143     for (int i = 0; i < n; i++) {
144       for (int j = i + 1; j < n; j++) {
145         int state = (1 << i) | (1 << j);
146         dp[state] = cost[i][j];
147         pairStates[k] = state;
148         pairCost[k] = cost[i][j];
149         k++;
150       }
151     }
152 
153     for (int state = 0b11; state < (1 << n); state++) { // O(2^n)
154       // Skip states with an odd number of bits (nodes). It's easier (and faster) to
155       // check dp[state] instead of calling `Integer.bitCount` for the bit count.
156       if (dp[state] == null) {
157         continue;
158       }
159       for (int i = 0; i < numPairs; i++) { // O(n^2)
160         int pair = pairStates[i];
161         // Ignore states which overlap
162         if ((state & pair) != 0) continue;
163 
164         int newState = state | pair;
165         double newCost = dp[state] + pairCost[i];
166         if (dp[newState] == null || newCost < dp[newState]) {
167           dp[newState] = newCost;
168           // Save the fact that we went from 'state' -> 'newState'. From this we will be able to
169           // reconstruct which pairs of nodes were taken by looking at 'state' xor 'newState' which
170           // should give us the binary representation (state) of the pair used.
171           history[newState] = state;
172         }
173       }
174     }
175 
176     reconstructMatching(history);
177 
178     minWeightCost = dp[END_STATE];
179     solved = true;
180   }

182   // Populates the `matching` array with a sorted deterministic matching sorted by lowest node
183   // index. For example, if the perfect matching consists of the pairs (3, 4), (1, 5), (0, 2).
184   // The matching is sorted such that the pairs appear in the ordering: (0, 2), (1, 5), (3, 4).
185   // Furthermore, it is guaranteed that for any pair (a, b) that a < b.
186   private void reconstructMatching(int[] history) {
187     // A map between pairs of nodes that were matched together.
188     int[] map = new int[n];
189     int[] leftNodes = new int[n / 2];
190 
191     // Reconstruct the matching of pairs of nodes working backwards through computed states.
192     for (int i = 0, state = END_STATE; state != 0; state = history[state]) {
193       // Isolate the pair used by xoring the state with the state used to generate it.
194       int pairUsed = state ^ history[state];
195 
196       int leftNode = getBitPosition(Integer.lowestOneBit(pairUsed));
197       int rightNode = getBitPosition(Integer.highestOneBit(pairUsed));
198 
199       leftNodes[i++] = leftNode;
200       map[leftNode] = rightNode;
201     }
202 
203     // Sort the left nodes in ascending order.
204     java.util.Arrays.sort(leftNodes);
205 
206     matching = new int[n];
207     for (int i = 0; i < n / 2; i++) {
208       matching[2 * i] = leftNodes[i];
209       int rightNode = map[leftNodes[i]];
210       matching[2 * i + 1] = rightNode;
211     }
212   }

214   // Gets the zero base index position of the 1 bit in `k`. `k` must be a power of 2, so there is
215   // only ever 1 bit in the binary representation of k.
216   private int getBitPosition(int k) {
217     int count = -1;
218     while (k > 0) {
219       count++;
220       k >>= 1;
221     }
222     return count;
223   }
