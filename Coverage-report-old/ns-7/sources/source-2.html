37     }
38

41   }

50   private void add(int i, long v) {
51     while (i < N) {
52       currentTree[i] += v;
53       i += lsb(i);
54     }
55   }

57   // Get the value at a specific index. The logic behind this method is the

59   // take the difference between the current tree and the original to get

66   private long prefixSum(int i, long[] tree) {
67     long sum = 0L;
68     while (i != 0) {
69       sum += tree[i];
