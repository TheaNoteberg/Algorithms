25   public static boolean containsPoint(Point2D[] hull, Point2D p) {
26 
27     Point2D p0 = hull[0];
28     int lo = 1, hi = hull.length - 2, mid = (lo + hi) >>> 1;
29 
30     while (lo != hi) {
31       Point2D p1 = hull[mid];
32       double sign = collinear(p0, p1, p);
33       if (sign >= 0) lo = mid;
34       else if (sign < 0) hi = mid;
35       mid = (lo + hi) >>> 1;
36       if (hi - lo == 1) {
37         if (collinear(p0, hull[hi], p) >= 0) lo = hi;
38         else hi = lo;
39       }
40     }
41 
42     Point2D p1 = hull[lo], p2 = hull[lo + 1];
43     double boundSign = collinear(p1, p2, p);
44     double segSign1 = collinear(p0, p1, p);
45     double segSign2 = collinear(p0, p2, p);
46 
47     return (boundSign >= 0 && segSign1 >= 0 && segSign2 <= 0);
48   }

50   // Suppose a != b and the points a & b form an infinite line and we want
51   // to determine if c is a point on that line. This method returns 0 if
52   // it is on the line, -1 if c is to the right of the line and +1 if it's
53   // to the left from the frame of reference of standing at point a
54   // and facing point b.
55   private static int collinear(Point2D a, Point2D b, Point2D c) {
56     double ax = a.getX(), ay = a.getY();
57     double bx = b.getX(), by = b.getY();
58     double cx = c.getX(), cy = c.getY();
59     double area = (bx - ax) * (cy - ay) - (by - ay) * (cx - ax);
60     if (abs(area) < EPS) return 0;
61     return (int) signum(area);
