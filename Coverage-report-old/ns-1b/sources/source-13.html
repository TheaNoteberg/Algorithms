18   // A very same epsilon value used as a threshold
19   public static final double EPS = 0.0000001;
20 
21   // Due to double rounding precision the value passed into the asin
22   // function may be outside its domain of [-1, +1] which would return
23   // the value Double.NaN which we do not want.
24   private static double arcsinSafe(double x) {
25     if (x <= -1.0) return -PI / 2.0;
26     if (x >= +1.0) return +PI / 2.0;
27     return asin(x);
28   }

30   // Finds the point(s) of intersection of the lines tangent to the circle centered
31   // at 'center' from the point 'point'.
32   public static Point2D[] pointCircleTangentPoints(Point2D center, double radius, Point2D pt) {
33 
34     double px = pt.getX(), py = pt.getY();
35     double cx = center.getX(), cy = center.getY();
36 
37     // Compute the distance to the circle center
38     double dx = cx - px;
39     double dy = cy - py;
40     double dist = sqrt(dx * dx + dy * dy);
41 
42     // Point is strictly contained within the circle
43     if (dist < radius) return new Point2D[] {};
44 
45     double angle, angle1, angle2;
46 
47     angle1 = arcsinSafe(radius / dist);
48     angle2 = atan2(dy, dx);
49 
50     angle = angle2 - angle1;
51     Point2D p1 = new Point2D.Double(cx + radius * sin(angle), cy + radius * -cos(angle));
52 
53     angle = angle1 + angle2;
54     Point2D p2 = new Point2D.Double(cx + radius * -sin(angle), cy + radius * cos(angle));
55 
56     // Points are sufficiently close to be considered the same point
57     // (i.e the original point is on the circle circumference)
58     if (p1.distance(p2) < EPS) return new Point2D[] {pt};
59     return new Point2D[] {p1, p2};
60   }
