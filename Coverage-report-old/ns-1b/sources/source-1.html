15   public static double angleBetweenVectors(double v1x, double v1y, double v2x, double v2y) {
16 
17     // To determine the angle between two vectors v1 and v2 we can use
18     // the following formula: dot(v1,v2) = len(v1)*len(v2)*cosθ and solve
19     // for θ where dot(a,b) is the dot product and len(c) is the length of c.
20     double dotproduct = (v1x * v2x) + (v1y * v2y);
21     double v1Length = sqrt(v1x * v1x + v1y * v1y);
22     double v2Length = sqrt(v2x * v2x + v2y * v2y);
23 
24     double value = dotproduct / (v1Length * v2Length);
25 
26     // Double value rounding precision may lead to the value we're about to pass into
27     // the arccos function to be slightly outside its domain, so do a safety check.
28     if (value <= -1.0) return PI;
29     if (value >= +1.0) return 0;
30     return acos(value);
31   }
