15 public class CollinearPoints {

17   // Epsilon is a small double value used for double comparisons
18   private static final double EPS = 1e-9;
19 
20   // Suppose a != b and the points a & b form an infinite line and we want
21   // to determine if c is a point on that line. This method returns 0 if
22   // it is on the line, -1 if c is to the right of the line and +1 if it's
23   // to the left from the frame of reference of standing at point a
24   // and facing point b.
25   public static int collinear(Point2D a, Point2D b, Point2D c) {
26 
27     if (a.distance(b) < EPS) throw new IllegalArgumentException("a cannot equal b");
28 
29     // To determine if c is on the line we will use the determinant of
30     // the two vectors formed by 'ab' and 'ac' (could also have done 'ba' and 'bc').
31     // The determinant is a particularly useful property in linear algebra because
32     // it can tell us the signed area of a parallelogram between two vectors.
33     // Furthermore, we know that two vectors are parallel if the area of the
34     // parallelogram between them is zero and we will use this fact to determine
35     // if the point c is on the line formed by a & b
36 
37     // If a = (a_x, a_y), b = (b_x, b_y), c = (c_x, c_y) then our two vectors
38     // 'ab' and 'ac' are v1 = <b_x-a_x, b_y-a_y> and v2 = <c_x-a_x, c_y-a_y>
39     double v1_x = b.getX() - a.getX();
40     double v1_y = b.getY() - a.getY();
41     double v2_x = c.getX() - a.getX();
42     double v2_y = c.getY() - a.getY();
43 
44     //                                                           | v1_x v2_x |
45     // When we place our two vectors inside a 2x2 matrix we get: | v1_y v2_y |, and
46     // hence we get that the determinant is (v1_x*v2_y)-(v2_x*v1_y)
47     double determinant = (v1_x * v2_y) - (v2_x * v1_y);
48 
49     // The points a, b, c are collinear
50     if (abs(determinant) < EPS) return 0;
51 
52     // Return -1 for right and +1 for left
53     return (determinant < 0 ? -1 : +1);
54   }
55 
56   // Compressed version of collinear code above
57   public static int collinear2(Point2D a, Point2D b, Point2D c) {
58     double area =
59         (b.getX() - a.getX()) * (c.getY() - a.getY())
60             - (b.getY() - a.getY()) * (c.getX() - a.getX());
61     if (abs(area) < EPS) return 0;
62     return (int) signum(area);
63   }
