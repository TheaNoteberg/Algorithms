17   public static double circularSegmentArea(Point2D center, double r, Point2D p1, Point2D p2) {
18 
19     double cx = center.getX(), cy = center.getY();
20     double x1 = p1.getX(), y1 = p1.getY();
21     double x2 = p2.getX(), y2 = p2.getY();
22 
23     double circleArea = PI * r * r;
24 
25     // To find the area of the circular segment cap we will find the area not occupied
26     // by the cap and compute pi*r^2 minus that area. The area not occupied by the circular
27     // cap can be divided into three parts:
28     // 1) A section pertaining to half the circle's area
29     // 2) The area of the triangle formed by (cx,cy), (x1,y1), (x2,y2)
30     // 3) The remaining slice which looks like a slice of pizza
31 
32     double halfCircleArea = circleArea / 2.0;
33     double threePointsArea = triangleArea(cx, cy, x1, y1, x2, y2);
34 
35     // Compute both vectors from the center of the circle
36     // to the points on the circumference of the circle.
37     double v1x = x1 - cx, v1y = y1 - cy;
38     double v2x = x2 - cx, v2y = y2 - cy;
39 
40     // Turn one of the vector around to point in the opposite direction
41     v1x = -v1x;
42     v1y = -v1y;
43 
44     // Compute the angle between the vectors and find the slice area
45     double angle = angleBetweenVectors(v1x, v1y, v2x, v2y);
46     double pizzaSliceArea = circleArea * (angle / (2.0 * PI));
47 
48     return circleArea - (halfCircleArea + threePointsArea + pizzaSliceArea);
49   }

51   // Given three points a, b, c find the area of the triangle
52   public static double triangleArea(
53       double ax, double ay, double bx, double by, double cx, double cy) {
54 
55     // Let v1 = <bx-ax, by-ay> and v2 = <cx-ax, cy-ay> be vectors
56     // Take the determinant of v1 and v2 to find the area of the
57     // parallelogram of the two vectors.
58 
59     double v1_x = bx - ax;
60     double v1_y = by - ay;
61     double v2_x = cx - ax;
62     double v2_y = cy - ay;
63 
64     //                | v1_x v2_x |
65     // Determinant of | v1_y v2_y | is v1_x*v2_y - v2_x*v1_y
66     double determinant = v1_x * v2_y - v2_x * v1_y;
67 
68     // Divide by two because we computed the area of the parallelogram
69     return abs(determinant) / 2.0;
70   }

72   // Return the smaller of the two angles between two 2D vectors in radians
73   private static double angleBetweenVectors(double v1x, double v1y, double v2x, double v2y) {
74 
75     // To determine the angle between two vectors v1 and v2 we can use
76     // the following formula: dot(v1,v2) = len(v1)*len(v2)*cosθ and solve
77     // for θ where dot(a,b) is the dot product and len(c) is the length of c.
78     double dotproduct = (v1x * v2x) + (v1y * v2y);
79     double v1Length = sqrt(v1x * v1x + v1y * v1y);
80     double v2Length = sqrt(v2x * v2x + v2y * v2y);
81 
82     double value = dotproduct / (v1Length * v2Length);
83 
84     // Double value rounding precision may lead to the value we're about to pass into
85     // the arccos function to be slightly outside its domain, so do a safety check.
86     if (value <= -1.0) return PI;
87     if (value >= +1.0) return 0;
88     return acos(value);
89   }
