30   // Sorts points by first x coordinate and then y coordinate.
31   private static class PointComparator implements Comparator<Point2D> {
32     public int compare(Point2D p1, Point2D p2) {
33       if (abs(p1.getX() - p2.getX()) < EPS) {
34         if (abs(p1.getY() - p2.getY()) < EPS) return 0;
35         else if (p1.getY() > p2.getY()) return 1;
36       } else if (p1.getX() > p2.getX()) return 1;
37       return -1;
38     }

41   // Use the monotone chains algorithm to find the
42   // convex hull of a set of points in O(nlogn) time.
43   public static Point2D[] convexHull(Point2D[] pts) {
44 
45     int n = pts.length, k = 0;
46     if (n <= 1) return pts;
47 
48     Point2D[] hull = new Point2D[2 * n];
49     Arrays.sort(pts, new PointComparator());
50 
51     // Build upper chain.
52     for (int i = 0; i < n; i++) {
53       while (k >= 2 && orientation(hull[k - 2], hull[k - 1], pts[i]) <= 0) k--;
54       hull[k++] = pts[i];
55     }
56 
57     int lastUpperChainIndex = k;
58 
59     // Build lower chain.
60     for (int i = n - 2; i >= 0; i--) {
61       while (k > lastUpperChainIndex && orientation(hull[k - 2], hull[k - 1], pts[i]) <= 0) k--;
62       hull[k++] = pts[i];
63     }
64 
65     // Conserve only unique points.
66     int index = 1;
67     Point2D lastPt = hull[0];
68     for (int i = 1; i < k - 1; i++) {
69       if (!hull[i].equals(lastPt)) {
70         hull[index++] = lastPt = hull[i];
71       }
72     }
73 
74     return Arrays.copyOfRange(hull, 0, index);
75   }

77   // To find orientation of point 'c' relative to the line segment (a, b).
78   // Imagine yourself standing at point 'a' looking out towards point 'b'.
79   // Returns  0 if all three points are collinear.
80   // Returns -1 if 'c' is clockwise to segment (a, b), i.e right of line formed by the segment.
81   // Returns +1 if 'c' is counter clockwise to segment (a, b), i.e left of line
82   // formed by the segment.
83   private static int orientation(Point2D a, Point2D b, Point2D c) {
84     double value =
85         (b.getY() - a.getY()) * (c.getX() - b.getX())
86             - (b.getX() - a.getX()) * (c.getY() - b.getY());
87     if (abs(value) < EPS) return 0;
88     return (value > 0) ? -1 : +1;
89   }
