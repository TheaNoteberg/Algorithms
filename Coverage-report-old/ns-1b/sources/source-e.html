19   // Given a line in the form ax + by = c (NOTE: this is not ax + by + c = 0)
20   // Given a circle with a center at (x,y) of radius r
21   // This method finds the intersection of this line and this circle
22   public static Point2D[] lineCircleIntersection(
23       double a, double b, double c, double x, double y, double r) {
24 
25     // Set ax + by = c equal to (x - X)^2 + (y - Y)^2 = r^2 and expand:
26     // (a^2 + b^2)x^2 + (2abY - 2ac + - 2b^2X)x + b^2X^2 + b^2Y^2 - 2bcY + c^2 - b^2r^2 = 0
27     // Then use quadratic formula X = (-b +- sqrt(a^2 - 4ac))/2a to find the
28     // roots of the equation (if they exist).
29 
30     // Ax^2 + Bx + C = 0;
31     double A = a * a + b * b;
32     double B = 2 * a * b * y - 2 * a * c - 2 * b * b * x;
33     double C = b * b * x * x + b * b * y * y - 2 * b * c * y + c * c - b * b * r * r;
34 
35     double D = B * B - 4 * A * C;
36     double x1, y1, x2, y2;
37 
38     // Vertical line case :0
39     if (abs(b) < EPS) {
40 
41       // ax + by = c, but b = 0, so x = c/a
42       double vx = c / a;
43 
44       // No intersection
45       if (abs(x - vx) > r) return new Point2D[] {};
46 
47       // Vertical line is tangent to circle
48       if (abs((vx - r) - x) < EPS || abs((vx + r) - x) < EPS)
49         return new Point2D[] {new Point2D.Double(vx, y)};
50 
51       double dx = abs(vx - x);
52       double dy = sqrt(r * r - dx * dx);
53 
54       return new Point2D[] {new Point2D.Double(vx, y + dy), new Point2D.Double(vx, y - dy)};
55 
56       // Line is tangent to circle
57     } else if (abs(D) < EPS) {
58 
59       x1 = -B / (2 * A);
60       y1 = (c - a * x1) / b;
61 
62       return new Point2D[] {new Point2D.Double(x1, y1)};
63 
64       // No intersection point
65     } else if (D < 0) {
66       return new Point2D[] {};
67 
68       // Two unique intersection points
69     } else {
70 
71       D = sqrt(D);
72 
73       x1 = (-B + D) / (2 * A);
74       y1 = (c - a * x1) / b;
75 
76       x2 = (-B - D) / (2 * A);
77       y2 = (c - a * x2) / b;
78 
79       return new Point2D[] {new Point2D.Double(x1, y1), new Point2D.Double(x2, y2)};
80     }
81   }
