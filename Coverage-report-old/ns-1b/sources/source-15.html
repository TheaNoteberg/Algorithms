17   // Rotate point 'pt' a certain number of radians clockwise
18   // relative to some fixed point 'fp'. Note that the angle
19   // should be specified in radians, not degrees.
20   public static Point2D rotatePoint(Point2D fp, Point2D pt, double angle) {
21 
22     double fpx = fp.getX();
23     double fpy = fp.getY();
24     double ptx = pt.getX();
25     double pty = pt.getY();
26 
27     // Compute the vector <x, y> from the fixed point
28     // to the point of rotation.
29     double x = ptx - fpx;
30     double y = pty - fpy;
31 
32     // Apply the clockwise rotation matrix to the vector <x, y>
33     // |  cosθ sinθ ||x|   |  xcosθ + ysinθ |
34     // | -sinθ cosθ ||y| = | -xsinθ + ycosθ |
35     double xRotated = x * cos(angle) + y * sin(angle);
36     double yRotated = y * cos(angle) - x * sin(angle);
37 
38     // The rotation matrix rotated the vector about the origin, so we
39     // need to offset it by the point (fpx, fpy) to get the right answer
40     return new Point2D.Double(fpx + xRotated, fpy + yRotated);
41   }
