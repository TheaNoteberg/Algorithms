19   // Determine if the point 'p' lies inside or exactly on the border of the
20   // triangle formed by the three points a,b,c. If you do want to exclude
21   // the border modify this method so that instead of checking against '<=' and
22   // '>=' it checks against strictly less than and strictly greater than.
23   public static boolean pointInsideTriangle(Point2D a, Point2D b, Point2D c, Point2D p) {
24 
25     // Points a,b,c form a degenerate triangle
26     if (collinear(a, b, c) == 0) {
27       throw new IllegalArgumentException("points a,b,c do not form a triangle!");
28     }
29 
30     // Compute the directions the point 'p' is relative to
31     // the three half planes formed by the points of the triangle
32     int dir1 = collinear(a, b, p);
33     int dir2 = collinear(b, c, p);
34     int dir3 = collinear(c, a, p);
35 
36     // Check if two of the half planes overlap in
37     // the area where the triangle is found.
38     return (dir1 <= 0 && dir2 <= 0 && dir3 <= 0) || (dir1 >= 0 && dir2 >= 0 && dir3 >= 0);
39   }

57   // Suppose a != b and the points a & b form an infinite line and we want
58   // to determine if c is a point on that line. This method returns 0 if
59   // it is on the line, -1 if c is to the right of the line and +1 if it's
60   // to the left from the frame of reference of standing at point a
61   // and facing point b.
62   private static int collinear(Point2D a, Point2D b, Point2D c) {
63     double ax = a.getX(), ay = a.getY();
64     double bx = b.getX(), by = b.getY();
65     double cx = c.getX(), cy = c.getY();
66     double area = (bx - ax) * (cy - ay) - (by - ay) * (cx - ax);
67     if (abs(area) < EPS) return 0;
68     return (int) signum(area);
69   }
