25   private static void quicksort(int[] ar, int lo, int hi) {
26     if (lo < hi) {
27       int splitPoint = partition(ar, lo, hi);
28       quicksort(ar, lo, splitPoint);
29       quicksort(ar, splitPoint + 1, hi);
30     }
31   }

33   // Performs Hoare partition algorithm for quicksort
34   private static int partition(int[] ar, int lo, int hi) {
35     int pivot = ar[lo];
36     int i = lo - 1, j = hi + 1;
37     while (true) {
38       do {
39         i++;
40       } while (ar[i] < pivot);
41       do {
42         j--;
43       } while (ar[j] > pivot);
44       if (i < j) swap(ar, i, j);
45       else return j;
46     }
47   }

51     int tmp = ar[i];
52     ar[i] = ar[j];
