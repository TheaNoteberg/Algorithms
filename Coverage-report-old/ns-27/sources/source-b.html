19   public static void selectionSort(int[] array) {
20     if (array == null) return;
21     final int N = array.length;
22 
23     for (int i = 0; i < N; i++) {
24       // Find the index beyond i with a lower value than i
25       int swapIndex = i;
26       for (int j = i + 1; j < N; j++) {
27         if (array[j] < array[swapIndex]) {
28           swapIndex = j;
29         }
30       }
31       swap(array, i, swapIndex);
32     }
33   }
