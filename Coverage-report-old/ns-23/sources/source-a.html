2  * This simplex algorithm maximizes an expression subject to a set of constraints
3  *
4  * <p>Time complexity: O(n^3)

10 public class Simplex {
11 
12   static final double EPS = 1e-9;
13 
14   // The matrix given as an argument represents the function to be maximized
15   // and each of the constraints. Constraints and objective function must be
16   // normalized first through the following steps:
17   // 1) RHS must be non-negative so multiply any inequalities failing this by -1
18   // 2) Add positive coefficient slack variable on LHS of any <= inequality
19   // 3) Add negative coefficient surplus variable on LHS of any >= inequality
20   // 4) Add positive coefficient artificial variable on LHS of any >= inequality and any = equality.
21   //
22   // If any artificial variables were added, perform simplex once, maximizing the
23   // negated sum of the artificial variables. If the maximum value is 0, take the
24   // resulting matrix and remove the artificial variable columns and replace function
25   // to maximise with original and run simplex again. If maximum value of simplex with
26   // artificial variables is non-zero there is no solution. First column of m is the constants
27   // on the RHS of all constraints. First row is the expression to maximise with all
28   // coefficients negated. M[i][j] is the coefficient of the (j-1)th term in the
29   // (i-1)th constraint (0 based).
30   public static double simplex(double[][] m) {
31     while (true) {
32       double min = -EPS;
33       int c = -1;
34       for (int j = 1; j < m[0].length; j++) {
35         if (m[0][j] < min) {
36           min = m[0][j];
37           c = j;
38         }
39       }
40       if (c < 0) break;
41       min = Double.MAX_VALUE;
42       int r = -1;
43       for (int i = 1; i < m.length; i++) {
44         if (m[i][c] > EPS) {
45           double v = m[i][0] / m[i][c];
46           if (v < min) {
47             min = v;
48             r = i;
49           }
50         }
51       }
52       double v = m[r][c];
53       for (int j = 0; j < m[r].length; j++) m[r][j] /= v;
54       for (int i = 0; i < m.length; i++) {
55         if (i != r) {
56           v = m[i][c];
57           for (int j = 0; j < m[i].length; j++) m[i][j] -= m[r][j] * v;
58         }
59       }
60     }
61     return m[0][0];
62   }
