75   // Given an n*n matrix, this method finds the determinant using Laplace/cofactor expansion.
76   // Time Complexity: ~O((n+2)!)
77   public static double determinant(double[][] matrix) {
78 
79     final int n = matrix.length;
80 
81     // Use closed form for 1x1 determinant
82     if (n == 1) return matrix[0][0];
83 
84     // Use closed form for 2x2 determinant
85     if (n == 2) return matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[1][0];
86 
87     // For 3x3 matrices and up use Laplace/cofactor expansion
88     return laplace(matrix);
89   }

91   // This method uses cofactor expansion to compute the determinant
92   // of a matrix. Unfortunately, this method is very slow and uses
93   // A LOT of memory, hence it is not too practical for large matrices.
94   private static double laplace(double[][] m) {
95 
96     final int n = m.length;
97 
98     // Base case is 3x3 determinant
99     if (n == 3) {
100       double a = m[0][0], b = m[0][1], c = m[0][2];
101       double d = m[1][0], e = m[1][1], f = m[1][2];
102       double g = m[2][0], h = m[2][1], i = m[2][2];
103       return a * (e * i - f * h) - b * (d * i - f * g) + c * (d * h - e * g);
104     }
105 
106     int det = 0;
107 
108     for (int i = 0; i < n; i++) {
109       for (int j = 0; j < n; j++) {
110 
111         double c = m[i][j];
112         if (Math.abs(c) > EPS) {
113           double[][] newMatrix = constructMatrix(m, j);
114           double parity = ((j & 1) == 0) ? +1 : -1;
115           det += parity * c * laplace(newMatrix);
116         }
117       }
118     }
119 
120     return det;
121   }
