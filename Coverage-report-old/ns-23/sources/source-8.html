14     int n = arr.length, m = arr[0].length;

16     for (int i = 0; i < m - 1 && r < n; i++) {
17       if (arr[r][i] == 0) {

19           if (arr[k][i] != 0) {

23             break;
24           }
25         }
26       }
27       if (arr[r][i] == 0) {
28         continue;
29       }
30       int inverse = inv[arr[r][i]];
31       for (int k = i; k < m; k++) arr[r][k] = (arr[r][k] * inverse) % prime;
32       for (int j = 0; j < n; j++) {
