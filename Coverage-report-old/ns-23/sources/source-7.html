36   // NOTE: Make sure the matrix is a square matrix and
37   // also watch out for overflow as the numbers climb quickly!
38   static long[][] matrixPower(long[][] matrix, long p) {
39 
40     if (p < 0) return null;
41 
42     final int N = matrix.length;
43     long[][] newMatrix = null;
44 
45     // Return identity matrix
46     if (p == 0) {
47       newMatrix = new long[N][N];
48       for (int i = 0; i < N; i++) newMatrix[i][i] = 1L;
49     } else {
50 
51       long[][] P = matrixDeepCopy(matrix);
52 
53       while (p > 0) {
54 
55         if ((p & 1L) == 1L) {
56           if (newMatrix == null) newMatrix = matrixDeepCopy(P);
57           else newMatrix = squareMatrixMult(newMatrix, P);
58         }
59 
60         // Repeatedly square P every loop, O(n^3)
61         P = squareMatrixMult(P, P);
62         p >>= 1L;
63       }
64     }
65 
66     return newMatrix;
67   }
