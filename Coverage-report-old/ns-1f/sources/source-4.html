43   private long bfs() {
44     Edge[] prev = new Edge[n];
45 
46     // The queue can be optimized to use a faster queue
47     Queue<Integer> q = new ArrayDeque<>(n);
48     visit(s);
49     q.offer(s);
50 
51     // Perform BFS from source to sink
52     while (!q.isEmpty()) {
53       int node = q.poll();
54       if (node == t) break;
55 
56       for (Edge edge : graph[node]) {
57         long cap = edge.remainingCapacity();
58         if (cap > 0 && !visited(edge.to)) {
59           visit(edge.to);
60           prev[edge.to] = edge;
61           q.offer(edge.to);
62         }
63       }
64     }
65 
66     // Sink not reachable!
67     if (prev[t] == null) return 0;
68 
69     long bottleNeck = Long.MAX_VALUE;
70 
71     // Find augmented path and bottle neck
72     for (Edge edge = prev[t]; edge != null; edge = prev[edge.from])
73       bottleNeck = min(bottleNeck, edge.remainingCapacity());
74 
75     // Retrace augmented path and update flow values.
76     for (Edge edge = prev[t]; edge != null; edge = prev[edge.from]) edge.augment(bottleNeck);
77 
78     // Return bottleneck flow
79     return bottleNeck;
80   }
