64     // Sum up the bottlenecks on each augmenting path to find the max flow and min cost.
65     List<Edge> path;
66     while ((path = getAugmentingPath()).size() != 0) {
67 
68       // Find bottle neck edge value along path.
69       long bottleNeck = Long.MAX_VALUE;
70       for (Edge edge : path) bottleNeck = min(bottleNeck, edge.remainingCapacity());

72       // Retrace path while augmenting the flow
73       for (Edge edge : path) {

82   // shortest path algorithm. First, Bellman-Ford was ran to get the shortest
83   // path from the source to every node, and then the graph was cost adjusted
84   // to remove negative edge weights so that Dijkstra's can be used in
