43   private void solve() {
44     if (n <= 1) return;
45 
46     colors = new int[n];
47     int nodesVisited = colorGraph(0, RED);
48 
49     // The graph is not bipartite. Either not all the nodes were visited or the
50     // colorGraph method returned -1 meaning the graph is not 2-colorable.
51     isBipartite = (nodesVisited == n);
52     solved = true;
53   }
54 
55   // Do a depth first search coloring the nodes of the graph as we go.
56   // This method returns the count of the number of nodes visited while
57   // coloring the graph or -1 if this graph is not bipartite.
58   private int colorGraph(int i, int color) {
59     colors[i] = color;
60 
61     // Toggles the color between RED and BLACK by exploiting the binary representation
62     // of the constants and flipping the least significant bit on and off.
63     int nextColor = (color ^ 1);
64 
65     int visitCount = 1;
66     List<Integer> edges = graph.get(i);
67 
68     for (int to : edges) {
69       // Contradiction found. In a bipartite graph no two
70       // nodes of the same color can be next to each other!
71       if (colors[to] == color) return -1;
72       if (colors[to] == nextColor) continue;
73 
74       // If a contradiction is found propagate return -1
75       // otherwise keep track of the number of visited nodes.
76       int count = colorGraph(to, nextColor);
77       if (count == -1) return -1;
78       visitCount += count;
79     }
80 
81     return visitCount;
82   }
