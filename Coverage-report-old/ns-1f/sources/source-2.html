45   // Performs the Ford-Fulkerson method applying a depth first search as
46   // a means of finding an augmenting path.
47   @Override
48   public void solve() {
49     // Start delta at the largest power of 2 <= the largest capacity.
50     // Equivalent of: delta = (long) pow(2, (int)floor(log(delta)/log(2)))
51     delta = Long.highestOneBit(delta);
52 
53     // Repeatedly find augmenting paths from source to sink using only edges
54     // with a remaining capacity >= delta. Half delta every time we become unable
55     // to find an augmenting path from source to sink until the graph is saturated.
56     for (long f = 0; delta > 0; delta /= 2) {
57       do {
58         markAllNodesAsUnvisited();
59         f = dfs(s, INF);
60         maxFlow += f;
61       } while (f != 0);
62     }
63 
64     // Find min cut.
65     for (int i = 0; i < n; i++) if (visited(i)) minCut[i] = true;
66   }

68   private long dfs(int node, long flow) {
69     // At sink node, return augmented path flow.
70     if (node == t) return flow;
71 
72     List<Edge> edges = graph[node];
73     visit(node);
74 
75     for (Edge edge : edges) {
76       long cap = edge.remainingCapacity();
77       if (cap >= delta && !visited(edge.to)) {
78 
79         long bottleNeck = dfs(edge.to, min(flow, cap));
80 
81         // Augment flow with bottle neck value
82         if (bottleNeck > 0) {
83           edge.augment(bottleNeck);
84           return bottleNeck;
85         }
86       }
87     }
88     return 0;
89   }
