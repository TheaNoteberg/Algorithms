33   public static double[] bellmanFord(Edge[] edges, int V, int start) {
34 
35     double[] dist = new double[V];
36     java.util.Arrays.fill(dist, Double.POSITIVE_INFINITY);
37     dist[start] = 0;
38 
39     // Only in the worst case does it take V-1 iterations for the Bellman-Ford
40     // algorithm to complete. Another stopping condition is when we're unable to
41     // relax an edge, this means we have reached the optimal solution early.
42     boolean relaxedAnEdge = true;
43 
44     // For each vertex, apply relaxation for all the edges
45     for (int v = 0; v < V - 1 && relaxedAnEdge; v++) {
46       relaxedAnEdge = false;
47       for (Edge edge : edges) {
48         if (dist[edge.from] + edge.cost < dist[edge.to]) {
49           dist[edge.to] = dist[edge.from] + edge.cost;
50           relaxedAnEdge = true;
51         }
52       }
53     }
54 
55     // Run algorithm a second time to detect which nodes are part
56     // of a negative cycle. A negative cycle has occurred if we
57     // can find a better path beyond the optimal solution.
58     relaxedAnEdge = true;
59     for (int v = 0; v < V - 1 && relaxedAnEdge; v++) {
60       relaxedAnEdge = false;
61       for (Edge edge : edges) {
62         if (dist[edge.from] + edge.cost < dist[edge.to]) {
63           dist[edge.to] = Double.NEGATIVE_INFINITY;
64           relaxedAnEdge = true;
65         }
66       }
67     }
68 
69     // Return the array containing the shortest distance to every node
70     return dist;
71   }
