17   /**
18    * Finds the cheapest cost to connect a given subset of nodes (which we will refer to as terminal
19    * nodes). These nodes may be either directly or indirectly connected, possibly connecting to
20    * intermediate nodes which are not terminal nodes.
21    *
22    * @param distances - The adjacency matrix for the undirected graph
23    * @param subsetToConnect - The 0-based indices of the terminal nodes
24    * @return the minimum cost required to connect the terminal nodes
25    */
26   public static double minLengthSteinerTree(double[][] distances, int[] subsetToConnect) {
27 
28     int v = distances.length;
29     int t = subsetToConnect.length;
30 
31     // Already implicitly connected
32     if (t <= 1) {
33       return 0;
34     }
35 
36     // Find the shortest distance between all pairs of nodes
37     floydWarshall(distances);
38 
39     // This array is indexed using a mask (which says which terminal nodes are
40     // connected so far) and node we are currently at (our root)
41     double[][] dp = new double[1 << t][v];
42     for (int i = 0; i < dp.length; i++) {
43       Arrays.fill(dp[i], Double.POSITIVE_INFINITY);
44     }
45 
46     // Initialize the dynamic programming array with our base cases (starting with
47     // each terminal node and going to all other nodes)
48     for (int mask = 0; mask < t; mask++) {
49       for (int j = 0; j < v; j++) {
50         dp[1 << mask][j] = distances[subsetToConnect[mask]][j];
51       }
52     }
53 
54     // Iterate over all of the sets of terminal nodes
55     for (int mask = 1; mask < 1 << t; mask++) {
56 
57       // Iterate over all of the nodes
58       for (int j = 0; j < v; j++) {
59 
60         // Effeciently iterate over all subsets of the mask
61         for (int subMask = (mask - 1) & mask; subMask > 0; subMask = (subMask - 1) & mask) {
62 
63           // Find the distance between the mask and the submask and see if we
64           // can use it to get a better answer
65           dp[mask][j] = Math.min(dp[mask][j], dp[subMask][j] + dp[mask ^ subMask][j]);
66         }
67       }
68 
69       // Try moving our roots to see if we can get a better answer
70       for (int j = 0; j < v; j++) {
71         for (int k = 0; k < v; k++) {
72           dp[mask][j] = Math.min(dp[mask][j], dp[mask][k] + distances[k][j]);
73         }
74       }
75     }
76 
77     // Return answer by looking up the mask with all of the bits set (which
78     // represents that all terminal nodes are connected)
79     return dp[(1 << t) - 1][subsetToConnect[0]];
80   }

82   /**
83    * Given an adjacency matrix with edge weights between nodes, where Double.POSITIVE_INFINITY is
84    * used to indicate that two nodes are not, connected, this method mutates the given matrix in
85    * order to give the shortest distance between all pairs of nodes. Double.NEGATIVE_INFINITY is
86    * used to indicate that the edge between node i and node j is part of a negative cycle.
87    *
88    * <p>NOTE: Usually the diagonal of the adjacency matrix is all zeros (i.e. distance[i][i] = 0 for
89    * all i) since there is typically no cost to go from a node to itself, but this may depend on
90    * your graph and the problem you are trying to solve.
91    */
92   public static void floydWarshall(double[][] distance) {
93 
94     int n = distance.length;
95 
96     // Compute all pairs shortest paths
97     for (int k = 0; k < n; k++)
98       for (int i = 0; i < n; i++)
99         for (int j = 0; j < n; j++)
100           if (distance[i][k] + distance[k][j] < distance[i][j])
101             distance[i][j] = distance[i][k] + distance[k][j];
102 
103     // Identify negative cycles (you can comment this
104     // out if you know that no negative cycles exist)
105     for (int k = 0; k < n; k++)
106       for (int i = 0; i < n; i++)
107         for (int j = 0; j < n; j++)
108           if (distance[i][k] + distance[k][j] < distance[i][j])
109             distance[i][j] = Double.NEGATIVE_INFINITY;
110   }
