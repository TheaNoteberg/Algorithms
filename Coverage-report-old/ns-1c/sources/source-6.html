220     int components;

223     public UnionFind(int n) {
224       components = n;
225       id = new int[n];
226       sz = new int[n];
227       for (int i = 0; i < n; i++) {
228         id[i] = i;
229         sz[i] = 1;
230       }
231     }
232 
233     public int find(int p) {
234       int root = p;
235       while (root != id[root]) root = id[root];
236       while (p != root) { // Do path compression
237         int next = id[p];
238         id[p] = root;
239         p = next;
240       }
241       return root;
242     }
243 
244     public boolean connected(int p, int q) {
245       return find(p) == find(q);
246     }
247 
248     public int size(int p) {
249       return sz[find(p)];
250     }

252     public void union(int p, int q) {
253       int root1 = find(p), root2 = find(q);
254       if (root1 == root2) return;
255       if (sz[root1] < sz[root2]) {
256         sz[root2] += sz[root1];
257         id[root1] = root2;
258       } else {
259         sz[root1] += sz[root2];
260         id[root2] = root1;
261       }
262       components--;
263     }
