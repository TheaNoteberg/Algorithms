40     if (solved) return;
41 
42     visited = new boolean[n];
43     components = new int[n];
44     for (int i = 0; i < n; i++) {
45       if (!visited[i]) {
46         componentCount++;
47         dfs(i);
48       }

52   }

54   private void dfs(int at) {
55     visited[at] = true;

58   }
