40   public static double computeTourCost(int[] tour, double[][] matrix) {
41 
42     double cost = 0;
43 
44     // Compute the cost of going to each city
45     for (int i = 1; i < matrix.length; i++) {

58   // Calling this when the array is already at the highest permutation returns false.
59   // Recommended usage is to start with the smallest permutations and use a do while
60   // loop to generate each successive permutations (see main for example).
61   public static boolean nextPermutation(int[] sequence) {
62     int first = getFirst(sequence);
63     if (first == -1) return false;
64     int toSwap = sequence.length - 1;
65     while (sequence[first] >= sequence[toSwap]) --toSwap;
66     swap(sequence, first++, toSwap);
67     toSwap = sequence.length - 1;
68     while (first < toSwap) swap(sequence, first++, toSwap--);
69     return true;
70   }
