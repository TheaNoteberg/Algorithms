82       low[u] = pre++;
83       int min = low[u];
84       stack.push(u);
85       for (int v = 0; v < n; v++) {
86         if (adj[u][v]) {

88           if (low[v] < min) min = low[v];
89         }
90       }
91       if (min < low[u]) {
92         low[u] = min;

94       }

96       do {

100       } while (v != u);

102     }
