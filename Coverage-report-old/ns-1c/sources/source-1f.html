62   private void addEdges(int nodeIndex) {
63     visited[nodeIndex] = true;
64 
65     // edges will never be null if the createEmptyGraph method was used to build the graph.
66     List<Edge> edges = graph.get(nodeIndex);
67     for (Edge e : edges)
68       if (!visited[e.to]) {
69         // System.out.printf("(%d, %d, %d)\n", e.from, e.to, e.cost);
70         pq.offer(e);
71       }
72   }

74   // Computes the minimum spanning tree and minimum spanning tree cost.
75   private void solve() {
76     if (solved) return;
77     solved = true;
78 
79     int m = n - 1, edgeCount = 0;
80     pq = new PriorityQueue<>();
81     visited = new boolean[n];
82     mstEdges = new Edge[m];
83 
84     // Add initial set of edges to the priority queue starting at node 0.
85     addEdges(0);
86 
87     // Loop while the MST is not complete.
88     while (!pq.isEmpty() && edgeCount != m) {
89       Edge edge = pq.poll();
90       int nodeIndex = edge.to;
91 
92       // Skip any edge pointing to an already visited node.
93       if (visited[nodeIndex]) continue;
94 
95       mstEdges[edgeCount++] = edge;
96       minCostSum += edge.cost;
97 
98       addEdges(nodeIndex);
99     }
100 
101     // Check if MST spans entire graph.
102     mstExists = (edgeCount == m);
103   }
