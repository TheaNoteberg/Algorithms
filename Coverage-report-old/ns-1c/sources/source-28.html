1 /**
2  * This file contains a recursive implementation of the TSP problem using dynamic programming. The
3  * main idea is that since we need to do all n! permutations of nodes to find the optimal solution
4  * that caching the results of sub paths can improve performance.
5  *
6  * <p>For example, if one permutation is: '... D A B C' then later when we need to compute the value
7  * of the permutation '... E B A C' we should already have cached the answer for the subgraph
8  * containing the nodes {A, B, C}.
9  *
10  * <p>Time Complexity: O(n^2 * 2^n) Space Complexity: O(n * 2^n)
11  *
12  * @author Steven & Felix Halim, William Fiset, Micah Stairs
13  */

34   public TspDynamicProgrammingRecursive(int startNode, double[][] distance) {
35 
36     this.distance = distance;
37     N = distance.length;
38     START_NODE = startNode;
39 
40     // Validate inputs.
41     if (N <= 2) throw new IllegalStateException("TSP on 0, 1 or 2 nodes doesn't make sense.");
42     if (N != distance[0].length)
43       throw new IllegalArgumentException("Matrix must be square (N x N)");
44     if (START_NODE < 0 || START_NODE >= N)

68   public void solve() {
69 
70     // Run the solver
71     int state = 1 << START_NODE;
72     Double[][] memo = new Double[N][1 << N];
73     Integer[][] prev = new Integer[N][1 << N];
74     minTourCost = tsp(START_NODE, state, memo, prev);
75 
76     // Regenerate path
77     int index = START_NODE;
78     while (true) {
79       tour.add(index);
80       Integer nextIndex = prev[index][state];
81       if (nextIndex == null) break;
82       int nextState = state | (1 << nextIndex);
83       state = nextState;
84       index = nextIndex;
85     }
86     tour.add(START_NODE);
87     ranSolver = true;
88   }

90   private double tsp(int i, int state, Double[][] memo, Integer[][] prev) {
91 
92     // Done this tour. Return cost of going back to start node.
93     if (state == FINISHED_STATE) return distance[i][START_NODE];
94 
95     // Return cached answer if already computed.
96     if (memo[i][state] != null) return memo[i][state];
97 
98     double minCost = Double.POSITIVE_INFINITY;
99     int index = -1;
100     for (int next = 0; next < N; next++) {
101 
102       // Skip if the next node has already been visited.
103       if ((state & (1 << next)) != 0) continue;
104 
105       int nextState = state | (1 << next);
106       double newCost = distance[i][next] + tsp(next, nextState, memo, prev);
107       if (newCost < minCost) {
108         minCost = newCost;
109         index = next;
110       }
111     }
112 
113     prev[i][state] = index;
114     return memo[i][state] = minCost;
115   }
