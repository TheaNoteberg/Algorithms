77   // Run Dijkstra's algorithm on a directed graph to find the shortest path
78   // from a starting node to an ending node. If there is no path between the
79   // starting node and the destination node the returned value is set to be
80   // Double.POSITIVE_INFINITY.
81   public double dijkstra(int start, int end) {
82 
83     // Keep an Indexed Priority Queue (ipq) of the next most promising node
84     // to visit.
85     int degree = edgeCount / n;
86     MinIndexedDHeap<Double> ipq = new MinIndexedDHeap<>(degree, n);
87     ipq.insert(start, 0.0);
88 
89     // Maintain an array of the minimum distance to each node.
90     dist = new double[n];
91     Arrays.fill(dist, Double.POSITIVE_INFINITY);
92     dist[start] = 0.0;
93 
94     boolean[] visited = new boolean[n];
95     prev = new Integer[n];
96 
97     while (!ipq.isEmpty()) {
98       int nodeId = ipq.peekMinKeyIndex();
99 
100       visited[nodeId] = true;
101       double minValue = ipq.pollMinValue();
102 
103       // We already found a better path before we got to
104       // processing this node so we can ignore it.
105       if (minValue > dist[nodeId]) continue;
106 
107       for (Edge edge : graph.get(nodeId)) {
108 
109         // We cannot get a shorter path by revisiting
110         // a node we have already visited before.
111         if (visited[edge.to]) continue;
112 
113         // Relax edge by updating minimum cost if applicable.
114         double newDist = dist[nodeId] + edge.cost;
115         if (newDist < dist[edge.to]) {
116           prev[edge.to] = nodeId;
117           dist[edge.to] = newDist;
118           // Insert the cost of going to a node for the first time in the PQ,
119           // or try and update it to a better value by calling decrease.
120           if (!ipq.contains(edge.to)) ipq.insert(edge.to, newDist);
121           else ipq.decrease(edge.to, newDist);
122         }
123       }
124       // Once we've processed the end node we can return early (without
125       // necessarily visiting the whole graph) because we know we cannot get a
126       // shorter path by routing through any other nodes since Dijkstra's is
127       // greedy and there are no negative edge weights.
128       if (nodeId == end) return dist[end];
129     }
130     // End node is unreachable.
131     return Double.POSITIVE_INFINITY;
132   }

138    *     'end' are not connected then an empty array is returned.

140   public List<Integer> reconstructPath(int start, int end) {
141     if (end < 0 || end >= n) throw new IllegalArgumentException("Invalid node index");
142     if (start < 0 || start >= n) throw new IllegalArgumentException("Invalid node index");
143     List<Integer> path = new ArrayList<>();
144     double dist = dijkstra(start, end);
145     if (dist == Double.POSITIVE_INFINITY) return path;
146     for (Integer at = end; at != null; at = prev[at]) path.add(at);
147     Collections.reverse(path);
148     return path;
149   }

311       int index = -1, from = child[i], to = min(sz, from + D);
