47    */
48   public static double[] bellmanFord(List<Edge>[] graph, int V, int start) {
49 
50     // Initialize the distance to all nodes to be infinity
51     // except for the start node which is zero.
52     double[] dist = new double[V];
53     java.util.Arrays.fill(dist, Double.POSITIVE_INFINITY);
54     dist[start] = 0;
55 
56     // For each vertex, apply relaxation for all the edges
57     for (int i = 0; i < V - 1; i++)
58       for (List<Edge> edges : graph)
59         for (Edge edge : edges)
60           if (dist[edge.from] + edge.cost < dist[edge.to])
61             dist[edge.to] = dist[edge.from] + edge.cost;
62 
63     // Run algorithm a second time to detect which nodes are part
64     // of a negative cycle. A negative cycle has occurred if we
65     // can find a better path beyond the optimal solution.
66     for (int i = 0; i < V - 1; i++)
67       for (List<Edge> edges : graph)
68         for (Edge edge : edges)
69           if (dist[edge.from] + edge.cost < dist[edge.to]) dist[edge.to] = Double.NEGATIVE_INFINITY;
70 
71     // Return the array containing the shortest distance to every node
72     return dist;
73   }
