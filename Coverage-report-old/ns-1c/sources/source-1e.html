65   private void kruskals() {
66     if (solved) return;
67 
68     // Heapify operation in constructor transforms list of edges into a binary heap in O(n)
69     PriorityQueue<Edge> pq = new PriorityQueue<>(edges);
70     UnionFind uf = new UnionFind(n);
71 
72     int index = 0;
73     mst = new Edge[n - 1];
74 
75     while (!pq.isEmpty()) {
76       // Use heap to poll the next cheapest edge. Polling avoids the need to sort
77       // the edges before loop in the event that the algorithm terminates early.
78       Edge edge = pq.poll();
79 
80       // Skip this edge to avoid creating a cycle in MST.
81       if (uf.connected(edge.u, edge.v)) continue;
82 
83       // Include this edge
84       uf.union(edge.u, edge.v);
85       mstCost += edge.cost;
86       mst[index++] = edge;
87 
88       // Stop early if we found a MST that includes all the nodes.
89       if (uf.size(0) == n) break;
90     }
91 
92     mstExists = (uf.size(0) == n);
93     solved = true;
94   }

97   private static class UnionFind {
98     private int[] id, sz;
99 
100     public UnionFind(int n) {
101       id = new int[n];
102       sz = new int[n];
103       for (int i = 0; i < n; i++) {
104         id[i] = i;
105         sz[i] = 1;
106       }
107     }
108 
109     public int find(int p) {
110       int root = p;
111       while (root != id[root]) root = id[root];
112       // Path compression
113       while (p != root) {
114         int next = id[p];
115         id[p] = root;
116         p = next;
117       }
118       return root;
119     }
120 
121     public boolean connected(int p, int q) {
122       return find(p) == find(q);
123     }
124 
125     public int size(int p) {
126       return sz[find(p)];
127     }
128 
129     public void union(int p, int q) {
130       int root1 = find(p);
131       int root2 = find(q);
132       if (root1 == root2) return;
133       if (sz[root1] < sz[root2]) {
134         sz[root2] += sz[root1];
135         id[root1] = root2;
136       } else {
137         sz[root1] += sz[root2];
138         id[root2] = root1;
139       }
140     }
141   }
