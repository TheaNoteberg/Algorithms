9 public class UnionFind {
10 
11   // The number of elements in this union find
12   private int size;
13 
14   // Used to track the size of each of the component
15   private int[] sz;
16 
17   // id[i] points to the parent of i, if id[i] = i then i is a root node
18   private int[] id;
19 
20   // Tracks the number of components in the union find
21   private int numComponents;
22 
23   public UnionFind(int size) {
24 
25     if (size <= 0) throw new IllegalArgumentException("Size <= 0 is not allowed");
26 
27     this.size = numComponents = size;
28     sz = new int[size];
29     id = new int[size];
30 
31     for (int i = 0; i < size; i++) {
32       id[i] = i; // Link to itself (self root)
33       sz[i] = 1; // Each component is originally of size one
34     }
35   }
36 
37   // Find which component/set 'p' belongs to, takes amortized constant time.
38   public int find(int p) {
39 
40     // Find the root of the component/set
41     int root = p;
42     while (root != id[root]) root = id[root];
43 
44     // Compress the path leading back to the root.
45     // Doing this operation is called "path compression"
46     // and is what gives us amortized time complexity.
47     while (p != root) {
48       int next = id[p];
49       id[p] = root;
50       p = next;
51     }
52 
53     return root;
54   }

83   // Unify the components/sets containing elements 'p' and 'q'
84   public void unify(int p, int q) {
85 
86     // These elements are already in the same group!
87     if (connected(p, q)) return;
88 
89     int root1 = find(p);
90     int root2 = find(q);
91 
92     // Merge smaller component/set into the larger one.
93     if (sz[root1] < sz[root2]) {
94       sz[root2] += sz[root1];
95       id[root1] = root2;
96       sz[root1] = 0;
97     } else {
98       sz[root1] += sz[root2];
99       id[root2] = root1;
100       sz[root2] = 0;
101     }
102 
103     // Since the roots found are different we know that the
104     // number of components/sets has decreased by one
105     numComponents--;
106   }
