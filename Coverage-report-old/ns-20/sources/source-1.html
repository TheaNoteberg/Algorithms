170   private static class CapacityScalingSolver extends NetworkFlowSolverBase {
171 
172     private long delta;
173 
174     /**
175      * Creates an instance of a flow network solver. Use the {@link #addEdge(int, int, int)} method
176      * to add edges to the graph.
177      *
178      * @param n - The number of nodes in the graph including source and sink nodes.
179      * @param s - The index of the source node, 0 <= s < n
180      * @param t - The index of the sink node, 0 <= t < n, t != s
181      */
182     public CapacityScalingSolver(int n, int s, int t) {
183       super(n, s, t);
184     }
185 
186     /**
187      * Adds a directed edge (and residual edge) to the flow graph.
188      *
189      * @param from - The index of the node the directed edge starts at.
190      * @param to - The index of the node the directed edge end at.
191      * @param capacity - The capacity of the edge.
192      */
193     @Override
194     public void addEdge(int from, int to, long capacity) {
195       super.addEdge(from, to, capacity);
196       delta = max(delta, capacity);
197     }
198 
199     // Performs the Ford-Fulkerson method applying a depth first search as
200     // a means of finding an augmenting path.
201     @Override
202     public void solve() {
203       // Start delta at the largest power of 2 <= the largest capacity.
204       // Equivalent of: delta = (long) pow(2, (int)floor(log(delta)/log(2)))
205       delta = Long.highestOneBit(delta);
206 
207       // Repeatedly find augmenting paths from source to sink using only edges
208       // with a remaining capacity >= delta. Half delta every time we become unable
209       // to find an augmenting path from source to sink until the graph is saturated.
210       for (long f = 0; delta > 0; delta /= 2) {
211         do {
212           markAllNodesAsUnvisited();
213           f = dfs(s, INF);
214           maxFlow += f;
215         } while (f != 0);
216       }
217     }
218 
219     private long dfs(int node, long flow) {
220       // At sink node, return augmented path flow.
221       if (node == t) return flow;
222       visit(node);
223 
224       for (Edge edge : graph[node]) {
225         long cap = edge.remainingCapacity();
226         if (cap >= delta && !visited(edge.to)) {
227 
228           long bottleNeck = dfs(edge.to, min(flow, cap));
229 
230           // Augment flow with bottle neck value
231           if (bottleNeck > 0) {
232             edge.augment(bottleNeck);
233             return bottleNeck;
234           }
235         }
236       }
237       return 0;
238     }
239   }
