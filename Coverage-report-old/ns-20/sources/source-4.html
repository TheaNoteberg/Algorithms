62     // To avoid overflow, set infinity to a value less than Long.MAX_VALUE;
63     static final long INF = Long.MAX_VALUE / 2;
64 
65     // Inputs: n = number of nodes, s = source, t = sink

68     // 'visited' and 'visitedToken' are variables used in graph sub-routines to
69     // track whether a node has been visited or not. In particular, node 'i' was
70     // recently visited if visited[i] == visitedToken is true. This is handy
71     // because to mark all nodes as unvisited simply increment the visitedToken.
72     protected int visitedToken = 1;
73     protected int[] visited;

89      * @param n - The number of nodes in the graph including s and t.
90      * @param s - The index of the source node, 0 <= s < n
91      * @param t - The index of the sink node, 0 <= t < n and t != s

102     @SuppressWarnings("unchecked")

108     /**
109      * Adds a directed edge (and its residual edge) to the flow graph.
110      *
111      * @param from - The index of the node the directed edge starts at.
112      * @param to - The index of the node the directed edge ends at.
113      * @param capacity - The capacity of the edge
114      */
115     public void addEdge(int from, int to, long capacity) {
116       if (capacity <= 0) throw new IllegalArgumentException("Forward edge capacity <= 0");
117       Edge e1 = new Edge(from, to, capacity);
118       Edge e2 = new Edge(to, from, 0);
119       e1.residual = e2;
120       e2.residual = e1;
121       graph[from].add(e1);
122       graph[to].add(e2);
123     }

152   private static class FordFulkersonDfsSolver extends NetworkFlowSolverBase {
153 
154     /**
155      * Creates an instance of a flow network solver. Use the {@link #addEdge} method to add edges to
156      * the graph.
157      *
158      * @param n - The number of nodes in the graph including s and t.
159      * @param s - The index of the source node, 0 <= s < n
160      * @param t - The index of the sink node, 0 <= t < n and t != s
161      */
162     public FordFulkersonDfsSolver(int n, int s, int t) {
163       super(n, s, t);
164     }
165 
166     // Performs the Ford-Fulkerson method applying a depth first search as
167     // a means of finding an augmenting path.
168     @Override
169     public void solve() {
170       // Find max flow by adding all augmenting path flows.
171       for (long f = dfs(s, INF); f != 0; f = dfs(s, INF)) {
172         visitedToken++;
173         maxFlow += f;
174       }
175     }
176 
177     private long dfs(int node, long flow) {
178       // At sink node, return augmented path flow.
179       if (node == t) return flow;
180 
181       // Mark the current node as visited.
182       visited[node] = visitedToken;
183 
184       List<Edge> edges = graph[node];
185       for (Edge edge : edges) {
186         if (edge.remainingCapacity() > 0 && visited[edge.to] != visitedToken) {
187           long bottleNeck = dfs(edge.to, min(flow, edge.remainingCapacity()));
188 
189           // If we made it from s -> t (a.k.a bottleNeck > 0) then
190           // augment flow with bottleneck value.
191           if (bottleNeck > 0) {
192             edge.augment(bottleNeck);
193             return bottleNeck;
194           }
195         }
196       }
197       return 0;
198     }
199   }
