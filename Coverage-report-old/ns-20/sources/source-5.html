52   static void solve(Mouse[] mice, Hole[] holes, int radius) {
53     final int M = mice.length;
54     final int H = holes.length;
55 
56     // N = number of nodes, S = source node, T = sink node
57     final int N = M + H + 2;
58     final int S = N - 1;
59     final int T = N - 2;
60 
61     NetworkFlowSolverBase solver;
62     solver = new FordFulkersonDfsSolver(N, S, T);
63 
64     // Source to mice
65     for (int i = 0; i < M; i++) {
66       solver.addEdge(S, i, 1);
67     }
68 
69     // Hook up each mouse with the holes they are able to reach
70     for (int i = 0; i < M; i++) {
71       Point2D mouse = mice[i].point;
72       for (int j = 0; j < H; j++) {
73         Point2D hole = holes[j].point;
74         if (mouse.distance(hole) <= radius) {
75           solver.addEdge(i, M + j, 1);
76         }
77       }
78     }
79 
80     // Holes to sink
81     for (int i = 0; i < H; i++) {
82       solver.addEdge(M + i, T, holes[i].capacity);
83     }
84 
85     // Output: "Number of safe mice: 4"
86     System.out.println("Number of safe mice: " + solver.getMaxFlow());

217   private static class FordFulkersonDfsSolver extends NetworkFlowSolverBase {
218 
219     /**
220      * Creates an instance of a flow network solver. Use the {@link #addEdge} method to add edges to
221      * the graph.
222      *
223      * @param n - The number of nodes in the graph including s and t.
224      * @param s - The index of the source node, 0 <= s < n
225      * @param t - The index of the sink node, 0 <= t < n and t != s
226      */
227     public FordFulkersonDfsSolver(int n, int s, int t) {
228       super(n, s, t);
229     }
230 
231     // Performs the Ford-Fulkerson method applying a depth first search as
232     // a means of finding an augmenting path.
233     @Override
234     public void solve() {
235       // Find max flow by adding all augmenting path flows.
236       for (long f = dfs(s, INF); f != 0; f = dfs(s, INF)) {
237         visitedToken++;
238         maxFlow += f;
239       }
240     }
241 
242     private long dfs(int node, long flow) {
243       // At sink node, return augmented path flow.
244       if (node == t) return flow;
245 
246       // Mark the current node as visited.
247       visited[node] = visitedToken;
248 
249       List<Edge> edges = graph[node];
250       for (Edge edge : edges) {
251         if (edge.remainingCapacity() > 0 && visited[edge.to] != visitedToken) {
252           long bottleNeck = dfs(edge.to, min(flow, edge.remainingCapacity()));
253 
254           // If we made it from s -> t (a.k.a bottleNeck > 0) then
255           // augment flow with bottleneck value.
256           if (bottleNeck > 0) {
257             edge.augment(bottleNeck);
258             return bottleNeck;
259           }
260         }
261       }
262       return 0;
263     }
