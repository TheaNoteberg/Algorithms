147   // Inserts an entry in a given bucket only if the entry does not already
148   // exist in the given bucket, but if it does then update the entry value
149   private V bucketInsertEntry(int bucketIndex, Entry<K, V> entry) {
150 
151     LinkedList<Entry<K, V>> bucket = table[bucketIndex];
152     if (bucket == null) table[bucketIndex] = bucket = new LinkedList<>();
153

155     if (existentEntry == null) {
156       bucket.add(entry);
157       if (++size > threshold) resizeTable();
158       return null; // Use null to indicate that there was no previous entry
159     } else {

161       existentEntry.value = entry.value;
162       return oldVal;
163     }
164   }

166   // Finds and returns a particular entry in a given bucket if it exists, returns null otherwise
167   private Entry<K, V> bucketSeekEntry(int bucketIndex, K key) {

169     if (key == null) return null;
170     LinkedList<Entry<K, V>> bucket = table[bucketIndex];
171     if (bucket == null) return null;
172     for (Entry<K, V> entry : bucket) if (entry.key.equals(key)) return entry;
173     return null;
174   }

176   // Resizes the internal table holding buckets of entries
177   private void resizeTable() {
178 
179     capacity *= 2;
180     threshold = (int) (capacity * maxLoadFactor);
181 
182     LinkedList<Entry<K, V>>[] newTable = new LinkedList[capacity];
183 
184     for (int i = 0; i < table.length; i++) {
185       if (table[i] != null) {
186

188           int bucketIndex = normalizeIndex(entry.hash);
189           LinkedList<Entry<K, V>> bucket = newTable[bucketIndex];
190           if (bucket == null) newTable[bucketIndex] = bucket = new LinkedList<>();
191           bucket.add(entry);
192         }
193 
194         // Avoid memory leak. Help the GC
195         table[i].clear();
196         table[i] = null;
197       }
198     }

201   }
