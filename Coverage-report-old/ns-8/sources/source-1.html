197     // 2  if z &ne; NIL
198     if (isEmpty()) {
199       return null;

203     FibonacciHeapNode<E> z = minimumNode;

205

258   private void consolidate() {
259     if (isEmpty()) {
260       return;
261     }
262 
263     // D( n[H] ) <= log_phi( n[H] )
264     // -> log_phi( n[H] ) = log( n[H] ) / log( phi )
265     // -> D( n[H] ) = log( n[H] ) / log( phi )
266     int arraySize = ((int) floor(log(size) / LOG_PHI));
267 
268     // 1  for i <- 0 to D(n[H])
269     List<FibonacciHeapNode<E>> nodeSequence = new ArrayList<FibonacciHeapNode<E>>(arraySize);
270     for (int i = 0; i < arraySize; i++) {
271       // 2      do A[i] <- NIL
272       nodeSequence.add(i, null);
273     }
274 
275     int numRoots = 0;
276 
277     // 3  for each node x in the root list of H
278     // 4  do x &larr; w
279     FibonacciHeapNode<E> x = minimumNode;
280 
281     if (x != null) {
282       numRoots++;
283       x = x.getRight();
284 
285       while (x != minimumNode) {
286         numRoots++;
287         x = x.getRight();
288       }
289     }
290 
291     while (numRoots > 0) {
292       // 5  d <- degree[x]
293       int degree = x.getDegree();
294       FibonacciHeapNode<E> next = x.getRight();
295 
296       // 6  while A[d] != NIL
297       while (nodeSequence.get(degree) != null) {
298         // 7  do y <- A[d]
299         FibonacciHeapNode<E> y = nodeSequence.get(degree);
300 
301         // 8  if key[x] > key[y]
302         if (compare(x, y) > 0) {
303           // 9  exchange x <-> y
304           FibonacciHeapNode<E> pointer = y;
305           y = x;
306           x = pointer;
307         }
308 
309         // 10  FIB-HEAP-LINK(H,y,x)
310         link(y, x);
311 
312         // 11  A[d] <- NIL
313         nodeSequence.set(degree, null);
314 
315         // 12  d <- d + 1
316         degree++;
317       }
318 
319       // 13  A[d] <- x
320       nodeSequence.set(degree, x);
321 
322       x = next;
323       numRoots--;
324     }
325 
326     // 14  min[H] <- NIL
327     minimumNode = null;
328 
329     // 15  for i <- 0 to D(n[H])
330     for (FibonacciHeapNode<E> pointer : nodeSequence) {
331       if (pointer == null) {
332         continue;
333       }
334       if (minimumNode == null) {
335         minimumNode = pointer;
336       }
337 
338       // 16 if A[i] != NIL
339       // We've got a live one, add it to root list.
340       if (minimumNode != null) {
341         //  First remove node from root list.
342         moveToRoot(pointer);
343       }
344     }
345   }

347   private void link(FibonacciHeapNode<E> y, FibonacciHeapNode<E> x) {
348     // 1 remove y from the root list of H
349     y.getLeft().setRight(y.getRight());
350     y.getRight().setLeft(y.getLeft());
351 
352     y.setParent(x);
353 
354     if (x.getChild() == null) {
355       // 2 make y a child of x, incrementing degree[x]
356       x.setChild(y);
357       y.setRight(y);
358       y.setLeft(y);
359     } else {
360       y.setLeft(x.getChild());
361       y.setRight(x.getChild().getRight());
362       x.getChild().setRight(y);
363       y.getRight().setLeft(y);
364     }
365 
366     x.incraeseDegree();
367 
368     // 3 mark[y] <- FALSE
369     y.setMarked(false);
370     markedNodes++;
371   }

388     // z <- p[y]
389     FibonacciHeapNode<E> z = y.getParent();

392     if (z != null) {
